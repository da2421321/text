const fs = require('fs');
const path = require('path');

class FileManager {
    constructor() {
        //当前目录
        this.basePath = process.cwd();
    }
    
    createFile(filePath, content) {
        try {
            //将路径解析为绝对路径
            const fullPath = path.resolve(this.basePath, filePath);
            //创建文件
            fs.writeFileSync(fullPath, content, 'utf8');
            console.log(`文件创建成功: ${fullPath}`);
            return true;
        } catch (error) {
            console.error('创建文件失败:', error.message);
            return false;
        }
    }
    
    readFile(filePath) {
        try {
            const fullPath = path.resolve(this.basePath, filePath);
            const content = fs.readFileSync(fullPath, 'utf8');
            console.log(`文件内容: ${content}`);
            return content;
        } catch (error) {
            console.error('读取文件失败:', error.message);
            return null;
        }
    }
    
    copyFile(sourcePath, targetPath) {
        try {
            const sourceFullPath = path.resolve(this.basePath, sourcePath);
            const targetFullPath = path.resolve(this.basePath, targetPath);
            
            if (!fs.existsSync(sourceFullPath)) {
                throw new Error('源文件不存在');
            }
            
            fs.copyFileSync(sourceFullPath, targetFullPath);
            console.log(`文件复制成功: ${sourceFullPath} -> ${targetFullPath}`);
            return true;
        } catch (error) {
            console.error('复制文件失败:', error.message);
            return false;
        }
    }
    
    deleteFile(filePath) {
        try {
            const fullPath = path.resolve(this.basePath, filePath);
            
            if (!fs.existsSync(fullPath)) {
                throw new Error('文件不存在');
            }
            
            fs.unlinkSync(fullPath);
            console.log(`文件删除成功: ${fullPath}`);
            return true;
        } catch (error) {
            console.error('删除文件失败:', error.message);
            return false;
        }
    }
    
    listDirectory(dirPath = '.') {
        try {
            const fullPath = path.resolve(this.basePath, dirPath);
            const items = fs.readdirSync(fullPath);
            
            console.log(`目录内容 (${fullPath}):`);
            items.forEach(item => {
                const itemPath = path.join(fullPath, item);
                const stats = fs.statSync(itemPath);
                const type = stats.isDirectory() ? '📁' : '📄';
                console.log(`${type} ${item}`);
            });
            
            return items;
        } catch (error) {
            console.error('列出目录失败:', error.message);
            return [];
        }
    }
}

const EventEmitter = require('events');

class EventManager extends EventEmitter {
    constructor() {
        super();
        this.listeners = new Map();
    }
    
    on(event, listener) {
        super.on(event, listener);
        
        if (!this.listeners.has(event)) {
            this.listeners.set(event, new Set());
        }
        this.listeners.get(event).add(listener);
        
        console.log(`注册监听器: ${event}`);
    }
    
    emit(event, ...args) {
        console.log(`触发事件: ${event}`, args);
        return super.emit(event, ...args);
    }
    
    removeListener(event, listener) {
        super.removeListener(event, listener);
        
        if (this.listeners.has(event)) {
            this.listeners.get(event).delete(listener);
            if (this.listeners.get(event).size === 0) {
                this.listeners.delete(event);
            }
        }
        
        console.log(`移除监听器: ${event}`);
    }
    
    getListenerCount(event) {
        const count = this.listeners.has(event) ? this.listeners.get(event).size : 0;
        console.log(`事件 ${event} 的监听器数量: ${count}`);
        return count;
    }
    
    removeAllListeners(event) {
        if (event) {
            super.removeAllListeners(event);
            this.listeners.delete(event);
            console.log(`清空事件 ${event} 的所有监听器`);
        } else {
            super.removeAllListeners();
            this.listeners.clear();
            console.log('清空所有监听器');
        }
    }
}

// 测试代码

const fs = require('fs');
const path = require('path');

class Logger {
    constructor(options = {}) {
        this.level = options.level || 'info';
        this.filePath = options.filePath || 'app.log';
        this.maxSize = options.maxSize || 1024 * 1024; // 1MB
        this.levels = {
            debug: 0,
            info: 1,
            warn: 2,
            error: 3
        };
    }
    
    log(level, message, ...args) {
        if (this.levels[level] < this.levels[this.level]) {
            return;
        }
        
        const timestamp = new Date().toISOString();
        const formattedMessage = `[${timestamp}] [${level.toUpperCase()}] ${message}`;
        const fullMessage = args.length > 0 ? formattedMessage + ' ' + args.join(' ') : formattedMessage;
        
        // 控制台输出
        console.log(fullMessage);
        
        // 文件输出
        this.writeToFile(fullMessage);
    }
    
    debug(message, ...args) {
        this.log('debug', message, ...args);
    }
    
    info(message, ...args) {
        this.log('info', message, ...args);
    }
    
    warn(message, ...args) {
        this.log('warn', message, ...args);
    }
    
    error(message, ...args) {
        this.log('error', message, ...args);
    }
    
    writeToFile(message) {
        try {
            // 检查文件大小
            if (fs.existsSync(this.filePath)) {
                const stats = fs.statSync(this.filePath);
                if (stats.size > this.maxSize) {
                    this.rotateLog();
                }
            }
            
            fs.appendFileSync(this.filePath, message + '\n', 'utf8');
        } catch (error) {
            console.error('写入日志文件失败:', error.message);
        }
    }
    
    rotateLog() {
        try {
            const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
            const backupPath = `${this.filePath}.${timestamp}`;
            
            if (fs.existsSync(this.filePath)) {
                fs.renameSync(this.filePath, backupPath);
                console.log(`日志轮转: ${this.filePath} -> ${backupPath}`);
            }
        } catch (error) {
            console.error('日志轮转失败:', error.message);
        }
    }
}

const http = require('http');
const url = require('url');
const fs = require('fs');
const path = require('path');

class HttpServer {
    constructor(port = 3000) {
        this.port = port;
        this.middlewares = [];
        this.routes = new Map();
    }
    
    use(middleware) {
        this.middlewares.push(middleware);
    }
    
    route(method, path, handler) {
        const key = `${method.toUpperCase()}:${path}`;
        this.routes.set(key, handler);
    }
    
    static(staticPath) {
        this.use((req, res, next) => {
            if (req.method === 'GET') {
                const parsedUrl = url.parse(req.url, true);
                const filePath = path.join(staticPath, parsedUrl.pathname);
                
                if (fs.existsSync(filePath) && fs.statSync(filePath).isFile()) {
                    const ext = path.extname(filePath);
                    const contentType = this.getContentType(ext);
                    
                    res.setHeader('Content-Type', contentType);
                    res.writeHead(200);
                    res.end(fs.readFileSync(filePath));
                    return;
                }
            }
            next();
        });
    }
    
    getContentType(ext) {
        const types = {
            '.html': 'text/html',
            '.css': 'text/css',
            '.js': 'application/javascript',
            '.json': 'application/json',
            '.png': 'image/png',
            '.jpg': 'image/jpeg',
            '.gif': 'image/gif'
        };
        return types[ext] || 'text/plain';
    }
    
    start() {
        const server = http.createServer((req, res) => {
            // 执行中间件
            let index = 0;
            const next = () => {
                if (index < this.middlewares.length) {
                    const middleware = this.middlewares[index++];
                    middleware(req, res, next);
                } else {
                    this.handleRequest(req, res);
                }
            };
            next();
        });
        
        server.listen(this.port, () => {
            console.log(`HTTP服务器运行在 http://localhost:${this.port}`);
        });
    }
    
    handleRequest(req, res) {
        const parsedUrl = url.parse(req.url, true);
        const method = req.method.toUpperCase();
        const pathname = parsedUrl.pathname;
        const key = `${method}:${pathname}`;
        
        const handler = this.routes.get(key);
        if (handler) {
            handler(req, res);
        } else {
            res.writeHead(404, { 'Content-Type': 'text/html; charset=utf-8' });
            res.end('<h1>404 - 页面未找到</h1>');
        }
    }
}

// 测试代码

// 添加中间件

    console.log(`${new Date().toISOString()} - ${req.method} ${req.url}`);

server.route('GET', '/api/users', (req, res) => {

const fs = require('fs');
const path = require('path');
const readline = require('readline');

class FileManagerCLI {
    constructor() {
        this.currentPath = process.cwd();
        this.rl = readline.createInterface({
            input: process.stdin,
            output: process.stdout
        });
    }
    
    start() {
        console.log('文件管理器CLI启动');
        console.log('输入 help 查看帮助');
        this.prompt();
    }
    
    prompt() {
        this.rl.question(`${this.currentPath}> `, (input) => {
            this.handleCommand(input.trim());
        });
    }
    
    handleCommand(input) {
        const [command, ...args] = input.split(' ');
        
        switch (command) {
            case 'help':
                this.showHelp();
                break;
            case 'ls':
                this.listFiles();
                break;
            case 'cd':
                this.changeDirectory(args[0]);
                break;
            case 'mkdir':
                this.createDirectory(args[0]);
                break;
            case 'touch':
                this.createFile(args[0]);
                break;
            case 'rm':
                this.deleteFile(args[0]);
                break;
            case 'search':
                this.searchFiles(args[0]);
                break;
            case 'batch':
                this.batchOperation(args[0], args.slice(1));
                break;
            case 'exit':
                this.rl.close();
                return;
            default:
                console.log('未知命令，输入 help 查看帮助');
        }
        
        this.prompt();
    }
    
    showHelp() {
        console.log(`
可用命令:
  help          - 显示帮助
  ls            - 列出当前目录文件
  cd <path>     - 切换目录
  mkdir <name>  - 创建目录
  touch <name>  - 创建文件
  rm <name>     - 删除文件
  search <pattern> - 搜索文件
  batch <op> <files> - 批量操作
  exit          - 退出
        `);
    }
    
    listFiles() {
        try {
            const files = fs.readdirSync(this.currentPath);
            files.forEach(file => {
                const filePath = path.join(this.currentPath, file);
                const stats = fs.statSync(filePath);
                const type = stats.isDirectory() ? '📁' : '📄';
                const size = stats.isFile() ? ` (${stats.size} bytes)` : '';
                console.log(`${type} ${file}${size}`);
            });
        } catch (error) {
            console.error('列出文件失败:', error.message);
        }
    }
    
    changeDirectory(newPath) {
        try {
            const targetPath = path.resolve(this.currentPath, newPath || '');
            if (fs.existsSync(targetPath) && fs.statSync(targetPath).isDirectory()) {
                this.currentPath = targetPath;
                console.log(`切换到目录: ${this.currentPath}`);
            } else {
                console.error('目录不存在');
            }
        } catch (error) {
            console.error('切换目录失败:', error.message);
        }
    }
    
    createDirectory(name) {
        try {
            const dirPath = path.join(this.currentPath, name);
            fs.mkdirSync(dirPath);
            console.log(`目录创建成功: ${dirPath}`);
        } catch (error) {
            console.error('创建目录失败:', error.message);
        }
    }
    
    createFile(name) {
        try {
            const filePath = path.join(this.currentPath, name);
            fs.writeFileSync(filePath, '');
            console.log(`文件创建成功: ${filePath}`);
        } catch (error) {
            console.error('创建文件失败:', error.message);
        }
    }
    
    deleteFile(name) {
        try {
            const filePath = path.join(this.currentPath, name);
            if (fs.existsSync(filePath)) {
                fs.unlinkSync(filePath);
                console.log(`文件删除成功: ${filePath}`);
            } else {
                console.error('文件不存在');
            }
        } catch (error) {
            console.error('删除文件失败:', error.message);
        }
    }
    
    searchFiles(pattern) {
        try {
            const results = this.searchFilesRecursive(this.currentPath, pattern);
            if (results.length > 0) {
                console.log('搜索结果:');
                results.forEach(file => {
                    console.log(`  ${file}`);
                });
            } else {
                console.log('未找到匹配的文件');
            }
        } catch (error) {
            console.error('搜索失败:', error.message);
        }
    }
    
    searchFilesRecursive(dirPath, pattern) {
        const results = [];
        const files = fs.readdirSync(dirPath);
        
        for (const file of files) {
            const filePath = path.join(dirPath, file);
            const stats = fs.statSync(filePath);
            
            if (stats.isDirectory()) {
                const subResults = this.searchFilesRecursive(filePath, pattern);
                results.push(...subResults);
            } else if (file.includes(pattern)) {
                results.push(filePath);
            }
        }
        
        return results;
    }
    
    batchOperation(operation, files) {
        if (!operation || files.length === 0) {
            console.error('用法: batch <operation> <file1> <file2> ...');
            return;
        }
        
        files.forEach(file => {
            try {
                const filePath = path.join(this.currentPath, file);
                
                switch (operation) {
                    case 'delete':
                        if (fs.existsSync(filePath)) {
                            fs.unlinkSync(filePath);
                            console.log(`删除: ${file}`);
                        }
                        break;
                    case 'copy':
                        const copyPath = filePath + '.copy';
                        fs.copyFileSync(filePath, copyPath);
                        console.log(`复制: ${file} -> ${file}.copy`);
                        break;
                    default:
                        console.error(`未知操作: ${operation}`);
                }
            } catch (error) {
                console.error(`操作失败 ${file}:`, error.message);
            }
        });
    }
}
