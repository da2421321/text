# 第十一章：自定义插件开发

## 11.1 插件基础

TailwindCSS v4 的插件通过 `@plugin` 指令加载：

```css
@import "tailwindcss";
@plugin "./my-plugin.js";
@plugin "@tailwindcss/typography";
```

插件文件结构：

```js
// my-plugin.js
import plugin from 'tailwindcss/plugin'

export default plugin(function({ addUtilities, addComponents, addVariant, theme }) {
  // 添加工具类
  addUtilities({ ... })

  // 添加组件
  addComponents({ ... })

  // 添加变体
  addVariant('hocus', ['&:hover', '&:focus'])
})
```

## 11.2 添加工具类（`addUtilities`）

```js
import plugin from 'tailwindcss/plugin'

export default plugin(function({ addUtilities }) {
  addUtilities({
    // 隐藏滚动条
    '.scrollbar-hide': {
      'scrollbar-width': 'none',
      '&::-webkit-scrollbar': {
        display: 'none',
      },
    },

    // 文字渐变
    '.text-gradient': {
      'background-clip': 'text',
      '-webkit-background-clip': 'text',
      '-webkit-text-fill-color': 'transparent',
    },

    // 绝对居中
    '.center-absolute': {
      position: 'absolute',
      top: '50%',
      left: '50%',
      transform: 'translate(-50%, -50%)',
    },

    // 安全区域内边距（移动端）
    '.safe-top': { 'padding-top': 'env(safe-area-inset-top)' },
    '.safe-bottom': { 'padding-bottom': 'env(safe-area-inset-bottom)' },
  })
})
```

## 11.3 添加组件（`addComponents`）

```js
import plugin from 'tailwindcss/plugin'

export default plugin(function({ addComponents, theme }) {
  addComponents({
    // 按钮基础样式
    '.btn': {
      display: 'inline-flex',
      alignItems: 'center',
      justifyContent: 'center',
      fontWeight: '600',
      borderRadius: theme('borderRadius.lg'),
      transition: 'all 150ms',
      '&:focus': {
        outline: 'none',
        boxShadow: `0 0 0 3px ${theme('colors.blue.500')}40`,
      },
      '&:disabled': {
        opacity: '0.5',
        cursor: 'not-allowed',
      },
    },

    '.btn-primary': {
      backgroundColor: theme('colors.blue.600'),
      color: '#fff',
      '&:hover:not(:disabled)': {
        backgroundColor: theme('colors.blue.700'),
      },
    },

    // 卡片
    '.card': {
      backgroundColor: '#fff',
      borderRadius: theme('borderRadius.2xl'),
      boxShadow: theme('boxShadow.sm'),
      border: `1px solid ${theme('colors.gray.100')}`,
    },
  })
})
```

## 11.4 添加变体（`addVariant`）

```js
import plugin from 'tailwindcss/plugin'

export default plugin(function({ addVariant }) {
  // 悬停或聚焦
  addVariant('hocus', ['&:hover', '&:focus'])

  // 父元素有特定 class 时
  addVariant('sidebar-open', '.sidebar-open &')

  // 支持某 CSS 特性时
  addVariant('supports-grid', '@supports (display: grid)')

  // 打印模式
  addVariant('print', '@media print')

  // 高对比度模式
  addVariant('high-contrast', '@media (forced-colors: active)')

  // 减少动画偏好
  addVariant('motion-safe', '@media (prefers-reduced-motion: no-preference)')
  addVariant('motion-reduce', '@media (prefers-reduced-motion: reduce)')
})
```

使用：

```html
<button class="hocus:bg-blue-700 bg-blue-600 text-white px-4 py-2 rounded">
  悬停或聚焦变深
</button>

<div class="print:hidden">
  仅屏幕显示，打印时隐藏
</div>

<div class="motion-safe:animate-bounce motion-reduce:animate-none">
  尊重用户动画偏好
</div>
```

## 11.5 带配置的插件

```js
import plugin from 'tailwindcss/plugin'

// 可配置的插件
export default plugin.withOptions(
  function(options = {}) {
    const {
      prefix = 'tw',
      variants = ['hover', 'focus'],
    } = options

    return function({ addUtilities, addVariant }) {
      addUtilities({
        [`.${prefix}-glass`]: {
          'backdrop-filter': 'blur(12px)',
          'background': 'rgb(255 255 255 / 0.1)',
          'border': '1px solid rgb(255 255 255 / 0.2)',
        },
      })
    }
  },
  function(options = {}) {
    // 返回主题扩展
    return {
      theme: {
        extend: {
          backdropBlur: {
            xs: '2px',
          },
        },
      },
    }
  }
)
```

使用带配置的插件：

```css
@import "tailwindcss";
@plugin "./my-plugin.js" {
  prefix: "ui";
}
```

## 11.6 官方插件

### `@tailwindcss/typography`

为富文本内容提供精美排版：

```bash
npm install @tailwindcss/typography
```

```css
@import "tailwindcss";
@plugin "@tailwindcss/typography";
```

```html
<article class="prose prose-lg prose-blue dark:prose-invert max-w-none">
  <h1>文章标题</h1>
  <p>文章内容，自动获得精美排版样式...</p>
  <pre><code>代码块</code></pre>
</article>
```

### `@tailwindcss/forms`

重置表单元素样式：

```bash
npm install @tailwindcss/forms
```

```css
@import "tailwindcss";
@plugin "@tailwindcss/forms";
```

### `@tailwindcss/aspect-ratio`

宽高比工具类（v4 已内置，无需安装）：

```html
<div class="aspect-video">16:9 视频容器</div>
<div class="aspect-square">正方形</div>
<div class="aspect-[4/3]">4:3 比例</div>
```

## 11.7 实战：构建 UI 组件插件

```js
// plugins/ui-components.js
import plugin from 'tailwindcss/plugin'

export default plugin(function({ addComponents, addUtilities, theme }) {
  // Badge 徽章
  addComponents({
    '.badge': {
      display: 'inline-flex',
      alignItems: 'center',
      padding: '0.125rem 0.5rem',
      borderRadius: '9999px',
      fontSize: '0.75rem',
      fontWeight: '500',
      lineHeight: '1rem',
    },
    '.badge-blue': {
      backgroundColor: theme('colors.blue.100'),
      color: theme('colors.blue.700'),
    },
    '.badge-green': {
      backgroundColor: theme('colors.green.100'),
      color: theme('colors.green.700'),
    },
    '.badge-red': {
      backgroundColor: theme('colors.red.100'),
      color: theme('colors.red.700'),
    },
    '.badge-gray': {
      backgroundColor: theme('colors.gray.100'),
      color: theme('colors.gray.700'),
    },
  })

  // Divider 分割线
  addComponents({
    '.divider': {
      display: 'flex',
      alignItems: 'center',
      gap: '1rem',
      '&::before, &::after': {
        content: '""',
        flex: '1',
        height: '1px',
        backgroundColor: theme('colors.gray.200'),
      },
    },
  })
})
```

## 小结

- `@plugin` 指令加载插件
- `addUtilities`：添加工具类
- `addComponents`：添加组件样式
- `addVariant`：添加自定义变体
- `plugin.withOptions`：创建可配置插件
- 官方插件：`@tailwindcss/typography`、`@tailwindcss/forms`
