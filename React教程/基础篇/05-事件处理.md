# 事件处理

## React 事件处理基础

React 事件处理与原生 JavaScript 事件处理类似，但有一些重要的区别。

## 基本语法

### 事件绑定

```jsx
function Button() {
  const handleClick = () => {
    console.log('按钮被点击了');
  };

  return <button onClick={handleClick}>点击我</button>;
}
```

### 内联事件处理

```jsx
function Button() {
  return (
    <button onClick={() => console.log('按钮被点击了')}>
      点击我
    </button>
  );
}
```

### 带参数的事件处理

```jsx
function Button() {
  const handleClick = (message) => {
    console.log(message);
  };

  return (
    <button onClick={() => handleClick('Hello, React!')}>
      点击我
    </button>
  );
}
```

## 常见事件类型

### 鼠标事件

```jsx
function MouseEventDemo() {
  const handleClick = (e) => {
    console.log('点击事件:', e.type);
    console.log('坐标:', e.clientX, e.clientY);
  };

  const handleDoubleClick = () => {
    console.log('双击事件');
  };

  const handleMouseEnter = () => {
    console.log('鼠标进入');
  };

  const handleMouseLeave = () => {
    console.log('鼠标离开');
  };

  return (
    <div>
      <button onClick={handleClick}>点击</button>
      <button onDoubleClick={handleDoubleClick}>双击</button>
      <div
        onMouseEnter={handleMouseEnter}
        onMouseLeave={handleMouseLeave}
        style={{ padding: '20px', border: '1px solid #ccc' }}
      >
        鼠标悬停区域
      </div>
    </div>
  );
}
```

### 键盘事件

```jsx
function KeyboardEventDemo() {
  const handleKeyDown = (e) => {
    console.log('按键:', e.key);
    console.log('键码:', e.keyCode);
  };

  const handleKeyPress = (e) => {
    console.log('按键按下:', e.key);
  };

  const handleKeyUp = (e) => {
    console.log('按键释放:', e.key);
  };

  return (
    <div>
      <input
        type="text"
        onKeyDown={handleKeyDown}
        onKeyPress={handleKeyPress}
        onKeyUp={handleKeyUp}
        placeholder="输入文字"
      />
    </div>
  );
}
```

### 表单事件

```jsx
function FormEventDemo() {
  const handleChange = (e) => {
    console.log('值变化:', e.target.value);
  };

  const handleSubmit = (e) => {
    e.preventDefault(); // 阻止表单默认提交行为
    console.log('表单提交');
  };

  const handleFocus = () => {
    console.log('输入框获得焦点');
  };

  const handleBlur = () => {
    console.log('输入框失去焦点');
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="text"
        onChange={handleChange}
        onFocus={handleFocus}
        onBlur={handleBlur}
        placeholder="输入文字"
      />
      <button type="submit">提交</button>
    </form>
  );
}
```

## 事件对象

React 事件对象是 SyntheticEvent，它包装了原生事件对象。

```jsx
function EventObjectDemo() {
  const handleClick = (e) => {
    console.log('事件类型:', e.type);
    console.log('目标元素:', e.target);
    console.log('当前元素:', e.currentTarget);
    console.log('阻止默认行为:', e.preventDefault());
    console.log('阻止事件冒泡:', e.stopPropagation());
  };

  return <button onClick={handleClick}>点击我</button>;
}
```

## 阻止默认行为

```jsx
function PreventDefaultDemo() {
  const handleClick = (e) => {
    e.preventDefault(); // 阻止链接默认跳转行为
    console.log('链接被点击，但不会跳转');
  };

  return (
    <a href="https://example.com" onClick={handleClick}>
      点击链接
    </a>
  );
}
```

## 阻止事件冒泡

```jsx
function StopPropagationDemo() {
  const handleParentClick = () => {
    console.log('父元素被点击');
  };

  const handleChildClick = (e) => {
    e.stopPropagation(); // 阻止事件冒泡到父元素
    console.log('子元素被点击');
  };

  return (
    <div onClick={handleParentClick} style={{ padding: '20px', border: '1px solid #ccc' }}>
      <button onClick={handleChildClick}>点击子元素</button>
    </div>
  );
}
```

## 条件事件处理

```jsx
function ConditionalEventDemo() {
  const [isEnabled, setIsEnabled] = useState(true);

  const handleClick = () => {
    if (isEnabled) {
      console.log('按钮被点击');
    }
  };

  return (
    <div>
      <button onClick={handleClick}>
        {isEnabled ? '点击我' : '已禁用'}
      </button>
      <button onClick={() => setIsEnabled(!isEnabled)}>
        切换状态
      </button>
    </div>
  );
}
```

## 实战示例：交互式计数器

```jsx
import { useState } from 'react';
import './Counter.css';

function Counter() {
  const [count, setCount] = useState(0);
  const [step, setStep] = useState(1);

  const handleIncrement = () => {
    setCount(count + step);
  };

  const handleDecrement = () => {
    setCount(count - step);
  };

  const handleReset = () => {
    setCount(0);
  };

  const handleStepChange = (e) => {
    setStep(Number(e.target.value));
  };

  const handleKeyPress = (e) => {
    if (e.key === 'ArrowUp') {
      handleIncrement();
    } else if (e.key === 'ArrowDown') {
      handleDecrement();
    } else if (e.key === 'r' || e.key === 'R') {
      handleReset();
    }
  };

  return (
    <div className="counter" tabIndex="0" onKeyDown={handleKeyPress}>
      <h1>计数器</h1>
      <div className="counter-display">
        <span className="count">{count}</span>
      </div>

      <div className="counter-controls">
        <button onClick={handleDecrement}>-</button>
        <button onClick={handleReset}>重置</button>
        <button onClick={handleIncrement}>+</button>
      </div>

      <div className="counter-step">
        <label htmlFor="step">步长:</label>
        <input
          id="step"
          type="number"
          value={step}
          onChange={handleStepChange}
          min="1"
          max="10"
        />
      </div>

      <div className="counter-hint">
        <p>提示: 使用键盘 ↑ ↓ 键或 R 键</p>
      </div>
    </div>
  );
}

export default Counter;
```

## 实战示例：拖拽功能

```jsx
import { useState, useRef } from 'react';
import './Draggable.css';

function Draggable() {
  const [position, setPosition] = useState({ x: 0, y: 0 });
  const [isDragging, setIsDragging] = useState(false);
  const [dragStart, setDragStart] = useState({ x: 0, y: 0 });
  const boxRef = useRef(null);

  const handleMouseDown = (e) => {
    setIsDragging(true);
    setDragStart({
      x: e.clientX - position.x,
      y: e.clientY - position.y
    });
  };

  const handleMouseMove = (e) => {
    if (!isDragging) return;

    const newX = e.clientX - dragStart.x;
    const newY = e.clientY - dragStart.y;

    setPosition({ x: newX, y: newY });
  };

  const handleMouseUp = () => {
    setIsDragging(false);
  };

  return (
    <div
      className="draggable-container"
      onMouseMove={handleMouseMove}
      onMouseUp={handleMouseUp}
    >
      <div
        ref={boxRef}
        className={`draggable-box ${isDragging ? 'dragging' : ''}`}
        style={{
          transform: `translate(${position.x}px, ${position.y}px)`
        }}
        onMouseDown={handleMouseDown}
      >
        拖拽我
      </div>
    </div>
  );
}

export default Draggable;
```

## 事件处理最佳实践

1. **使用箭头函数**：避免 `this` 绑定问题
2. **避免内联函数**：在渲染时创建新函数会影响性能
3. **合理使用事件委托**：减少事件监听器数量
4. **清理事件监听器**：在组件卸载时移除监听器
5. **使用 TypeScript**：为事件处理函数添加类型

## 性能优化

```jsx
import { useCallback } from 'react';

function OptimizedButton() {
  const handleClick = useCallback(() => {
    console.log('按钮被点击');
  }, []); // 空依赖数组，函数不会改变

  return <button onClick={handleClick}>点击我</button>;
}
```
