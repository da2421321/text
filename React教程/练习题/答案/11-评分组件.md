# 练习 11：评分组件

## 题目回顾

创建一个星级评分组件。

**要求：**
- 显示 5 颗星
- 支持点击评分
- 支持半星评分
- 显示评分数值
- 支持只读模式

---

## 参考答案

### Rating.tsx

```tsx
import React, { useState } from 'react';
import './Rating.css';

interface RatingProps {
  value: number;
  onChange?: (value: number) => void;
  readOnly?: boolean;
  max?: number;
  size?: 'small' | 'medium' | 'large';
}

function Rating({
  value,
  onChange,
  readOnly = false,
  max = 5,
  size = 'medium'
}: RatingProps) {
  const [hoverValue, setHoverValue] = useState<number | null>(null);

  const displayValue = hoverValue ?? value;

  const handleClick = (e: React.MouseEvent<SVGSVGElement>) => {
    if (readOnly || !onChange) return;

    const svg = e.currentTarget;
    const rect = svg.getBoundingClientRect();
    const isHalf = e.clientX - rect.left < rect.width / 2;
    const newValue = Number(svg.dataset.index!) + (isHalf ? 0.5 : 1);
    onChange(Math.min(Math.max(newValue, 0.5), max));
  };

  const handleMouseMove = (e: React.MouseEvent<SVGSVGElement>) => {
    if (readOnly) return;
    const svg = e.currentTarget;
    const rect = svg.getBoundingClientRect();
    const isHalf = e.clientX - rect.left < rect.width / 2;
    const index = Number(svg.dataset.index!);
    setHoverValue(index + (isHalf ? 0.5 : 1));
  };

  const sizeMap = { small: 16, medium: 24, large: 32 };
  const starSize = sizeMap[size];

  return (
    <div className={`rating rating--${size} ${readOnly ? 'rating--readonly' : ''}`}>
      <div className="rating__stars">
        {Array.from({ length: max }).map((_, i) => {
          const starValue = i + 1;
          const isFull = displayValue >= starValue;
          const isHalf = !isFull && displayValue > i && displayValue < starValue;

          return (
            <svg
              key={i}
              data-index={i}
              width={starSize}
              height={starSize}
              viewBox="0 0 24 24"
              className={`rating__star ${isFull ? 'rating__star--full' : ''} ${isHalf ? 'rating__star--half' : ''}`}
              onClick={handleClick}
              onMouseMove={handleMouseMove}
              onMouseLeave={() => setHoverValue(null)}
            >
              <defs>
                <linearGradient id={`half-${i}`}>
                  <stop offset="50%" stopColor="#f59e0b" />
                  <stop offset="50%" stopColor="#e5e7eb" />
                </linearGradient>
              </defs>
              <path
                d="M12 2l3.09 6.26L22 9.27l-5 4.87 1.18 6.88L12 17.77l-6.18 3.25L7 14.14 2 9.27l6.91-1.01L12 2z"
                fill={isFull ? '#f59e0b' : isHalf ? `url(#half-${i})` : '#e5e7eb'}
              />
            </svg>
          );
        })}
      </div>
      <span className="rating__value">{displayValue.toFixed(1)}</span>
    </div>
  );
}

export default Rating;
```

### Rating.css

```css
.rating {
  display: inline-flex;
  align-items: center;
  gap: 8px;
}

.rating__stars {
  display: flex;
  gap: 2px;
}

.rating__star {
  cursor: pointer;
  transition: transform 0.1s;
}

.rating__star:hover {
  transform: scale(1.2);
}

.rating--readonly .rating__star {
  cursor: default;
}

.rating--readonly .rating__star:hover {
  transform: none;
}

.rating__value {
  font-size: 14px;
  font-weight: 600;
  color: #f59e0b;
  min-width: 30px;
}
```

### 使用示例

```tsx
import React, { useState } from 'react';
import Rating from './Rating';

function App() {
  const [rating, setRating] = useState(3.5);

  return (
    <div>
      <h2>点击评分</h2>
      <Rating value={rating} onChange={setRating} />

      <h2>只读模式</h2>
      <Rating value={4} readOnly />

      <h2>不同尺寸</h2>
      <Rating value={4.5} onChange={setRating} size="small" />
      <Rating value={4.5} onChange={setRating} size="medium" />
      <Rating value={4.5} onChange={setRating} size="large" />
    </div>
  );
}
```

---

## 知识点

- `useState` 管理评分状态
- SVG 图标绘制五角星
- `getBoundingClientRect` 计算鼠标点击位置判断半星
- CSS `linearGradient` 实现半星填充效果
- `hoverValue` 实现悬停预览效果
