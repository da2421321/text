# 练习 15：日历组件

## 题目回顾

创建一个日历组件。

**要求：**
- 显示当前月份
- 切换上一个月/下一个月
- 标记今天
- 支持选择日期
- 显示选中日期

---

## 参考答案

### Calendar.tsx

```tsx
import React, { useState, useMemo } from 'react';
import './Calendar.css';

interface CalendarProps {
  value?: Date;
  onChange?: (date: Date) => void;
  locale?: string;
  highlightToday?: boolean;
  disabledDate?: (date: Date) => boolean;
}

const WEEKDAYS = ['日', '一', '二', '三', '四', '五', '六'];
const MONTHS = [
  '一月', '二月', '三月', '四月', '五月', '六月',
  '七月', '八月', '九月', '十月', '十一月', '十二月',
];

function Calendar({
  value,
  onChange,
  locale = 'zh-CN',
  highlightToday = true,
  disabledDate,
}: CalendarProps) {
  const [currentYear, setCurrentYear] = useState(value?.getFullYear() || new Date().getFullYear());
  const [currentMonth, setCurrentMonth] = useState(value?.getMonth() || new Date().getMonth());
  const [selectedDate, setSelectedDate] = useState<Date | null>(value || null);

  const today = useMemo(() => new Date(), []);

  const days = useMemo(() => {
    const firstDay = new Date(currentYear, currentMonth, 1);
    const lastDay = new Date(currentYear, currentMonth + 1, 0);
    const startWeekday = firstDay.getDay();
    const totalDays = lastDay.getDate();

    const cells: (number | null)[] = [];

    // Fill empty cells before the first day
    for (let i = 0; i < startWeekday; i++) {
      cells.push(null);
    }

    // Fill all days
    for (let day = 1; day <= totalDays; day++) {
      cells.push(day);
    }

    return cells;
  }, [currentYear, currentMonth]);

  const prevMonth = () => {
    if (currentMonth === 0) {
      setCurrentMonth(11);
      setCurrentYear((y) => y - 1);
    } else {
      setCurrentMonth((m) => m - 1);
    }
  };

  const nextMonth = () => {
    if (currentMonth === 11) {
      setCurrentMonth(0);
      setCurrentYear((y) => y + 1);
    } else {
      setCurrentMonth((m) => m + 1);
    }
  };

  const goToToday = () => {
    const now = new Date();
    setCurrentYear(now.getFullYear());
    setCurrentMonth(now.getMonth());
    handleSelect(now);
  };

  const handleSelect = (day: number | null) => {
    if (day === null) return;
    const date = new Date(currentYear, currentMonth, day);
    if (disabledDate && disabledDate(date)) return;
    setSelectedDate(date);
    onChange?.(date);
  };

  const isToday = (day: number | null) => {
    if (!highlightToday || day === null) return false;
    return (
      day === today.getDate() &&
      currentMonth === today.getMonth() &&
      currentYear === today.getFullYear()
    );
  };

  const isSelected = (day: number | null) => {
    if (day === null || !selectedDate) return false;
    return (
      day === selectedDate.getDate() &&
      currentMonth === selectedDate.getMonth() &&
      currentYear === selectedDate.getFullYear()
    );
  };

  const isDisabled = (day: number | null) => {
    if (day === null || !disabledDate) return false;
    return disabledDate(new Date(currentYear, currentMonth, day));
  };

  const formatSelected = () => {
    if (!selectedDate) return '请选择日期';
    return selectedDate.toLocaleDateString(locale, {
      year: 'numeric',
      month: 'long',
      day: 'numeric',
      weekday: 'long',
    });
  };

  return (
    <div className="calendar">
      <div className="calendar__header">
        <button onClick={prevMonth} className="calendar__nav">
          &lt;
        </button>
        <h3 className="calendar__title">
          {currentYear}年 {MONTHS[currentMonth]}
        </h3>
        <button onClick={nextMonth} className="calendar__nav">
          &gt;
        </button>
      </div>

      <div className="calendar__weekdays">
        {WEEKDAYS.map((day) => (
          <div key={day} className={`calendar__weekday ${day === '日' || day === '六' ? 'calendar__weekday--weekend' : ''}`}>
            {day}
          </div>
        ))}
      </div>

      <div className="calendar__grid">
        {days.map((day, index) => {
          const disabled = isDisabled(day);
          const todayFlag = isToday(day);
          const selectedFlag = isSelected(day);
          const isWeekend = index % 7 === 0 || index % 7 === 6;

          return (
            <div
              key={index}
              className={`calendar__day ${day === null ? 'calendar__day--empty' : ''} ${todayFlag ? 'calendar__day--today' : ''} ${selectedFlag ? 'calendar__day--selected' : ''} ${disabled ? 'calendar__day--disabled' : ''} ${isWeekend && day !== null ? 'calendar__day--weekend' : ''}`}
              onClick={() => !disabled && handleSelect(day)}
            >
              {day}
            </div>
          );
        })}
      </div>

      <div className="calendar__footer">
        <button onClick={goToToday} className="calendar__today">
          今天
        </button>
        <div className="calendar__selected">
          {formatSelected()}
        </div>
      </div>
    </div>
  );
}

export default Calendar;
```

### 使用示例

```tsx
import React from 'react';
import Calendar from './Calendar';

function App() {
  const handleChange = (date: Date) => {
    console.log('选中日期:', date);
  };

  // 禁用周末
  const disabledDate = (date: Date) => {
    const day = date.getDay();
    return day === 0 || day === 6;
  };

  return (
    <div>
      <Calendar onChange={handleChange} />
      <Calendar onChange={handleChange} disabledDate={disabledDate} />
    </div>
  );
}
```

## 知识点

- 日期计算（`new Date(year, month, 1)` 获取月首，`new Date(year, month + 1, 0)` 获取月末）
- `getDay()` 获取星期几（0=周日）
- 网格布局（`grid-template-columns: repeat(7, 1fr)`）
- `useMemo` 缓存日期计算结果
- `locale` 参数支持多语言
- `disabledDate` 函数支持禁用特定日期
