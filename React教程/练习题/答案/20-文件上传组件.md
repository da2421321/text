import { useState, useRef, useCallback, DragEvent, ChangeEvent } from 'react';

// ============ 类型定义 ============
interface FileItem {
  id: string;
  file: File;
  preview?: string;
  progress: number;
  status: 'pending' | 'uploading' | 'success' | 'error';
  error?: string;
}

interface UploadConfig {
  accept?: string;
  maxSize?: number; // bytes
  maxFiles?: number;
  multiple?: boolean;
}

interface FileUploadProps {
  config?: UploadConfig;
  onUpload?: (files: File[]) => Promise<void>;
  onChange?: (files: FileItem[]) => void;
}

// ============ 工具函数 ============
function formatFileSize(bytes: number): string {
  if (bytes === 0) return '0 B';
  const k = 1024;
  const sizes = ['B', 'KB', 'MB', 'GB', 'TB'];
  const i = Math.floor(Math.log(bytes) / Math.log(k));
  return `${parseFloat((bytes / Math.pow(k, i)).toFixed(2))} ${sizes[i]}`;
}

function getFileIcon(file: File): string {
  if (file.type.startsWith('image/')) return '🖼';
  if (file.type.startsWith('video/')) return '🎬';
  if (file.type.startsWith('audio/')) return '🎵';
  if (file.type.startsWith('text/')) return '📄';
  if (file.type.includes('pdf')) return '📕';
  if (file.type.includes('zip') || file.type.includes('rar') || file.type.includes('tar')) return '📦';
  return '📎';
}

// ============ 单个文件项组件 ============
function FileItemComponent({
  item,
  onRemove,
  onRetry,
}: {
  item: FileItem;
  onRemove: (id: string) => void;
  onRetry: (id: string) => void;
}) {
  const isImage = item.file.type.startsWith('image/');

  const getStatusText = () => {
    switch (item.status) {
      case 'pending': return '等待上传';
      case 'uploading': return `上传中 ${item.progress}%`;
      case 'success': return '上传成功';
      case 'error': return item.error || '上传失败';
    }
  };

  return (
    <div className={`file-item ${item.status}`}>
      <div className="file-preview">
        {isImage && item.preview ? (
          <img src={item.preview} alt={item.file.name} />
        ) : (
          <span className="file-icon">{getFileIcon(item.file)}</span>
        )}
      </div>

      <div className="file-info">
        <div className="file-name">{item.file.name}</div>
        <div className="file-meta">
          <span className="file-size">{formatFileSize(item.file.size)}</span>
          <span className="file-status">{getStatusText()}</span>
        </div>

        {(item.status === 'uploading' || item.status === 'pending') && (
          <div className="file-progress">
            <div
              className="progress-bar"
              style={{ width: `${item.progress}%` }}
            />
          </div>
        )}
      </div>

      <div className="file-actions">
        {item.status === 'error' && (
          <button
            className="btn-retry"
            onClick={() => onRetry(item.id)}
            title="重试"
          >
            ↻
          </button>
        )}
        <button
          className="btn-remove"
          onClick={() => onRemove(item.id)}
          title="移除"
        >
          ×
        </button>
      </div>
    </div>
  );
}

// ============ 主文件上传组件 ============
function FileUpload({
  config = {},
  onUpload,
  onChange,
}: FileUploadProps) {
  const {
    accept,
    maxSize = 10 * 1024 * 1024, // 10MB
    maxFiles = 10,
    multiple = true,
  } = config;

  const [files, setFiles] = useState<FileItem[]>([]);
  const [isDragging, setIsDragging] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const inputRef = useRef<HTMLInputElement>(null);

  // 更新文件列表
  const updateFiles = useCallback((newFiles: FileItem[]) => {
    setFiles(newFiles);
    onChange?.(newFiles);
  }, [onChange]);

  // 验证文件
  const validateFile = (file: File): string | null => {
    if (maxSize > 0 && file.size > maxSize) {
      return `文件 ${file.name} 超过最大大小限制 (${formatFileSize(maxSize)})`;
    }
    return null;
  };

  // 生成预览
  const generatePreview = (file: File): string | undefined => {
    if (file.type.startsWith('image/')) {
      return URL.createObjectURL(file);
    }
    return undefined;
  };

  // 添加文件
  const addFiles = useCallback((newFiles: FileList | File[]) => {
    setError(null);
    const fileArray = Array.from(newFiles);

    if (maxFiles > 0 && files.length + fileArray.length > maxFiles) {
      setError(`最多只能上传 ${maxFiles} 个文件`);
      return;
    }

    const validFiles: FileItem[] = [];
    const errors: string[] = [];

    fileArray.forEach(file => {
      const validationError = validateFile(file);
      if (validationError) {
        errors.push(validationError);
      } else {
        validFiles.push({
          id: `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,
          file,
          preview: generatePreview(file),
          progress: 0,
          status: 'pending',
        });
      }
    });

    if (errors.length > 0) {
      setError(errors.join('；'));
    }

    if (validFiles.length > 0) {
      updateFiles([...files, ...validFiles]);
    }
  }, [files, maxFiles, maxSize, updateFiles]);

  // 移除文件
  const removeFile = useCallback((id: string) => {
    const file = files.find(f => f.id === id);
    if (file?.preview) {
      URL.revokeObjectURL(file.preview);
    }
    updateFiles(files.filter(f => f.id !== id));
  }, [files, updateFiles]);

  // 模拟上传
  const uploadFile = async (fileItem: FileItem): Promise<void> => {
    return new Promise((resolve, reject) => {
      const interval = setInterval(() => {
        setFiles(prev => prev.map(f => {
          if (f.id === fileItem.id) {
            const newProgress = Math.min(f.progress + Math.random() * 20, 100);
            if (newProgress >= 100) {
              clearInterval(interval);
              if (Math.random() > 0.1) { // 90% 成功率
                return { ...f, progress: 100, status: 'success' as const };
              } else {
                return { ...f, progress: 0, status: 'error' as const, error: '网络错误' };
              }
            }
            return { ...f, progress: newProgress, status: 'uploading' as const };
          }
          return f;
        }));
      }, 300);
    });
  };

  // 重试上传
  const retryUpload = useCallback((id: string) => {
    setFiles(prev => prev.map(f =>
      f.id === id ? { ...f, progress: 0, status: 'pending' as const, error: undefined } : f
    ));
  }, []);

  // 上传所有文件
  const uploadAll = async () => {
    const pendingFiles = files.filter(f => f.status === 'pending' || f.status === 'error');
    for (const file of pendingFiles) {
      try {
        await uploadFile(file);
      } catch {
        setFiles(prev => prev.map(f =>
          f.id === file.id ? { ...f, status: 'error' as const, error: '上传失败' } : f
        ));
      }
    }
    onUpload?.(pendingFiles.map(f => f.file));
  };

  // 拖拽处理
  const handleDragOver = useCallback((e: DragEvent<HTMLDivElement>) => {
    e.preventDefault();
    setIsDragging(true);
  }, []);

  const handleDragLeave = useCallback((e: DragEvent<HTMLDivElement>) => {
    e.preventDefault();
    setIsDragging(false);
  }, []);

  const handleDrop = useCallback((e: DragEvent<HTMLDivElement>) => {
    e.preventDefault();
    setIsDragging(false);
    addFiles(e.dataTransfer.files);
  }, [addFiles]);

  // 文件选择
  const handleInputChange = useCallback((e: ChangeEvent<HTMLInputElement>) => {
    if (e.target.files) {
      addFiles(e.target.files);
      e.target.value = ''; // 允许重复选择同一文件
    }
  }, [addFiles]);

  // 清除全部
  const clearAll = () => {
    files.forEach(f => {
      if (f.preview) URL.revokeObjectURL(f.preview);
    });
    updateFiles([]);
    setError(null);
  };

  const pendingCount = files.filter(f => f.status === 'pending').length;
  const uploadingCount = files.filter(f => f.status === 'uploading').length;

  return (
    <div className="file-upload-container">
      {/* 拖拽上传区域 */}
      <div
        className={`upload-zone ${isDragging ? 'dragging' : ''}`}
        onDragOver={handleDragOver}
        onDragLeave={handleDragLeave}
        onDrop={handleDrop}
        onClick={() => inputRef.current?.click()}
      >
        <input
          ref={inputRef}
          type="file"
          accept={accept}
          multiple={multiple}
          onChange={handleInputChange}
          style={{ display: 'none' }}
        />
        <div className="upload-icon">📁</div>
        <p className="upload-text">
          拖拽文件到此处，或 <span className="upload-link">点击选择文件</span>
        </p>
        <p className="upload-hint">
          {accept && `支持格式: ${accept}`}
          {maxSize > 0 && ` · 最大 ${formatFileSize(maxSize)}`}
          {maxFiles > 0 && ` · 最多 ${maxFiles} 个文件`}
        </p>
      </div>

      {/* 错误提示 */}
      {error && (
        <div className="upload-error">
          <span>⚠</span> {error}
        </div>
      )}

      {/* 文件列表 */}
      {files.length > 0 && (
        <div className="file-list">
          <div className="file-list-header">
            <span>已选择 {files.length} 个文件</span>
            <div className="header-actions">
              {pendingCount > 0 && (
                <button className="btn-upload-all" onClick={uploadAll}>
                  上传全部
                </button>
              )}
              <button className="btn-clear" onClick={clearAll}>
                清除全部
              </button>
            </div>
          </div>

          {files.map(item => (
            <FileItemComponent
              key={item.id}
              item={item}
              onRemove={removeFile}
              onRetry={retryUpload}
            />
          ))}

          {uploadingCount > 0 && (
            <div className="upload-status">
              上传中... {uploadingCount} 个文件
            </div>
          )}
        </div>
      )}
    </div>
  );
}

// ============ 裁剪上传组件 ============
function ImageUploadWithCrop() {
  const [selectedFile, setSelectedFile] = useState<File | null>(null);
  const [preview, setPreview] = useState<string | null>(null);
  const [croppedImage, setCroppedImage] = useState<string | null>(null);
  const canvasRef = useRef<HTMLCanvasElement>(null);

  const handleFileSelect = (e: ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (file && file.type.startsWith('image/')) {
      setSelectedFile(file);
      setPreview(URL.createObjectURL(file));
    }
  };

  const handleCrop = () => {
    if (!preview || !canvasRef.current) return;
    const canvas = canvasRef.current;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const img = new Image();
    img.onload = () => {
      const size = Math.min(img.width, img.height);
      canvas.width = 200;
      canvas.height = 200;
      ctx.drawImage(
        img,
        (img.width - size) / 2,
        (img.height - size) / 2,
        size,
        size,
        0,
        0,
        200,
        200
      );
      setCroppedImage(canvas.toDataURL('image/jpeg', 0.9));
    };
    img.src = preview;
  };

  const handleUpload = async () => {
    if (!canvasRef.current) return;
    const blob = await new Promise<Blob>((resolve) => {
      canvasRef.current!.toBlob(b => resolve(b!), 'image/jpeg', 0.9);
    });
    const file = new File([blob], 'avatar.jpg', { type: 'image/jpeg' });
    console.log('上传文件:', file);
    alert('头像上传成功！');
  };

  return (
    <div className="image-upload-crop">
      <h3>头像上传（带裁剪）</h3>
      <input type="file" accept="image/*" onChange={handleFileSelect} />

      <div className="crop-preview">
        {preview && (
          <div className="original-preview">
            <img src={preview} alt="原图" />
          </div>
        )}
        <div className="cropped-preview">
          {croppedImage ? (
            <img src={croppedImage} alt="裁剪后" />
          ) : (
            <div className="placeholder">裁剪预览</div>
          )}
          <canvas ref={canvasRef} style={{ display: 'none' }} />
        </div>
      </div>

      {preview && (
        <div className="crop-actions">
          <button onClick={handleCrop}>裁剪</button>
          {croppedImage && <button onClick={handleUpload}>上传</button>}
        </div>
      )}
    </div>
  );
}

## 设计要点

1. **拖拽上传**：支持拖拽文件到上传区域
2. **点击上传**：传统文件选择器方式
3. **文件预览**：图片文件显示缩略图预览
4. **上传进度**：实时显示每个文件的上传进度
5. **批量操作**：支持一键上传全部、清除全部
6. **重试机制**：上传失败的文件支持重试
7. **文件验证**：支持大小限制、格式限制、数量限制
8. **裁剪功能**：额外的头像裁剪上传示例
9. **内存管理**：使用 `URL.revokeObjectURL` 释放对象 URL
