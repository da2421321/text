# 练习 5：商品列表

## 题目回顾

创建一个商品列表组件。

**要求：**
- 显示商品图片、名称、价格
- 实现商品筛选功能
- 实现商品排序功能
- 添加到购物车按钮
- 显示购物车商品数量

---

## 参考答案

### ProductList.tsx

```tsx
import React, { useState, useMemo } from 'react';
import './ProductList.css';

interface Product {
  id: number;
  name: string;
  price: number;
  category: string;
  image: string;
  rating: number;
}

interface CartItem extends Product {
  quantity: number;
}

const mockProducts: Product[] = [
  { id: 1, name: 'MacBook Pro', price: 12999, category: '电脑', image: '💻', rating: 4.8 },
  { id: 2, name: 'iPhone 15', price: 6999, category: '手机', image: '📱', rating: 4.7 },
  { id: 3, name: 'AirPods Pro', price: 1499, category: '配件', image: '🎧', rating: 4.6 },
  { id: 4, name: 'iPad Air', price: 4599, category: '平板', image: '📱', rating: 4.5 },
  { id: 5, name: 'Apple Watch', price: 2999, category: '配件', image: '⌚', rating: 4.4 },
  { id: 6, name: 'iMac 24', price: 10999, category: '电脑', image: '🖥️', rating: 4.9 },
];

type SortKey = 'default' | 'price-asc' | 'price-desc' | 'rating';

function ProductList() {
  const [searchTerm, setSearchTerm] = useState('');
  const [selectedCategory, setSelectedCategory] = useState('全部');
  const [sortBy, setSortBy] = useState<SortKey>('default');
  const [cart, setCart] = useState<CartItem[]>([]);

  // 获取所有分类
  const categories = ['全部', ...new Set(mockProducts.map(p => p.category))];

  // 筛选和排序商品
  const filteredProducts = useMemo(() => {
    let result = mockProducts.filter(product => {
      const matchSearch = product.name.toLowerCase().includes(searchTerm.toLowerCase());
      const matchCategory = selectedCategory === '全部' || product.category === selectedCategory;
      return matchSearch && matchCategory;
    });

    // 排序
    switch (sortBy) {
      case 'price-asc':
        result = [...result].sort((a, b) => a.price - b.price);
        break;
      case 'price-desc':
        result = [...result].sort((a, b) => b.price - a.price);
        break;
      case 'rating':
        result = [...result].sort((a, b) => b.rating - a.rating);
        break;
    }

    return result;
  }, [searchTerm, selectedCategory, sortBy]);

  // 添加到购物车
  const addToCart = (product: Product) => {
    setCart(prev => {
      const existing = prev.find(item => item.id === product.id);
      if (existing) {
        return prev.map(item =>
          item.id === product.id
            ? { ...item, quantity: item.quantity + 1 }
            : item
        );
      }
      return [...prev, { ...product, quantity: 1 }];
    });
  };

  // 购物车总数量
  const cartCount = cart.reduce((sum, item) => sum + item.quantity, 0);
  const cartTotal = cart.reduce((sum, item) => sum + item.price * item.quantity, 0);

  return (
    <div className="product-list">
      {/* 顶部购物车 */}
      <div className="cart-bar">
        <span className="cart-icon">🛒</span>
        <span className="cart-count">{cartCount}</span>
        <span className="cart-total">¥{cartTotal.toLocaleString()}</span>
      </div>

      {/* 搜索和筛选 */}
      <div className="filters">
        <input
          type="text"
          placeholder="搜索商品..."
          value={searchTerm}
          onChange={e => setSearchTerm(e.target.value)}
          className="search-input"
        />

        <div className="filter-group">
          <select
            value={selectedCategory}
            onChange={e => setSelectedCategory(e.target.value)}
          >
            {categories.map(cat => (
              <option key={cat} value={cat}>{cat}</option>
            ))}
          </select>

          <select
            value={sortBy}
            onChange={e => setSortBy(e.target.value as SortKey)}
          >
            <option value="default">默认排序</option>
            <option value="price-asc">价格从低到高</option>
            <option value="price-desc">价格从高到低</option>
            <option value="rating">评分最高</option>
          </select>
        </div>
      </div>

      {/* 商品网格 */}
      <div className="product-grid">
        {filteredProducts.length === 0 ? (
          <div className="no-results">没有找到相关商品</div>
        ) : (
          filteredProducts.map(product => (
            <div key={product.id} className="product-card">
              <div className="product-image">{product.image}</div>
              <div className="product-info">
                <h3 className="product-name">{product.name}</h3>
                <div className="product-meta">
                  <span className="product-category">{product.category}</span>
                  <span className="product-rating">⭐ {product.rating}</span>
                </div>
                <div className="product-price">¥{product.price.toLocaleString()}</div>
              </div>
              <button
                className="add-to-cart"
                onClick={() => addToCart(product)}
              >
                加入购物车
              </button>
            </div>
          ))
        )}
      </div>
    </div>
  );
}

export default ProductList;
```

## 知识点总结

1. **useMemo 优化**：缓存筛选和排序结果
2. **派生状态**：categories 从数据中计算得出
3. **复合筛选**：搜索 + 分类 + 排序组合
4. **购物车状态**：数量累加和总价计算
5. **响应式布局**：grid + auto-fill 自适应
