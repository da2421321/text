# 练习 13：拖拽排序

## 题目回顾

创建一个支持拖拽排序的列表。

**要求：**
- 拖拽列表项
- 实时更新顺序
- 拖拽时的视觉反馈
- 支持嵌套列表
- 保存排序结果

---

## 参考答案

### useDragAndDrop.ts（自定义 Hook）

```tsx
import { useState, useCallback, useRef } from 'react';

interface DragItem {
  id: string;
  index: number;
}

interface UseDragAndDropOptions<T> {
  initialItems: T[];
  onOrderChange?: (items: T[]) => void;
  keyExtractor: (item: T) => string;
}

export function useDragAndDrop<T>({
  initialItems,
  onOrderChange,
  keyExtractor,
}: UseDragAndDropOptions<T>) {
  const [items, setItems] = useState<T[]>(initialItems);
  const [draggedId, setDraggedId] = useState<string | null>(null);
  const [dragOverId, setDragOverId] = useState<string | null>(null);
  const dragNode = useRef<{ index: number; handle: string } | null>(null);

  const handleDragStart = useCallback(
    (e: React.DragEvent, index: number) => {
      dragNode.current = { index, handle: keyExtractor(items[index]) };
      setDraggedId(keyExtractor(items[index]));
      e.dataTransfer.effectAllowed = 'move';
      e.dataTransfer.setData('text/plain', keyExtractor(items[index]));
    },
    [items, keyExtractor]
  );

  const handleDragOver = useCallback(
    (e: React.DragEvent, index: number) => {
      e.preventDefault();
      e.dataTransfer.dropEffect = 'move';
      const dragIndex = dragNode.current?.index;
      if (dragIndex === undefined || dragIndex === index) return;
      setDragOverId(keyExtractor(items[index]));
    },
    [items, keyExtractor]
  );

  const handleDragEnter = useCallback(
    (e: React.DragEvent, index: number) => {
      e.preventDefault();
      const dragIndex = dragNode.current?.index;
      if (dragIndex === undefined || dragIndex === index) return;

      setItems((prev) => {
        const newItems = [...prev];
        const [draggedItem] = newItems.splice(dragIndex, 1);
        newItems.splice(index, 0, draggedItem);
        dragNode.current = { index, handle: keyExtractor(draggedItem) };
        return newItems;
      });
    },
    [keyExtractor]
  );

  const handleDragLeave = useCallback(() => {
    setDragOverId(null);
  }, []);

  const handleDrop = useCallback(
    (e: React.DragEvent) => {
      e.preventDefault();
      if (onOrderChange) {
        onOrderChange(items);
      }
      setDraggedId(null);
      setDragOverId(null);
      dragNode.current = null;
    },
    [items, onOrderChange]
  );

  const handleDragEnd = useCallback(() => {
    setDraggedId(null);
    setDragOverId(null);
    dragNode.current = null;
  }, []);

  return {
    items,
    setItems,
    draggedId,
    dragOverId,
    handleDragStart,
    handleDragOver,
    handleDragEnter,
    handleDragLeave,
    handleDrop,
    handleDragEnd,
  };
}
```

### DragSortableList.tsx

```tsx
import React, { useCallback } from 'react';
import { useDragAndDrop } from './useDragAndDrop';
import './DragSortableList.css';

interface TaskItem {
  id: string;
  title: string;
  priority: 'high' | 'medium' | 'low';
}

function DragSortableList() {
  const [taskItems, setTaskItems] = React.useState<TaskItem[]>([
    { id: '1', title: '完成项目报告', priority: 'high' },
    { id: '2', title: '回复邮件', priority: 'medium' },
    { id: '3', title: '整理文档', priority: 'low' },
    { id: '4', title: '开会讨论', priority: 'medium' },
    { id: '5', title: '代码审查', priority: 'high' },
  ]);

  const handleOrderChange = useCallback((newItems: TaskItem[]) => {
    setTaskItems(newItems);
    localStorage.setItem('taskOrder', JSON.stringify(newItems.map((t) => t.id)));
  }, []);

  const {
    items: displayItems,
    draggedId,
    dragOverId,
    handleDragStart,
    handleDragOver,
    handleDragEnter,
    handleDragLeave,
    handleDrop,
    handleDragEnd,
  } = useDragAndDrop<TaskItem>({
    initialItems: taskItems,
    onOrderChange: handleOrderChange,
    keyExtractor: (item) => item.id,
  });

  const priorityColors = { high: '#ef4444', medium: '#f59e0b', low: '#22c55e' };

  return (
    <div className="drag-list">
      <h2>任务列表（拖拽排序）</h2>
      <p className="drag-list__hint">拖拽项目以改变顺序</p>
      <ul className="drag-list__container">
        {displayItems.map((item, index) => (
          <li
            key={item.id}
            draggable
            className={`drag-list__item ${draggedId === item.id ? 'drag-list__item--dragging' : ''} ${dragOverId === item.id ? 'drag-list__item--dragover' : ''}`}
            onDragStart={(e) => handleDragStart(e, index)}
            onDragOver={(e) => handleDragOver(e, index)}
            onDragEnter={(e) => handleDragEnter(e, index)}
            onDragLeave={handleDragLeave}
            onDrop={handleDrop}
            onDragEnd={handleDragEnd}
          >
            <span className="drag-list__handle">&#8942;&#8942;</span>
            <span
              className="drag-list__priority"
              style={{ background: priorityColors[item.priority] }}
            >
              {item.priority === 'high' ? '高' : item.priority === 'medium' ? '中' : '低'}
            </span>
            <span className="drag-list__title">{item.title}</span>
          </li>
        ))}
      </ul>
      {displayItems.length === 0 && (
        <div className="drag-list__empty">暂无任务</div>
      )}
    </div>
  );
}

export default DragSortableList;
```

## 知识点

- HTML5 原生 Drag and Drop API（`draggable`、`onDragStart`、`onDragOver`、`onDrop`）
- 拖拽状态管理（`draggedId`、`dragOverId`）
- 数组元素交换实现排序
- CSS 视觉反馈（拖拽中、悬停目标）
- `localStorage` 持久化排序结果
