# 练习 19：进度条组件答案

## 完整代码实现

```tsx
import { useState, useEffect, useRef } from 'react';

// ============ 类型定义 ============
interface ProgressBarProps {
  value: number;
  max?: number;
  showLabel?: boolean;
  showPercentage?: boolean;
  color?: string;
  backgroundColor?: string;
  height?: number;
  animated?: boolean;
  striped?: boolean;
  transitionDuration?: number;
}

interface CircularProgressProps {
  value: number;
  max?: number;
  size?: number;
  strokeWidth?: number;
  showLabel?: boolean;
  color?: string;
  trackColor?: string;
}

interface MultiProgressProps {
  items: Array<{
    label: string;
    value: number;
    max?: number;
    color?: string;
  }>;
  showLabel?: boolean;
  height?: number;
}

// ============ 线性进度条 ============
function ProgressBar({
  value,
  max = 100,
  showLabel = true,
  showPercentage = true,
  color = '#4a90d9',
  backgroundColor = '#e9ecef',
  height = 20,
  animated = true,
  striped = false,
  transitionDuration = 300,
}: ProgressBarProps) {
  const percentage = Math.min(Math.max((value / max) * 100, 0), 100);

  const getStripedStyle = (): React.CSSProperties => {
    if (!striped) return {};
    return {
      backgroundImage: `linear-gradient(
        45deg,
        rgba(255, 255, 255, 0.15) 25%,
        transparent 25%,
        transparent 50%,
        rgba(255, 255, 255, 0.15) 50%,
        rgba(255, 255, 255, 0.15) 75%,
        transparent 75%,
        transparent
      )`,
      backgroundSize: '1rem 1rem',
    };
  };

  return (
    <div className="progress-container">
      {showLabel && (
        <div className="progress-header">
          <span className="progress-label">
            {showPercentage ? `${percentage.toFixed(0)}%` : `${value} / ${max}`}
          </span>
        </div>
      )}
      <div
        className="progress-track"
        style={{
          height: `${height}px`,
          backgroundColor,
          borderRadius: height / 2,
        }}
      >
        <div
          className={`progress-fill ${animated ? 'animated' : ''} ${striped ? 'striped' : ''}`}
          style={{
            width: `${percentage}%`,
            backgroundColor: striped ? undefined : color,
            height: '100%',
            borderRadius: height / 2,
            transitionDuration: `${transitionDuration}ms`,
            ...getStripedStyle(),
          }}
        />
      </div>
    </div>
  );
}

// ============ 圆形进度条 ============
function CircularProgress({
  value,
  max = 100,
  size = 120,
  strokeWidth = 10,
  showLabel = true,
  color = '#4a90d9',
  trackColor = '#e9ecef',
}: CircularProgressProps) {
  const percentage = Math.min(Math.max((value / max) * 100, 0), 100);
  const radius = (size - strokeWidth) / 2;
  const circumference = radius * 2 * Math.PI;
  const offset = circumference - (percentage / 100) * circumference;

  return (
    <div className="circular-progress" style={{ width: size, height: size }}>
      <svg width={size} height={size}>
        {/* 背景圆 */}
        <circle
          cx={size / 2}
          cy={size / 2}
          r={radius}
          fill="none"
          stroke={trackColor}
          strokeWidth={strokeWidth}
        />
        {/* 进度圆 */}
        <circle
          cx={size / 2}
          cy={size / 2}
          r={radius}
          fill="none"
          stroke={color}
          strokeWidth={strokeWidth}
          strokeLinecap="round"
          strokeDasharray={circumference}
          strokeDashoffset={offset}
          style={{
            transform: 'rotate(-90deg)',
            transformOrigin: '50% 50%',
            transition: 'stroke-dashoffset 0.5s ease-in-out',
          }}
        />
      </svg>
      {showLabel && (
        <div className="circular-label">
          <span className="percentage">{percentage.toFixed(0)}%</span>
        </div>
      )}
    </div>
  );
}

// ============ 多进度条组 ============
function MultiProgress({
  items,
  showLabel = true,
  height = 16,
}: MultiProgressProps) {
  return (
    <div className="multi-progress">
      {items.map((item, index) => {
        const max = item.max || 100;
        const percentage = Math.min(Math.max((item.value / max) * 100, 0), 100);

        return (
          <div key={index} className="multi-progress-item">
            {showLabel && (
              <div className="multi-progress-header">
                <span className="item-label">{item.label}</span>
                <span className="item-value">{percentage.toFixed(0)}%</span>
              </div>
            )}
            <div
              className="progress-track"
              style={{
                height: `${height}px`,
                backgroundColor: '#e9ecef',
                borderRadius: height / 2,
              }}
            >
              <div
                className="progress-fill"
                style={{
                  width: `${percentage}%`,
                  backgroundColor: item.color || `hsl(${(index * 60) % 360}, 70%, 50%)`,
                  height: '100%',
                  borderRadius: height / 2,
                  transition: 'width 0.5s ease-in-out',
                }}
              />
            </div>
          </div>
        );
      })}
    </div>
  );
}

// ============ 动态进度条（带动画） ============
function AnimatedProgressBar({
  duration = 5000,
  color = '#4a90d9',
}: {
  duration?: number;
  color?: string;
}) {
  const [progress, setProgress] = useState(0);
  const [isRunning, setIsRunning] = useState(false);
  const [isPaused, setIsPaused] = useState(false);
  const rafRef = useRef<number>();
  const startTimeRef = useRef<number>();
  const pausedTimeRef = useRef<number>(0);

  const animate = (timestamp: number) => {
    if (!startTimeRef.current) {
      startTimeRef.current = timestamp - pausedTimeRef.current;
    }

    const elapsed = timestamp - startTimeRef.current;
    const newProgress = Math.min((elapsed / duration) * 100, 100);

    setProgress(newProgress);

    if (newProgress < 100 && !isPaused) {
      rafRef.current = requestAnimationFrame(animate);
    }
  };

  const start = () => {
    setIsRunning(true);
    setIsPaused(false);
    pausedTimeRef.current = 0;
    startTimeRef.current = undefined;
    rafRef.current = requestAnimationFrame(animate);
  };

  const pause = () => {
    if (rafRef.current) {
      cancelAnimationFrame(rafRef.current);
    }
    setIsPaused(true);
    pausedTimeRef.current = (progress / 100) * duration;
  };

  const resume = () => {
    setIsPaused(false);
    startTimeRef.current = undefined;
    rafRef.current = requestAnimationFrame(animate);
  };

  const reset = () => {
    if (rafRef.current) {
      cancelAnimationFrame(rafRef.current);
    }
    setProgress(0);
    setIsRunning(false);
    setIsPaused(false);
    startTimeRef.current = undefined;
    pausedTimeRef.current = 0;
  };

  useEffect(() => {
    return () => {
      if (rafRef.current) {
        cancelAnimationFrame(rafRef.current);
      }
    };
  }, []);

  return (
    <div className="animated-progress">
      <ProgressBar value={progress} color={color} height={24} />
      <div className="progress-controls">
        {!isRunning && progress === 0 && (
          <button onClick={start}>开始</button>
        )}
        {isRunning && !isPaused && <button onClick={pause}>暂停</button>}
        {isRunning && isPaused && <button onClick={resume}>继续</button>}
        <button onClick={reset} disabled={!isRunning && progress === 0}>
          重置
        </button>
      </div>
      <p className="progress-hint">进度: {progress.toFixed(1)}%</p>
    </div>
  );
}

// ============ 分段进度条 ============
function SegmentedProgressBar({
  value,
  max = 100,
  segments = 10,
  activeColor = '#4a90d9',
  inactiveColor = '#e9ecef',
  height = 24,
}: {
  value: number;
  max?: number;
  segments?: number;
  activeColor?: string;
  inactiveColor?: string;
  height?: number;
}) {
  const activeSegments = Math.round((value / max) * segments);

  return (
    <div
      className="segmented-progress"
      style={{
        display: 'flex',
        gap: '4px',
        height: `${height}px`,
      }}
    >
      {Array.from({ length: segments }).map((_, index) => (
        <div
          key={index}
          style={{
            flex: 1,
            backgroundColor: index < activeSegments ? activeColor : inactiveColor,
            borderRadius: index === 0 ? '4px 0 0 4px' : index === segments - 1 ? '0 4px 4px 0' : '0',
            transition: 'background-color 0.3s ease',
          }}
        />
      ))}
    </div>
  );
}

// ============ 使用示例 ============
function ProgressDemo() {
  const [downloadProgress, setDownloadProgress] = useState(0);
  const [uploadProgress, setUploadProgress] = useState(0);

  // 模拟下载
  useEffect(() => {
    const interval = setInterval(() => {
      setDownloadProgress(prev => {
        if (prev >= 100) {
          clearInterval(interval);
          return 100;
        }
        return prev + Math.random() * 5;
      });
    }, 200);

    return () => clearInterval(interval);
  }, []);

  // 模拟上传
  useEffect(() => {
    const interval = setInterval(() => {
      setUploadProgress(prev => {
        if (prev >= 100) {
          clearInterval(interval);
          return 100;
        }
        return prev + Math.random() * 3;
      });
    }, 300);

    return () => clearInterval(interval);
  }, []);

  const storageData = [
    { label: '已使用', value: 75, max: 100, color: '#4a90d9' },
    { label: '图片', value: 35, max: 100, color: '#50c878' },
    { label: '视频', value: 25, max: 100, color: '#f39c12' },
    { label: '文档', value: 10, max: 100, color: '#9b59b6' },
  ];

  return (
    <div className="progress-demo">
      <h2>进度条组件演示</h2>

      <div className="demo-section">
        <h3>基本进度条</h3>
        <ProgressBar value={65} />
        <ProgressBar value={30} color="#e74c3c" height={12} showLabel={false} />
      </div>

      <div className="demo-section">
        <h3>带条纹动画进度条</h3>
        <ProgressBar value={75} striped animated color="#4a90d9" />
      </div>

      <div className="demo-section">
        <h3>动态进度条</h3>
        <ProgressBar
          value={downloadProgress}
          color="#50c878"
          showLabel
          showPercentage
        />
        <br />
        <ProgressBar
          value={uploadProgress}
          color="#f39c12"

      <div className="demo-section">
        <h3>圆形进度条</h3>
        <div style={{ display: 'flex', gap: '30px', alignItems: 'center' }}>
          <CircularProgress value={75} size={100} color="#4a90d9" />
          <CircularProgress value={45} size={100} color="#e74c3c" />

      <div className="demo-section">
        <h3>多进度条组</h3>

        <h3>分段进度条</h3>
        <SegmentedProgressBar value={75} segments={20} />
        <br />
        <SegmentedProgressBar value={45} segments={10} activeColor="#e74c3c" />
      </div>

      <div className="demo-section">
        <h3>可控制动画进度条</h3>
        <AnimatedProgressBar duration={10000} color="#9b59b6" />
      </div>
    </div>
  );
}

export default ProgressDemo;
```

## 设计要点

1. **多种样式**：支持线性进度条、圆形进度条、分段进度条
2. **动画效果**：
   - 平滑过渡动画（`transitionDuration`）
   - 条纹动画（`striped`）
3. **自定义配置**：
   - 高度、颜色、背景色
   - 是否显示标签和百分比
   - 最大值自定义
4. **RAF 动画**：使用 `requestAnimationFrame` 实现流畅的可控进度动画
5. **百分比计算**：自动限制在 0-100 范围内
