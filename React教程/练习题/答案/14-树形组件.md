# 练习 14：树形组件

## 题目回顾

创建一个树形组件。

**要求：**
- 支持多级嵌套
- 展开/折叠节点
- 显示节点图标
- 支持选择节点
- 支持搜索节点

---

## 参考答案

### Tree.tsx

```tsx
import React, { useState, useMemo } from 'react';
import './Tree.css';

export interface TreeNode {
  key: string;
  label: string;
  icon?: React.ReactNode;
  children?: TreeNode[];
  disabled?: boolean;
  selectable?: boolean;
}

interface TreeProps {
  data: TreeNode[];
  defaultExpandedKeys?: string[];
  defaultSelectedKeys?: string[];
  onSelect?: (selectedKeys: string[], info: { node: TreeNode }) => void;
  showIcon?: boolean;
  showLine?: boolean;
  searchable?: boolean;
}

interface TreeNodeProps {
  node: TreeNode;
  depth: number;
  expandedKeys: Set<string>;
  selectedKeys: Set<string>;
  onToggle: (key: string) => void;
  onSelect: (key: string) => void;
  showIcon: boolean;
  showLine: boolean;
}

function TreeNodeComponent({
  node,
  depth,
  expandedKeys,
  selectedKeys,
  onToggle,
  onSelect,
  showIcon,
  showLine,
}: TreeNodeProps) {
  const isExpanded = expandedKeys.has(node.key);
  const isSelected = selectedKeys.has(node.key);
  const hasChildren = node.children && node.children.length > 0;

  const paddingLeft = depth * 24 + 16;

  return (
    <div className="tree-node">
      <div
        className={`tree-node__row ${isSelected ? 'tree-node__row--selected' : ''} ${node.disabled ? 'tree-node__row--disabled' : ''}`}
        style={{ paddingLeft }}
        onClick={() => {
          if (node.disabled) return;
          if (hasChildren) onToggle(node.key);
          if (node.selectable !== false) onSelect(node.key);
        }}
      >
        <span
          className={`tree-node__switcher ${hasChildren ? 'tree-node__switcher--visible' : 'tree-node__switcher--hidden'}`}
          onClick={(e) => {
            e.stopPropagation();
            if (hasChildren) onToggle(node.key);
          }}
        >
          {hasChildren ? (isExpanded ? '−' : '+') : ''}
        </span>

        {showIcon && (
          <span className="tree-node__icon">
            {node.icon || (hasChildren ? (isExpanded ? '📂' : '📁') : '📄')}
          </span>
        )}

        <span className="tree-node__label">{node.label}</span>
      </div>

      {hasChildren && isExpanded && (
        <div className="tree-node__children">
          {node.children!.map((child) => (
            <TreeNodeComponent
              key={child.key}
              node={child}
              depth={depth + 1}
              expandedKeys={expandedKeys}
              selectedKeys={selectedKeys}
              onToggle={onToggle}
              onSelect={onSelect}
              showIcon={showIcon}
              showLine={showLine}
            />
          ))}
        </div>
      )}
    </div>
  );
}

function Tree({
  data,
  defaultExpandedKeys = [],
  defaultSelectedKeys = [],
  onSelect,
  showIcon = true,
  showLine = false,
  searchable = false,
}: TreeProps) {
  const [expandedKeys, setExpandedKeys] = useState<Set<string>>(
    new Set(defaultExpandedKeys)
  );
  const [selectedKeys, setSelectedKeys] = useState<Set<string>>(
    new Set(defaultSelectedKeys)
  );
  const [searchValue, setSearchValue] = useState('');

  const handleToggle = (key: string) => {
    setExpandedKeys((prev) => {
      const next = new Set(prev);
      if (next.has(key)) {
        next.delete(key);
      } else {
        next.add(key);
      }
      return next;
    });
  };

  const handleSelect = (key: string) => {
    setSelectedKeys((prev) => {
      const next = new Set(prev);
      if (next.has(key)) {
        next.delete(key);
      } else {
        next.add(key);
      }
      return next;
    });

    const node = findNode(data, key);
    if (node && onSelect) {
      onSelect(Array.from(selectedKeys), { node });
    }
  };

  const filteredData = useMemo(() => {
    if (!searchValue) return data;

    const filterNodes = (nodes: TreeNode[]): TreeNode[] => {
      return nodes.reduce<TreeNode[]>((acc, node) => {
        const matches = node.label.toLowerCase().includes(searchValue.toLowerCase());
        const filteredChildren = filterNodes(node.children || []);

        if (matches || filteredChildren.length > 0) {
          acc.push({
            ...node,
            children: filteredChildren.length > 0 ? filteredChildren : node.children,
          });
        }
        return acc;
      }, []);
    };

    return filterNodes(data);
  }, [data, searchValue]);

  const expandAll = () => {
    const allKeys = getAllKeys(data);
    setExpandedKeys(new Set(allKeys));
  };

  const collapseAll = () => {
    setExpandedKeys(new Set());
  };

  return (
    <div className={`tree ${showLine ? 'tree--show-line' : ''}`}>
      {searchable && (
        <div className="tree__search">
          <input
            type="text"
            placeholder="搜索节点..."
            value={searchValue}
            onChange={(e) => setSearchValue(e.target.value)}
            className="tree__search-input"
          />
          <button onClick={expandAll} className="tree__btn">展开全部</button>
          <button onClick={collapseAll} className="tree__btn">折叠全部</button>
        </div>
      )}
      <div className="tree__content">
        {filteredData.length === 0 ? (
          <div className="tree__empty">没有找到匹配的节点</div>
        ) : (
          filteredData.map((node) => (
            <TreeNodeComponent
              key={node.key}
              node={node}
              depth={0}
              expandedKeys={expandedKeys}
              selectedKeys={selectedKeys}
              onToggle={handleToggle}
              onSelect={handleSelect}
              showIcon={showIcon}
              showLine={showLine}
            />
          ))
        )}
      </div>
    </div>
  );
}

function findNode(nodes: TreeNode[], key: string): TreeNode | null {
  for (const node of nodes) {
    if (node.key === key) return node;
    if (node.children) {
      const found = findNode(node.children, key);
      if (found) return found;
    }
  }
  return null;
}

function getAllKeys(nodes: TreeNode[]): string[] {
  return nodes.reduce<string[]>((acc, node) => {
    acc.push(node.key);
    if (node.children) {
      acc.push(...getAllKeys(node.children));
    }
    return acc;
  }, []);
}

export default Tree;
```

## 知识点

- 递归组件渲染树形结构
- 状态管理（展开/折叠、选中状态）
- 搜索过滤（递归过滤匹配节点及其父节点）
- 展开全部/折叠全部功能
- `useMemo` 优化搜索过滤性能
