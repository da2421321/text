# 练习 12：无限滚动列表

## 题目回顾

创建一个无限滚动列表组件。

**要求：**
- 初始加载一定数量的数据
- 滚动到底部自动加载更多
- 显示加载状态
- 显示没有更多数据
- 支持刷新

---

## 参考答案

### useInfiniteScroll.ts（自定义 Hook）

```tsx
import { useState, useCallback, useRef } from 'react';

interface UseInfiniteScrollOptions<T> {
  fetchData: (page: number) => Promise<T[]>;
  initialPage?: number;
  pageSize?: number;
}

interface UseInfiniteScrollReturn<T> {
  data: T[];
  loading: boolean;
  hasMore: boolean;
  error: string | null;
  loadMore: () => void;
  refresh: () => void;
  sentinelRef: (node: HTMLDivElement | null) => void;
}

export function useInfiniteScroll<T>({
  fetchData,
  initialPage = 1,
  pageSize = 10,
}: UseInfiniteScrollOptions<T>): UseInfiniteScrollReturn<T> {
  const [data, setData] = useState<T[]>([]);
  const [loading, setLoading] = useState(false);
  const [hasMore, setHasMore] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [page, setPage] = useState(initialPage);
  const observerRef = useRef<IntersectionObserver | null>(null);
  const loadingRef = useRef(false);

  const loadMore = useCallback(async () => {
    if (loadingRef.current || !hasMore) return;
    loadingRef.current = true;
    setLoading(true);
    setError(null);

    try {
      const newData = await fetchData(page);
      if (newData.length < pageSize) {
        setHasMore(false);
      }
      setData((prev) => [...prev, ...newData]);
      setPage((prev) => prev + 1);
    } catch (err) {
      setError(err instanceof Error ? err.message : '加载失败');
    } finally {
      setLoading(false);
      loadingRef.current = false;
    }
  }, [fetchData, page, hasMore, pageSize]);

  const refresh = useCallback(() => {
    setData([]);
    setPage(initialPage);
    setHasMore(true);
    setError(null);
  }, [initialPage]);

  const sentinelRef = useCallback(
    (node: HTMLDivElement | null) => {
      if (observerRef.current) {
        observerRef.current.disconnect();
      }
      if (!node) return;

      observerRef.current = new IntersectionObserver(
        (entries) => {
          if (entries[0].isIntersecting && hasMore && !loadingRef.current) {
            loadMore();
          }
        },
        { threshold: 0.1 }
      );
      observerRef.current.observe(node);
    },
    [loadMore, hasMore]
  );

  return { data, loading, hasMore, error, loadMore, refresh, sentinelRef };
}
```

### InfiniteList.tsx

```tsx
import React, { useCallback } from 'react';
import { useInfiniteScroll } from './useInfiniteScroll';
import './InfiniteList.css';

interface Item {
  id: number;
  title: string;
  description: string;
}

function InfiniteList() {
  const fetchItems = useCallback(async (page: number): Promise<Item[]> => {
    const response = await fetch(`/api/items?page=${page}&limit=10`);
    const data = await response.json();
    return data.items;
  }, []);

  const {
    data,
    loading,
    hasMore,
    error,
    refresh,
    sentinelRef,
  } = useInfiniteScroll<Item>({
    fetchData: fetchItems,
    pageSize: 10,
  });

  // 模拟数据（当没有真实 API 时使用）
  const mockFetch = useCallback(async (page: number): Promise<Item[]> => {
    await new Promise((r) => setTimeout(r, 800));
    if (page > 5) return [];
    return Array.from({ length: 10 }, (_, i) => ({
      id: (page - 1) * 10 + i + 1,
      title: `商品 ${(page - 1) * 10 + i + 1}`,
      description: `这是商品 ${(page - 1) * 10 + i + 1} 的描述信息`,
    }));
  }, []);

  const { data: items, loading: mockLoading, hasMore: mockHasMore, sentinelRef: mockSentinel, refresh: mockRefresh } =
    useInfiniteScroll<Item>({ fetchData: mockFetch, pageSize: 10 });

  return (
    <div className="infinite-list">
      <div className="infinite-list__header">
        <h2>无限滚动列表</h2>
        <button onClick={mockRefresh} className="infinite-list__refresh">
          刷新
        </button>
      </div>

      <div className="infinite-list__content">
        {items.map((item) => (
          <div key={item.id} className="infinite-list__item">
            <h3>{item.title}</h3>
            <p>{item.description}</p>
          </div>
        ))}

        {error && <div className="infinite-list__error">{error}</div>}

        <div ref={mockSentinel} className="infinite-list__sentinel" />
      </div>

      {mockLoading && (
        <div className="infinite-list__loading">
          <span className="spinner" />
          <span>加载中...</span>
        </div>
      )}

      {!mockHasMore && !mockLoading && items.length > 0 && (
        <div className="infinite-list__nomore">— 没有更多数据了 —</div>
      )}
    </div>
  );
}

export default InfiniteList;
```

## 知识点

- `IntersectionObserver` API 监听元素是否进入视口
- 自定义 Hook 封装数据获取逻辑
- 防重入保护（`loadingRef` 防止重复触发加载）
- 状态提升（loading、hasMore、error）
- 刷新功能需要重置分页状态
