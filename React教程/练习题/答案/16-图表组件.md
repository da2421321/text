# 练习 16：图表组件答案

## 完整代码实现

```tsx
import { useState, useRef, useEffect } from 'react';

// 类型定义
interface ChartData {
  label: string;
  value: number;
  color?: string;
}

interface ChartProps {
  data: ChartData[];
  width?: number;
  height?: number;
}

// ============ 柱状图 ============
function BarChart({ data, width = 500, height = 300 }: ChartProps) {
  const maxValue = Math.max(...data.map(d => d.value));
  const barWidth = (width - 60) / data.length;
  const chartHeight = height - 60;

  return (
    <div className="chart-container">
      <h3>柱状图</h3>
      <svg width={width} height={height}>
        {/* Y 轴 */}
        <line x1="50" y1="10" x2="50" y2={chartHeight} stroke="#333" />
        {/* X 轴 */}
        <line x1="50" y1={chartHeight} x2={width - 10} y2={chartHeight} stroke="#333" />

        {data.map((item, index) => {
          const barHeight = (item.value / maxValue) * (chartHeight - 20);
          const x = 55 + index * barWidth;
          const y = chartHeight - barHeight;

          return (
            <g key={index}>
              <rect
                x={x}
                y={y}
                width={barWidth - 10}
                height={barHeight}
                fill={item.color || '#4a90d9'}
                rx="3"
                className="chart-bar"
              >
                <title>{item.label}: {item.value}</title>
              </rect>
              <text
                x={x + (barWidth - 10) / 2}
                y={chartHeight + 20}
                textAnchor="middle"
                fontSize="12"
              >
                {item.label}
              </text>
              <text
                x={x + (barWidth - 10) / 2}
                y={y - 5}
                textAnchor="middle"
                fontSize="11"
              >
                {item.value}
              </text>
            </g>
          );
        })}
      </svg>
    </div>
  );
}

// ============ 折线图 ============
function LineChart({ data, width = 500, height = 300 }: ChartProps) {
  const maxValue = Math.max(...data.map(d => d.value));
  const minValue = Math.min(...data.map(d => d.value));
  const chartWidth = width - 60;
  const chartHeight = height - 60;
  const padding = 30;

  const points = data.map((item, index) => {
    const x = padding + (index / (data.length - 1)) * chartWidth;
    const y = chartHeight - ((item.value - minValue) / (maxValue - minValue || 1)) * (chartHeight - padding);
    return { x, y };
  });

  const pathD = points.map((p, i) =>
    i === 0 ? `M ${p.x} ${p.y}` : `L ${p.x} ${p.y}`
  ).join(' ');

  const areaD = `${pathD} L ${points[points.length - 1].x} ${chartHeight} L ${points[0].x} ${chartHeight} Z`;

  return (
    <div className="chart-container">
      <h3>折线图</h3>
      <svg width={width} height={height}>
        {/* Y 轴 */}
        <line x1={padding} y1={padding} x2={padding} y2={chartHeight} stroke="#333" />
        {/* X 轴 */}
        <line x1={padding} y1={chartHeight} x2={width - 10} y2={chartHeight} stroke="#333" />

        {/* 面积填充 */}
        <path d={areaD} fill="rgba(74, 144, 217, 0.2)" />

        {/* 折线 */}
        <path d={pathD} fill="none" stroke="#4a90d9" strokeWidth="2" />

        {/* 数据点 */}
        {points.map((p, index) => (
          <g key={index}>
            <circle cx={p.x} cy={p.y} r="4" fill="#4a90d9" />
            <text x={p.x} y={chartHeight + 20} textAnchor="middle" fontSize="12">
              {data[index].label}
            </text>
            <text x={p.x} y={p.y - 10} textAnchor="middle" fontSize="11">
              {data[index].value}
            </text>
          </g>
        ))}
      </svg>
    </div>
  );
}

// ============ 饼图 ============
function PieChart({ data, width = 400, height = 400 }: ChartProps) {
  const total = data.reduce((sum, item) => sum + item.value, 0);
  const radius = Math.min(width, height) / 2 - 20;
  const centerX = width / 2;
  const centerY = height / 2;

  const colors = ['#4a90d9', '#50c878', '#e74c3c', '#f39c12', '#9b59b6', '#1abc9c'];

  let currentAngle = -90;

  const slices = data.map((item, index) => {
    const angle = (item.value / total) * 360;
    const startAngle = currentAngle;
    const endAngle = currentAngle + angle;
    currentAngle = endAngle;

    const startRad = (startAngle * Math.PI) / 180;
    const endRad = (endAngle * Math.PI) / 180;

    const x1 = centerX + radius * Math.cos(startRad);
    const y1 = centerY + radius * Math.sin(startRad);
    const x2 = centerX + radius * Math.cos(endRad);
    const y2 = centerY + radius * Math.sin(endRad);

    const largeArc = angle > 180 ? 1 : 0;

    const pathD = [
      `M ${centerX} ${centerY}`,
      `L ${x1} ${y1}`,
      `A ${radius} ${radius} 0 ${largeArc} 1 ${x2} ${y2}`,
      'Z'
    ].join(' ');

    const labelRad = ((startAngle + angle / 2) * Math.PI) / 180;
    const labelRadius = radius * 0.6;
    const labelX = centerX + labelRadius * Math.cos(labelRad);
    const labelY = centerY + labelRadius * Math.sin(labelRad);

    const percentage = ((item.value / total) * 100).toFixed(1);

    return (
      <g key={index}>
        <path
          d={pathD}
          fill={item.color || colors[index % colors.length]}
          className="chart-slice"
        >
          <title>{item.label}: {item.value} ({percentage}%)</title>
        </path>
        <text
          x={labelX}
          y={labelY}
          textAnchor="middle"
          fontSize="12"
          fill="#fff"
          fontWeight="bold"
        >
          {percentage}%
        </text>
      </g>
    );
  });

  return (
    <div className="chart-container">
      <h3>饼图</h3>
      <svg width={width} height={height}>
        {slices}
      </svg>
      <div className="chart-legend">
        {data.map((item, index) => (
          <div key={index} className="legend-item">
            <span
              className="legend-color"
              style={{ backgroundColor: item.color || colors[index % colors.length] }}
            />
            <span className="legend-label">{item.label}</span>
            <span className="legend-value">{item.value}</span>
          </div>
        ))}
      </div>
    </div>
  );
}

// ============ 动态数据更新示例 ============
function AnimatedChart({ data, width = 500, height = 300 }: ChartProps) {
  const [animatedData, setAnimatedData] = useState<ChartData[]>([]);
  const [show, setShow] = useState(false);

  useEffect(() => {
    if (show) {
      setAnimatedData([]);
      data.forEach((item, index) => {
        setTimeout(() => {
          setAnimatedData(prev => [...prev, item]);
        }, index * 300);
      });
    }
  }, [show, data]);

  const maxValue = Math.max(...data.map(d => d.value), 1);
  const barWidth = (width - 60) / data.length;
  const chartHeight = height - 60;

  return (
    <div className="chart-container">
      <h3>带动画的柱状图</h3>
      <button onClick={() => setShow(!show)}>
        {show ? '重置' : '开始动画'}
      </button>
      <svg width={width} height={height}>
        {animatedData.map((item, index) => {
          const barHeight = (item.value / maxValue) * (chartHeight - 20);
          const x = 55 + index * barWidth;
          const y = chartHeight - barHeight;

          return (
            <g key={index}>
              <rect
                x={x}
                y={y}
                width={barWidth - 10}
                height={barHeight}
                fill={item.color || '#4a90d9'}
                rx="3"
              >
                <animate
                  attributeName="height"
                  from="0"
                  to={barHeight}
                  dur="0.5s"
                  fill="freeze"
                />
                <animate
                  attributeName="y"
                  from={chartHeight}
                  to={y}
                  dur="0.5s"
                  fill="freeze"
                />
              </rect>
              <text x={x + (barWidth - 10) / 2} y={chartHeight + 20} textAnchor="middle" fontSize="12">
                {item.label}
              </text>
            </g>
          );
        })}
      </svg>
    </div>
  );
}

// ============ 图表组件使用示例 ============
function ChartDemo() {
  const [chartData] = useState<ChartData[]>([
    { label: '一月', value: 120, color: '#4a90d9' },
    { label: '二月', value: 80, color: '#50c878' },
    { label: '三月', value: 150, color: '#e74c3c' },
    { label: '四月', value: 200, color: '#f39c12' },
    { label: '五月', value: 170, color: '#9b59b6' },
    { label: '六月', value: 220, color: '#1abc9c' },
  ]);

  const pieData: ChartData[] = [
    { label: '电子产品', value: 35 },
    { label: '服装', value: 25 },
    { label: '食品', value: 20 },
    { label: '图书', value: 12 },
    { label: '其他', value: 8 },
  ];

  return (
    <div className="chart-demo">
      <h2>图表组件示例</h2>

      <BarChart data={chartData} />
      <LineChart data={chartData} />
      <PieChart data={pieData} />
      <AnimatedChart data={chartData} />
    </div>
  );
}

export default ChartDemo;
```

## 设计要点

1. **SVG 图表**：使用原生 SVG 绘制图表，无需第三方库
2. **响应式**：支持自定义宽高参数
3. **动画效果**：使用 SVG `<animate>` 标签实现数据更新动画
4. **可访问性**：使用 `<title>` 元素提供悬停提示
5. **类型安全**：使用 TypeScript 定义完整的类型接口
