# 练习 18：通知组件答案

## 完整代码实现

```tsx
import { useState, useEffect, useCallback, createContext, useContext } from 'react';

// ============ 类型定义 ============
type NotificationType = 'success' | 'error' | 'warning' | 'info';

interface Notification {
  id: string;
  type: NotificationType;
  message: string;
  duration?: number;
}

interface Toast {
  id: string;
  type: NotificationType;
  message: string;
  createdAt: number;
}

interface NotificationContextValue {
  addToast: (message: string, type?: NotificationType, duration?: number) => void;
  removeToast: (id: string) => void;
}

// ============ 样式配置 ============
const toastStyles: Record<NotificationType, { bg: string; border: string; icon: string }> = {
  success: { bg: '#d4edda', border: '#c3e6cb', icon: '✓' },
  error: { bg: '#f8d7da', border: '#f5c6cb', icon: '✕' },
  warning: { bg: '#fff3cd', border: '#ffeeba', icon: '!' },
  info: { bg: '#d1ecf1', border: '#bee5eb', icon: 'i' },
};

// ============ Toast 项组件 ============
function ToastItem({
  toast,
  onRemove,
}: {
  toast: Toast;
  onRemove: (id: string) => void;
}) {
  const style = toastStyles[toast.type];
  const [isExiting, setIsExiting] = useState(false);

  useEffect(() => {
    const timer = setTimeout(() => {
      setIsExiting(true);
      setTimeout(() => onRemove(toast.id), 300);
    }, 3000);

    return () => clearTimeout(timer);
  }, [toast.id, onRemove]);

  const handleClose = () => {
    setIsExiting(true);
    setTimeout(() => onRemove(toast.id), 300);
  };

  return (
    <div
      className={`toast-item ${isExiting ? 'exiting' : ''}`}
      style={{
        backgroundColor: style.bg,
        borderColor: style.border,
        animation: isExiting ? 'slideOut 0.3s ease-out' : 'slideIn 0.3s ease-out',
      }}
    >
      <span className="toast-icon" style={{ color: style.border.replace('#', '#') }}>
        {style.icon}
      </span>
      <span className="toast-message">{toast.message}</span>
      <button className="toast-close" onClick={handleClose}>
        ×
      </button>
    </div>
  );
}

// ============ Toast 容器组件 ============
function ToastContainer({
  toasts,
  position = 'top-right',
  onRemove,
}: {
  toasts: Toast[];
  position?: 'top-right' | 'top-left' | 'bottom-right' | 'bottom-left' | 'top-center' | 'bottom-center';
  onRemove: (id: string) => void;
}) {
  const positionStyles: Record<string, React.CSSProperties> = {
    'top-right': { top: '20px', right: '20px' },
    'top-left': { top: '20px', left: '20px' },
    'bottom-right': { bottom: '20px', right: '20px' },
    'bottom-left': { bottom: '20px', left: '20px' },
    'top-center': { top: '20px', left: '50%', transform: 'translateX(-50%)' },
    'bottom-center': { bottom: '20px', left: '50%', transform: 'translateX(-50%)' },
  };

  return (
    <div className="toast-container" style={positionStyles[position]}>
      {toasts.map(toast => (
        <ToastItem key={toast.id} toast={toast} onRemove={onRemove} />
      ))}
    </div>
  );
}

// ============ Toast Context ============
const ToastContext = createContext<NotificationContextValue | null>(null);

export function ToastProvider({ children }: { children: React.ReactNode }) {
  const [toasts, setToasts] = useState<Toast[]>([]);

  const addToast = useCallback((
    message: string,
    type: NotificationType = 'info',
    duration = 3000
  ) => {
    const id = `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
    const toast: Toast = {
      id,
      type,
      message,
      createdAt: Date.now(),
    };

    setToasts(prev => [...prev, toast]);

    if (duration > 0) {
      setTimeout(() => {
        setToasts(prev => prev.filter(t => t.id !== id));
      }, duration);
    }
  }, []);

  const removeToast = useCallback((id: string) => {
    setToasts(prev => prev.filter(t => t.id !== id));
  }, []);

  return (
    <ToastContext.Provider value={{ addToast, removeToast }}>
      {children}
      <ToastContainer toasts={toasts} position="top-right" onRemove={removeToast} />
    </ToastContext.Provider>
  );
}

export function useToast() {
  const context = useContext(ToastContext);
  if (!context) {
    throw new Error('useToast must be used within ToastProvider');
  }
  return context;
}

// ============ 通知确认框组件 ============
interface ConfirmOptions {
  title?: string;
  message: string;
  confirmText?: string;
  cancelText?: string;
  type?: NotificationType;
}

function ConfirmDialog({
  options,
  onConfirm,
  onCancel,
}: {
  options: ConfirmOptions;
  onConfirm: () => void;
  onCancel: () => void;
}) {
  const style = toastStyles[options.type || 'info'];

  return (
    <div className="confirm-overlay">
      <div className="confirm-dialog">
        <div className="confirm-header" style={{ borderBottomColor: style.border }}>
          <h3>{options.title || '确认'}</h3>
        </div>
        <div className="confirm-body">
          <p>{options.message}</p>
        </div>
        <div className="confirm-footer">
          <button className="btn-cancel" onClick={onCancel}>
            {options.cancelText || '取消'}
          </button>
          <button
            className="btn-confirm"
            style={{ backgroundColor: style.border.replace('#', '#') }}
            onClick={onConfirm}
          >
            {options.confirmText || '确认'}
          </button>
        </div>
      </div>
    </div>
  );
}

// ============ useConfirm Hook ============
function useConfirm() {
  const [dialog, setDialog] = useState<{
    isOpen: boolean;
    options: ConfirmOptions | null;
    resolve: ((value: boolean) => void) | null;
  }>({ isOpen: false, options: null, resolve: null });

  const confirm = useCallback((options: ConfirmOptions): Promise<boolean> => {
    return new Promise(resolve => {
      setDialog({ isOpen: true, options, resolve });
    });
  }, []);

  const handleConfirm = () => {
    dialog.resolve?.(true);
    setDialog({ isOpen: false, options: null, resolve: null });
  };

  const handleCancel = () => {
    dialog.resolve?.(false);
    setDialog({ isOpen: false, options: null, resolve: null });
  };

  return {
    confirm,
    dialog: dialog.isOpen && dialog.options ? (
      <ConfirmDialog
        options={dialog.options}
        onConfirm={handleConfirm}
        onCancel={handleCancel}
      />
    ) : null,
  };
}

// ============ 完整通知系统 ============
function NotificationSystem() {
  const { addToast } = useToast();
  const { confirm, dialog } = useConfirm();

  const handleSuccess = () => addToast('操作成功完成！', 'success');
  const handleError = () => addToast('发生了错误，请重试。', 'error');
  const handleWarning = () => addToast('您的会话即将过期，请保存工作。', 'warning');
  const handleInfo = () => addToast('新版本已发布，请更新。', 'info');

  const handleConfirm = async () => {
    const result = await confirm({
      title: '确认删除',
      message: '确定要删除这个项目吗？此操作无法撤销。',
      confirmText: '删除',
      cancelText: '取消',
      type: 'error',
    });
    if (result) {
      addToast('项目已删除', 'success');
    } else {
      addToast('已取消删除', 'info');
    }
  };

  const handleBatchActions = () => {
    addToast('正在处理您的请求...', 'info', 5000);
    setTimeout(() => addToast('处理完成！', 'success', 3000), 1500);
    setTimeout(() => addToast('有一项操作失败', 'error', 5000), 3000);
  };

  return (
    <div className="notification-demo">
      <h2>通知系统演示</h2>

      <div className="demo-section">
        <h3>基本通知</h3>
        <div className="button-group">
          <button className="btn-success" onClick={handleSuccess}>成功</button>
          <button className="btn-error" onClick={handleError}>错误</button>
          <button className="btn-warning" onClick={handleWarning}>警告</button>
          <button className="btn-info" onClick={handleInfo}>信息</button>
        </div>
      </div>

      <div className="demo-section">
        <h3>确认对话框</h3>
        <button className="btn-primary" onClick={handleConfirm}>打开确认框</button>
      </div>

      <div className="demo-section">
        <h3>批量通知</h3>
        <button className="btn-primary" onClick={handleBatchActions}>模拟批量操作</button>
      </div>

      {dialog}
    </div>
  );
}

// ============ 使用示例 ============
function App() {
  return (
    <ToastProvider>
      <NotificationSystem />
    </ToastProvider>
  );
}

export default App;
```

## 设计要点

1. **Context API**：使用 Context + Provider 模式全局管理通知状态
2. **自定义 Hook**：`useToast` 提供简洁的 API，`useConfirm` 返回 Promise
3. **动画效果**：入场和退场动画，提升用户体验
4. **自动关闭**：通知默认 3 秒后自动消失，支持自定义时长
5. **多位置支持**：支持 6 种位置：四角 + 上下居中
6. **确认对话框**：`useConfirm` hook 返回 Promise，支持异步确认
7. **批量通知**：可同时显示多个通知
