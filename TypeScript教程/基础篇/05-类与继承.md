## 基本类定义

### 访问修饰符

```typescript
class BankAccount {
    public accountNumber: string;      // 公共属性（默认）
    private balance: number;            // 私有属性
    protected owner: string;            // 受保护属性
    
    constructor(accountNumber: string, owner: string, initialBalance: number) {
        this.accountNumber = accountNumber;
        this.owner = owner;
        this.balance = initialBalance;
    }
    
    // 公共方法
    public deposit(amount: number): void {
        if (amount > 0) {
            this.balance += amount;
            console.log(`存款 ${amount}，当前余额：${this.balance}`);
        }
    }
    
    public withdraw(amount: number): boolean {
        if (this.canWithdraw(amount)) {
            this.balance -= amount;
            console.log(`取款 ${amount}，当前余额：${this.balance}`);
            return true;
        }
        return false;
    }
    
    public getBalance(): number {
        return this.balance;
    }
    
    // 私有方法
    private canWithdraw(amount: number): boolean {
        return amount > 0 && amount <= this.balance;
    }
}

const account = new BankAccount("123456", "Alice", 1000);
account.deposit(500);
account.withdraw(300);
console.log("余额:", account.getBalance());
// console.log(account.balance); // Error: 私有属性
```

## 继承

### 基本继承

```typescript
class Animal {
    name: string;
    
    constructor(name: string) {
        this.name = name;
    }
    
    move(distance: number = 0): void {
        console.log(`${this.name} moved ${distance}m.`);
    }
}

class Dog extends Animal {
    bark(): void {
        console.log("Woof! Woof!");
    }
}

class Bird extends Animal {
    fly(distance: number): void {
        console.log(`${this.name} flew ${distance}m.`);
    }
}

const dog = new Dog("Buddy");
dog.bark();
dog.move(10);

const bird = new Bird("Tweety");
bird.fly(100);
bird.move(5);
```

### super 关键字

```typescript
class Vehicle {
    constructor(public brand: string, protected speed: number) {}
    
    describe(): string {
        return `${this.brand} vehicle`;
    }
    
    move(): void {
        console.log(`Moving at ${this.speed} km/h`);
    }
}

class Car extends Vehicle {
    constructor(brand: string, speed: number, public doors: number) {
        super(brand, speed); // 调用父类构造函数
    }
    
    describe(): string {
        return `${super.describe()} with ${this.doors} doors`; // 调用父类方法
    }
    
    accelerate(increment: number): void {
        this.speed += increment;
        console.log(`Accelerated to ${this.speed} km/h`);
    }
}

const car = new Car("Toyota", 60, 4);
console.log(car.describe());
car.move();
car.accelerate(20);
car.move();
```

## 抽象类

```typescript
abstract class Shape {
    constructor(public color: string) {}
    
    // 抽象方法（子类必须实现）
    abstract getArea(): number;
    abstract getPerimeter(): number;
    
    // 具体方法（子类可以继承）
    describe(): string {
        return `A ${this.color} shape with area ${this.getArea()}`;
    }
}

class Rectangle extends Shape {
    constructor(color: string, private width: number, private height: number) {
        super(color);
    }
    
    getArea(): number {
        return this.width * this.height;
    }
    
    getPerimeter(): number {
        return 2 * (this.width + this.height);
    }
}

class Circle extends Shape {
    constructor(color: string, private radius: number) {
        super(color);
    }
    
    getArea(): number {
        return Math.PI * this.radius ** 2;
    }
    
    getPerimeter(): number {
        return 2 * Math.PI * this.radius;
    }
}

// const shape = new Shape("red"); // Error: 不能实例化抽象类

const rectangle = new Rectangle("blue", 10, 5);
console.log(rectangle.describe());
console.log("周长:", rectangle.getPerimeter());

const circle = new Circle("red", 7);
console.log(circle.describe());
console.log("周长:", circle.getPerimeter().toFixed(2));
```

## Getter 和 Setter

```typescript
class Employee {
    private _salary: number;
    
    constructor(
        public firstName: string,
        public lastName: string,
        salary: number
    ) {
        this._salary = salary;
    }
    
    // Getter
    get fullName(): string {
        return `${this.firstName} ${this.lastName}`;
    }
    
    get salary(): number {
        return this._salary;
    }
    
    // Setter
    set salary(newSalary: number) {
        if (newSalary < 0) {
            throw new Error("薪资不能为负数");
        }
        this._salary = newSalary;
    }
}

const employee = new Employee("John", "Doe", 50000);
console.log(employee.fullName); // 使用 getter
console.log("薪资:", employee.salary);

employee.salary = 60000; // 使用 setter
console.log("新薪资:", employee.salary);

// employee.salary = -1000; // Error: 抛出异常
```

## 实践示例：任务管理系统

```typescript
// src/task-manager.ts

/**
 * 任务管理系统
 */

// 任务状态枚举
enum TaskStatus {
    TODO = "待办",
    IN_PROGRESS = "进行中",
    DONE = "已完成",
    CANCELLED = "已取消"
}

// 任务优先级
enum Priority {
    LOW = 1,
    MEDIUM = 2,
    HIGH = 3,
    URGENT = 4
}

// 基础任务类
abstract class BaseTask {
    private static idCounter: number = 1;
    
    public readonly id: number;
    public readonly createdAt: Date;
    private _status: TaskStatus;
    
    constructor(
        public title: string,
        public description: string,
        public priority: Priority = Priority.MEDIUM
    ) {
        this.id = BaseTask.idCounter++;
        this.createdAt = new Date();
        this._status = TaskStatus.TODO;
    }
    
    get status(): TaskStatus {
        return this._status;
    }
    
    set status(newStatus: TaskStatus) {
        console.log(`任务 #${this.id} 状态变更: ${this._status} -> ${newStatus}`);
        this._status = newStatus;
    }
    
    abstract getTaskType(): string;
    
    getInfo(): string {
        return `[${this.getTaskType()}] #${this.id}: ${this.title} (${this._status})`;
    }
}

// 个人任务
class PersonalTask extends BaseTask {
    constructor(
        title: string,
        description: string,
        priority: Priority,
        public assignee: string
    ) {
        super(title, description, priority);
    }
    
    getTaskType(): string {
        return "个人任务";
    }
    
    getInfo(): string {
        return `${super.getInfo()} - 负责人: ${this.assignee}`;
    }
}

// 团队任务
class TeamTask extends BaseTask {
    private members: Set<string> = new Set();
    
    constructor(
        title: string,
        description: string,
        priority: Priority,
        public teamName: string
    ) {
        super(title, description, priority);
    }
    
    addMember(member: string): void {
        this.members.add(member);
        console.log(`添加成员 ${member} 到任务 #${this.id}`);
    }
    
    removeMember(member: string): void {
        this.members.delete(member);
        console.log(`从任务 #${this.id} 移除成员 ${member}`);
    }
    
    getMembers(): string[] {
        return Array.from(this.members);
    }
    
    getTaskType(): string {
        return "团队任务";
    }
    
    getInfo(): string {
        return `${super.getInfo()} - 团队: ${this.teamName} (${this.members.size} 成员)`;
    }
}

// 任务管理器
class TaskManager {
    private tasks: Map<number, BaseTask> = new Map();
    
    addTask(task: BaseTask): void {
        this.tasks.set(task.id, task);
        console.log(`添加任务: ${task.getInfo()}`);
    }
    
    getTask(id: number): BaseTask | undefined {
        return this.tasks.get(id);
    }
    
    getAllTasks(): BaseTask[] {
        return Array.from(this.tasks.values());
    }
    
    getTasksByStatus(status: TaskStatus): BaseTask[] {
        return this.getAllTasks().filter(task => task.status === status);
    }
    
    getTasksByPriority(priority: Priority): BaseTask[] {
        return this.getAllTasks().filter(task => task.priority === priority);
    }
    
    updateTaskStatus(id: number, newStatus: TaskStatus): boolean {
        const task = this.tasks.get(id);
        if (task) {
            task.status = newStatus;
            return true;
        }
        return false;
    }
    
    deleteTask(id: number): boolean {
        return this.tasks.delete(id);
    }
    
    printAllTasks(): void {
        console.log("\n===== 所有任务 =====");
        this.getAllTasks().forEach(task => {
            console.log(task.getInfo());
        });
    }
    
    getStatistics(): void {
        console.log("\n===== 任务统计 =====");
        console.log(`总任务数: ${this.tasks.size}`);
        console.log(`待办: ${this.getTasksByStatus(TaskStatus.TODO).length}`);
        console.log(`进行中: ${this.getTasksByStatus(TaskStatus.IN_PROGRESS).length}`);
        console.log(`已完成: ${this.getTasksByStatus(TaskStatus.DONE).length}`);
        console.log(`已取消: ${this.getTasksByStatus(TaskStatus.CANCELLED).length}`);
    }
}

// 使用示例
const manager = new TaskManager();

// 创建个人任务
const task1 = new PersonalTask(
    "完成TypeScript教程",
    "学习TypeScript类和继承",
    Priority.HIGH,
    "张三"
);
manager.addTask(task1);

const task2 = new PersonalTask(
    "代码审查",
    "审查项目代码",
    Priority.MEDIUM,
    "李四"
);
manager.addTask(task2);

// 创建团队任务
const task3 = new TeamTask(
    "开发新功能",
    "开发用户认证功能",
    Priority.URGENT,
    "开发团队"
);
task3.addMember("张三");
task3.addMember("李四");
task3.addMember("王五");
manager.addTask(task3);

// 更新任务状态
manager.updateTaskStatus(1, TaskStatus.IN_PROGRESS);
manager.updateTaskStatus(2, TaskStatus.DONE);
manager.updateTaskStatus(3, TaskStatus.IN_PROGRESS);

// 打印所有任务
manager.printAllTasks();

// 打印统计
manager.getStatistics();

// 获取高优先级任务
console.log("\n===== 高优先级任务 =====");
manager.getTasksByPriority(Priority.HIGH).forEach(task => {
    console.log(task.getInfo());
});
```
