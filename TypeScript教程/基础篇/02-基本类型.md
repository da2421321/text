# 02 - 基本类型与类型注解

TypeScript 提供了丰富的类型系统，帮助我们在编译时捕获错误。

## 基本类型

### 1. 布尔类型（boolean）

```typescript
let isDone: boolean = false;
let isActive: boolean = true;

// 错误示例
// let flag: boolean = 1; // Error: Type 'number' is not assignable to type 'boolean'
```

### 2. 数字类型（number）

```typescript
let decimal: number = 6;
let hex: number = 0xf00d;
let binary: number = 0b1010;
let octal: number = 0o744;
let big: bigint = 100n;
```

### 3. 字符串类型（string）

```typescript
let color: string = "blue";
let fullName: string = `Bob Smith`;
let age: number = 37;
let sentence: string = `Hello, my name is ${fullName}. I'll be ${age + 1} years old next month.`;
```

### 4. 数组类型（Array）

```typescript
// 方式一：类型 + []
let list1: number[] = [1, 2, 3];
let names: string[] = ["Alice", "Bob", "Charlie"];

// 方式二：泛型数组
let list2: Array<number> = [1, 2, 3];
let items: Array<string> = ["item1", "item2"];

// 只读数组
let readonlyList: ReadonlyArray<number> = [1, 2, 3];
// readonlyList[0] = 4; // Error: Index signature in type 'readonly number[]' only permits reading
```

### 5. 元组类型（Tuple）

元组允许表示一个已知元素数量和类型的数组。

```typescript
// 声明元组
let user: [string, number];
user = ["Alice", 25]; // 正确
// user = [25, "Alice"]; // Error: 类型不匹配

// 访问元组元素
console.log(user[0]); // "Alice"
console.log(user[1]); // 25

// 解构元组
let [username, userAge] = user;

// 可选元素
let optionalTuple: [string, number?];
optionalTuple = ["Bob"];
optionalTuple = ["Bob", 30];

// 剩余元素
let restTuple: [string, ...number[]];
restTuple = ["Alice", 1, 2, 3, 4];
```

### 6. 枚举类型（Enum）

```typescript
// 数字枚举
enum Direction {
    Up = 1,
    Down,
    Left,
    Right
}

let dir: Direction = Direction.Up;
console.log(dir); // 1
console.log(Direction[2]); // "Down"

// 字符串枚举
enum Color {
    Red = "RED",
    Green = "GREEN",
    Blue = "BLUE"
}

let myColor: Color = Color.Red;
console.log(myColor); // "RED"

// 常量枚举（编译时会被内联）
const enum Status {
    Active,
    Inactive,
    Pending
}

let status: Status = Status.Active;
```

### 7. Any 类型

```typescript
let notSure: any = 4;
notSure = "maybe a string";
notSure = false; // 都可以

// any 类型的数组
let mixedList: any[] = [1, true, "free"];
mixedList[1] = 100;

// 避免过度使用 any，它会失去类型检查的优势
```

### 8. Unknown 类型

```typescript
let value: unknown;

value = true;
value = 42;
value = "Hello";
value = [];

// unknown 比 any 更安全，使用前必须进行类型检查
// let value1: boolean = value; // Error

// 正确的使用方式
if (typeof value === "string") {
    console.log(value.toUpperCase()); // 类型收窄后可以安全使用
}
```

### 9. Void 类型

```typescript
// 函数没有返回值
function warnUser(): void {
    console.log("This is a warning message");
}

// void 类型的变量只能赋值 undefined 或 null（非严格模式）
let unusable: void = undefined;
```

### 10. Null 和 Undefined

```typescript
let u: undefined = undefined;
let n: null = null;

// 在严格模式下（strictNullChecks），null 和 undefined 只能赋值给它们各自的类型
let num: number = 10;
// num = null; // Error in strict mode
// num = undefined; // Error in strict mode

// 可选类型
let optionalNum: number | undefined;
optionalNum = 10;
optionalNum = undefined;
```

### 11. Never 类型

```typescript
// 永远不会有返回值的函数
function error(message: string): never {
    throw new Error(message);
}

// 推断返回值为 never
function fail() {
    return error("Something failed");
}

// 无限循环
function infiniteLoop(): never {
    while (true) {
        // ...
    }
}
```

### 12. Object 类型

```typescript
// object 表示非原始类型
declare function create(o: object | null): void;

create({ prop: 0 }); // OK
create(null); // OK

// create(42); // Error
// create("string"); // Error
// create(false); // Error
```

## 类型断言

类型断言允许你告诉编译器"相信我，我知道我在做什么"。

```typescript
// 方式一：尖括号语法
let someValue: any = "this is a string";
let strLength1: number = (<string>someValue).length;

// 方式二：as 语法（推荐，在 JSX 中只能用这种方式）
let strLength2: number = (someValue as string).length;

// 非空断言
function process(text: string | null) {
    // 确定 text 不为 null
    let length: number = text!.length;
}
```

## 类型推断

TypeScript 会自动推断变量的类型。

```typescript
// 类型推断
let num = 10; // 推断为 number
let str = "hello"; // 推断为 string
let flag = true; // 推断为 boolean

// num = "test"; // Error

// 最佳通用类型推断
let arr = [0, 1, null]; // 推断为 (number | null)[]

// 上下文类型推断
window.onmousedown = function(mouseEvent) {
    // mouseEvent 自动推断为 MouseEvent 类型
    console.log(mouseEvent.button);
};
```

## 联合类型和交叉类型

```typescript
// 联合类型（|）：可以是多个类型之一
let value: string | number;
value = "hello";
value = 42;

function padLeft(value: string, padding: string | number) {
    if (typeof padding === "number") {
        return Array(padding + 1).join(" ") + value;
    }
    if (typeof padding === "string") {
        return padding + value;
    }
    throw new Error(`Expected string or number, got '${padding}'.`);
}

// 交叉类型（&）：合并多个类型
interface Person {
    name: string;
}

interface Employee {
    employeeId: number;
}

type Staff = Person & Employee;

let staff: Staff = {
    name: "Alice",
    employeeId: 1001
};
```

## 字面量类型

```typescript
// 字符串字面量类型
type Direction = "north" | "south" | "east" | "west";

function move(direction: Direction) {
    console.log(`Moving ${direction}`);
}

move("north"); // OK
// move("up"); // Error

// 数字字面量类型
type DiceRoll = 1 | 2 | 3 | 4 | 5 | 6;

function rollDice(): DiceRoll {
    return (Math.floor(Math.random() * 6) + 1) as DiceRoll;
}

// 布尔字面量类型
type Yes = true;
let agree: Yes = true;
```

## 实践示例

```typescript
// src/types-demo.ts

// 用户信息
interface UserInfo {
    id: number;
    username: string;
    email: string;
    age?: number; // 可选属性
    tags: string[];
}

// 处理用户数据
function processUser(user: UserInfo): string {
    let info = `User ${user.username} (ID: ${user.id})`;
    
    if (user.age !== undefined) {
        info += `, Age: ${user.age}`;
    }
    
    if (user.tags.length > 0) {
        info += `, Tags: ${user.tags.join(", ")}`;
    }
    
    return info;
}

// 测试
const user1: UserInfo = {
    id: 1,
    username: "alice",
    email: "alice@example.com",
    age: 25,
    tags: ["developer", "typescript"]
};

const user2: UserInfo = {
    id: 2,
    username: "bob",
    email: "bob@example.com",
    tags: []
};

console.log(processUser(user1));
console.log(processUser(user2));

// 状态管理
type Status = "pending" | "success" | "error";

interface ApiResponse<T> {
    status: Status;
    data?: T;
    error?: string;
}

function handleResponse<T>(response: ApiResponse<T>): void {
    switch (response.status) {
        case "success":
            console.log("Success:", response.data);
            break;
        case "error":
            console.error("Error:", response.error);
            break;
        case "pending":
            console.log("Loading...");
            break;
    }
}

// 使用示例
const successResponse: ApiResponse<UserInfo> = {
    status: "success",
    data: user1
};

handleResponse(successResponse);
```
