# 03 - 接口与类型别名

接口（Interface）和类型别名（Type Alias）是 TypeScript 中定义自定义类型的两种主要方式。

## 接口（Interface）

### 基本接口

```typescript
// 定义对象的形状
interface Person {
    firstName: string;
    lastName: string;
    age: number;
}

function greet(person: Person): string {
    return `Hello, ${person.firstName} ${person.lastName}!`;
}

const user: Person = {
    firstName: "John",
    lastName: "Doe",
    age: 30
};

console.log(greet(user));
```

### 可选属性

```typescript
interface Config {
    host: string;
    port: number;
    username?: string; // 可选
    password?: string; // 可选
}

function connectDatabase(config: Config): void {
    console.log(`Connecting to ${config.host}:${config.port}`);
    
    if (config.username) {
        console.log(`Username: ${config.username}`);
    }
}

// 可以不提供可选属性
connectDatabase({ host: "localhost", port: 3306 });

// 也可以提供
connectDatabase({
    host: "localhost",
    port: 3306,
    username: "admin",
    password: "secret"
});
```

### 只读属性

```typescript
interface Point {
    readonly x: number;
    readonly y: number;
}

let p1: Point = { x: 10, y: 20 };
// p1.x = 5; // Error: Cannot assign to 'x' because it is a read-only property

// ReadonlyArray
let arr: ReadonlyArray<number> = [1, 2, 3, 4];
// arr[0] = 12; // Error
// arr.push(5); // Error
```

### 额外的属性检查

```typescript
interface SquareConfig {
    color?: string;
    width?: number;
}

function createSquare(config: SquareConfig): { color: string; area: number } {
    return {
        color: config.color || "red",
        area: (config.width || 10) ** 2
    };
}

// 字面量对象会进行额外的属性检查
// let mySquare = createSquare({ colour: "red", width: 100 }); // Error: 'colour' 拼写错误

// 解决方案1：类型断言
let mySquare1 = createSquare({ colour: "red", width: 100 } as SquareConfig);

// 解决方案2：添加字符串索引签名
interface SquareConfigWithIndex {
    color?: string;
    width?: number;
    [propName: string]: any;
}

// 解决方案3：将对象赋值给变量
let squareOptions = { colour: "red", width: 100 };
let mySquare2 = createSquare(squareOptions);
```

### 函数类型接口

```typescript
// 定义函数类型
interface SearchFunc {
    (source: string, subString: string): boolean;
}

let mySearch: SearchFunc;

mySearch = function(source: string, subString: string): boolean {
    return source.indexOf(subString) !== -1;
};

// 参数名可以不同，但类型必须匹配
mySearch = function(src: string, sub: string): boolean {
    return src.indexOf(sub) !== -1;
};

console.log(mySearch("Hello World", "World")); // true
```

### 可索引类型

```typescript
// 字符串索引
interface StringArray {
    [index: number]: string;
}

let myArray: StringArray = ["Bob", "Alice"];
let myStr: string = myArray[0];

// 字典类型
interface Dictionary {
    [key: string]: number;
}

let ages: Dictionary = {
    alice: 25,
    bob: 30,
    charlie: 35
};

// 混合使用
interface NumberOrString {
    [index: number]: string;
    [key: string]: string | number;
    length: number; // OK
    name: string;   // OK
}
```

### 类类型接口

```typescript
interface ClockInterface {
    currentTime: Date;
    setTime(d: Date): void;

class Clock implements ClockInterface {
    currentTime: Date = new Date();
    
    setTime(d: Date) {
        this.currentTime = d;
    }
    
    constructor(h: number, m: number) {
        // 构造函数逻辑
    }

```

### 接口继承

```typescript
// 单继承
interface Shape {
    color: string;
}

interface Square extends Shape {
    sideLength: number;
}

let square: Square = {
    color: "blue",
    sideLength: 10
};

// 多继承
interface PenStroke {
    penWidth: number;
}

interface FilledSquare extends Square, PenStroke {
    filled: boolean;
}

let filledSquare: FilledSquare = {
    color: "red",
    sideLength: 20,
    penWidth: 5,
    filled: true
};
```

### 混合类型

```typescript
// 对象既是函数也有属性
interface Counter {
    (start: number): string;
    interval: number;
    reset(): void;
}

function getCounter(): Counter {
    let counter = function(start: number) {
        return `Count: ${start}`;
    } as Counter;
    
    counter.interval = 123;
    counter.reset = function() {
        console.log("Reset");
    };
    
    return counter;
}

let c = getCounter();
console.log(c(10));
c.reset();
c.interval = 5.0;
```

## 类型别名（Type Alias）

### 基本类型别名

```typescript
type Name = string;
type Age = number;
type IsActive = boolean;

let userName: Name = "Alice";
let userAge: Age = 25;
let active: IsActive = true;
```

### 对象类型别名

```typescript
type User = {
    id: number;
    name: string;
    email: string;
};

const user: User = {
    id: 1,
    name: "Bob",
    email: "bob@example.com"
};
```

### 联合类型

```typescript
type Status = "success" | "error" | "pending";
type ID = string | number;

let taskStatus: Status = "pending";
let userId: ID = 12345;
userId = "abc123";
```

### 交叉类型

```typescript
type Draggable = {
    drag(): void;
};

type Resizable = {
    resize(): void;
};

type UIWidget = Draggable & Resizable;

const widget: UIWidget = {
    drag() {
        console.log("Dragging");
    },
    resize() {
        console.log("Resizing");
    }
};
```

### 函数类型别名

```typescript
type MathOperation = (x: number, y: number) => number;

const add: MathOperation = (x, y) => x + y;
const subtract: MathOperation = (x, y) => x - y;
const multiply: MathOperation = (x, y) => x * y;
const divide: MathOperation = (x, y) => x / y;

console.log(add(10, 5)); // 15
```

### 元组类型别名

```typescript
type Point2D = [number, number];
type Point3D = [number, number, number];
type RGB = [number, number, number];

let point: Point2D = [10, 20];
let color: RGB = [255, 0, 0];
```

### 泛型类型别名

```typescript
type Container<T> = { value: T };

let numberContainer: Container<number> = { value: 42 };
let stringContainer: Container<string> = { value: "hello" };

// 条件类型
type IsString<T> = T extends string ? true : false;

type A = IsString<string>; // true
type B = IsString<number>; // false
```

## Interface vs Type Alias

### 相同点

1. 都可以描述对象的形状
2. 都支持扩展

```typescript
// Interface 使用 extends
interface Animal {
    name: string;
}

interface Dog extends Animal {
    bark(): void;
}

// Type 使用交叉类型
type AnimalType = {
    name: string;
};

type DogType = AnimalType & {
    bark(): void;
};
```

### 不同点

```typescript
// 1. Interface 可以声明合并
interface Box {
    height: number;
}

interface Box {
    width: number;
}

// 合并后的 Box
let box: Box = { height: 5, width: 6 };

// Type 不能重复声明
// type BoxType = { height: number; }
// type BoxType = { width: number; } // Error

// 2. Type 可以表示原始类型、联合类型、元组
type StringOrNumber = string | number;
type Tuple = [string, number];

// Interface 不能表示原始类型和联合类型
// interface StringOrNumberInterface = string | number; // Error

// 3. Type 可以使用映射类型和条件类型
type ReadOnly<T> = {
    readonly [P in keyof T]: T[P];
};

type User = {
    name: string;
    age: number;
};

type ReadOnlyUser = ReadOnly<User>;
```

### 使用建议

- **优先使用 Interface**：
  - 定义对象的形状
  - 需要声明合并
  - 定义类的契约

- **使用 Type**：
  - 定义联合类型或交叉类型
  - 定义元组类型
  - 使用高级类型功能（映射类型、条件类型）

## 实践示例

```typescript
// src/user-system.ts

// 基础用户信息接口
interface BaseUser {
    id: number;
    username: string;
    email: string;
    createdAt: Date;
}

// 管理员用户（继承）
interface AdminUser extends BaseUser {
    role: "admin";
    permissions: string[];
}

// 普通用户（继承）
interface RegularUser extends BaseUser {
    role: "user";
    profile?: {
        avatar?: string;
        bio?: string;
    };
}

// 用户类型联合
type User = AdminUser | RegularUser;

// 用户仓库接口
interface UserRepository {
    findById(id: number): User | null;
    findByUsername(username: string): User | null;
    save(user: User): void;
    delete(id: number): boolean;
}

// API 响应类型
type ApiResponse<T> = {
    success: boolean;
    data?: T;
    error?: string;
    timestamp: Date;
};

// 用户服务
class UserService {
    constructor(private repository: UserRepository) {}
    
    getUser(id: number): ApiResponse<User> {
        try {
            const user = this.repository.findById(id);
            
            if (!user) {
                return {
                    success: false,
                    error: "User not found",
                    timestamp: new Date()
                };
            }
            
            return {
                success: true,
                data: user,
                timestamp: new Date()
            };
        } catch (error) {
            return {
                success: false,
                error: error instanceof Error ? error.message : "Unknown error",
                timestamp: new Date()
            };
        }
    }
    
    isAdmin(user: User): user is AdminUser {
        return user.role === "admin";
    }
}

// 实现用户仓库
class InMemoryUserRepository implements UserRepository {
    private users: Map<number, User> = new Map();
    
    findById(id: number): User | null {
        return this.users.get(id) || null;
    }
    
    findByUsername(username: string): User | null {
        for (const user of this.users.values()) {
            if (user.username === username) {
                return user;
            }
        }
        return null;
    }
    
    save(user: User): void {
        this.users.set(user.id, user);
    }
    
    delete(id: number): boolean {
        return this.users.delete(id);
    }
}

// 使用示例
const repository = new InMemoryUserRepository();
const userService = new UserService(repository);

const admin: AdminUser = {
    id: 1,
    username: "admin",
    email: "admin@example.com",
    role: "admin",
    permissions: ["read", "write", "delete"],
    createdAt: new Date()
};

const regularUser: RegularUser = {
    id: 2,
    username: "john",
    email: "john@example.com",
    role: "user",
    profile: {
        avatar: "avatar.jpg",
        bio: "Hello, I'm John"
    },
    createdAt: new Date()
};

repository.save(admin);
repository.save(regularUser);

const response = userService.getUser(1);
console.log(response);

if (response.success && response.data) {
    if (userService.isAdmin(response.data)) {
        console.log("User is admin with permissions:", response.data.permissions);
    }
}
```
