# 11 - 装饰器（Decorators）

装饰器是一种特殊类型的声明，它能够被附加到类、方法、访问符、属性或参数上，用于修改类的行为。

> 注意：装饰器是实验性特性，需要在 `tsconfig.json` 中启用 `experimentalDecorators`

## 启用装饰器

```json
// tsconfig.json
{
    "compilerOptions": {
        "target": "ES2020",
        "experimentalDecorators": true,
        "emitDecoratorMetadata": true
    }
}
```

## 类装饰器

### 基本类装饰器

```typescript
// 简单的类装饰器
function sealed(constructor: Function) {
    Object.seal(constructor);
    Object.seal(constructor.prototype);
}

@sealed
class Greeter {
    greeting: string;
    
    constructor(message: string) {
        this.greeting = message;
    }
    
    greet() {
        return `Hello, ${this.greeting}`;
    }
}

// 带参数的类装饰器
function component(options: { selector: string; template: string }) {
    return function(constructor: Function) {
        console.log(`Component registered:`, options);
        (constructor as any).selector = options.selector;
        (constructor as any).template = options.template;
    };
}

@component({
    selector: "my-component",
    template: "<div>Hello World</div>"
})
class MyComponent {}
```

### 类装饰器工厂

```typescript
// 装饰器工厂
function Entity(tableName: string) {
    return function <T extends { new(...args: any[]): {} }>(constructor: T) {
        return class extends constructor {
            tableName = tableName;
            createdAt = new Date();
        };
    };
}

@Entity("users")
class User {
    constructor(public name: string, public email: string) {}
}

const user = new User("Alice", "alice@example.com");
console.log(user);
console.log((user as any).tableName); // "users"
```

## 方法装饰器

```typescript
// 方法装饰器
function log(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;
    
    descriptor.value = function(...args: any[]) {
        console.log(`Calling ${propertyKey} with args:`, args);
        const result = originalMethod.apply(this, args);
        console.log(`Result:`, result);
        return result;
    };
    
    return descriptor;
}

class Calculator {
    @log
    add(a: number, b: number): number {
        return a + b;
    }
    
    @log
    multiply(a: number, b: number): number {
        return a * b;
    }
}

const calc = new Calculator();
calc.add(2, 3); // 会打印日志
calc.multiply(4, 5);

// 性能测试装饰器
function measure(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;
    
    descriptor.value = function(...args: any[]) {
        const start = performance.now();
        const result = originalMethod.apply(this, args);
        const end = performance.now();
        console.log(`${propertyKey} executed in ${(end - start).toFixed(2)}ms`);
        return result;
    };
    
    return descriptor;
}

class DataProcessor {
    @measure
    processLargeData(size: number): number[] {
        return Array.from({ length: size }, (_, i) => i * 2);
    }
}

// 缓存装饰器
function memoize(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;
    const cache = new Map<string, any>();
    
    descriptor.value = function(...args: any[]) {
        const key = JSON.stringify(args);
        
        if (cache.has(key)) {
            console.log(`Returning cached result for ${propertyKey}`);
            return cache.get(key);
        }
        
        const result = originalMethod.apply(this, args);
        cache.set(key, result);
        return result;
    };
    
    return descriptor;
}

class MathOperations {
    @memoize
    fibonacci(n: number): number {
        if (n <= 1) return n;
        return this.fibonacci(n - 1) + this.fibonacci(n - 2);
    }
}
```

## 访问器装饰器

```typescript
// 访问器装饰器
function configurable(value: boolean) {
    return function(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
        descriptor.configurable = value;
    };
}

class Point {
    private _x: number;
    private _y: number;
    
    constructor(x: number, y: number) {
        this._x = x;
        this._y = y;
    }
    
    @configurable(false)
    get x() {
        return this._x;
    }
    
    @configurable(false)
    get y() {
        return this._y;
    }
}

// 验证装饰器
function validate(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const originalSet = descriptor.set;
    
    if (originalSet) {
        descriptor.set = function(value: any) {
            if (value < 0) {
                throw new Error(`${propertyKey} cannot be negative`);
            }
            originalSet.call(this, value);
        };
    }
}

class Product {
    private _price: number = 0;
    
    @validate
    set price(value: number) {
        this._price = value;
    }
    
    get price() {
        return this._price;
    }
}

const product = new Product();
product.price = 100; // OK
// product.price = -50; // Error
```

## 属性装饰器

```typescript
// 属性装饰器
function format(formatString: string) {
    return function(target: any, propertyKey: string) {
        let value: string;
        
        const getter = function() {
            return value;
        };
        
        const setter = function(newVal: string) {
            value = formatString.replace("{value}", newVal);
        };
        
        Object.defineProperty(target, propertyKey, {
            get: getter,
            set: setter,
            enumerable: true,
            configurable: true
        });
    };
}

class Person {
    @format("Mr./Ms. {value}")
    name: string;
    
    constructor(name: string) {
        this.name = name;
    }
}

const person = new Person("Alice");
console.log(person.name); // "Mr./Ms. Alice"

// 必填属性装饰器
const requiredFields: Map<any, string[]> = new Map();

function required(target: any, propertyKey: string) {
    const constructor = target.constructor;
    
    if (!requiredFields.has(constructor)) {
        requiredFields.set(constructor, []);
    }
    
    requiredFields.get(constructor)!.push(propertyKey);
}

function validate(constructor: any) {
    const fields = requiredFields.get(constructor) || [];
    
    return function(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
        const originalMethod = descriptor.value;
        
        descriptor.value = function(...args: any[]) {
            for (const field of fields) {
                if (!(this as any)[field]) {
                    throw new Error(`${field} is required`);
                }
            }
            return originalMethod.apply(this, args);
        };
        
        return descriptor;
    };
}

class RegisterForm {
    @required
    username: string;
    
    @required
    email: string;
    
    password: string;
    
    @validate(RegisterForm)
    submit() {
        console.log("Form submitted");
    }
}
```

## 参数装饰器

```typescript
// 参数装饰器
function logParameter(target: any, propertyKey: string, parameterIndex: number) {
    const metadataKey = `log_${propertyKey}_parameters`;
    
    if (Array.isArray(target[metadataKey])) {
        target[metadataKey].push(parameterIndex);
    } else {
        target[metadataKey] = [parameterIndex];
    }
}

class UserService {
    getUser(@logParameter id: number, @logParameter includeDetails: boolean) {
        console.log(`Getting user ${id}, includeDetails: ${includeDetails}`);
    }
}

// 验证参数
function validateParam(validator: (value: any) => boolean) {
    return function(target: any, propertyKey: string, parameterIndex: number) {
        const metadataKey = `validate_${propertyKey}_parameters`;
        
        if (!target[metadataKey]) {
            target[metadataKey] = {};
        }
        
        target[metadataKey][parameterIndex] = validator;
    };
}

function validateMethod(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;
    const metadataKey = `validate_${propertyKey}_parameters`;
    
    descriptor.value = function(...args: any[]) {
        const validators = target[metadataKey] || {};
        
        for (const index in validators) {
            const validator = validators[index];
            if (!validator(args[index])) {
                throw new Error(`Validation failed for parameter ${index}`);
            }
        }
        
        return originalMethod.apply(this, args);
    };
    
    return descriptor;
}

class ArticleService {
    @validateMethod
    createArticle(
        @validateParam((value) => typeof value === "string" && value.length > 0) title: string,
        @validateParam((value) => typeof value === "string" && value.length >= 10) content: string
    ) {
        console.log("Creating article:", title);
    }
}
```

## 装饰器组合

```typescript
// 多个装饰器可以组合使用
function first() {
    console.log("first(): factory evaluated");
    return function(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
        console.log("first(): called");
    };
}

function second() {
    console.log("second(): factory evaluated");
    return function(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
        console.log("second(): called");
    };
}

class ExampleClass {
    @first()
    @second()
    method() {}
}

// 输出顺序：
// first(): factory evaluated
// second(): factory evaluated
// second(): called
// first(): called
```

## 实践示例：依赖注入容器

```typescript
// src/di-container.ts

/**
 * 简单的依赖注入容器实现
 */

// 元数据存储
const INJECTABLE_METADATA_KEY = Symbol("INJECTABLE_KEY");
const INJECT_METADATA_KEY = Symbol("INJECT_KEY");

// Injectable 装饰器
function Injectable() {
    return function<T extends { new(...args: any[]): {} }>(constructor: T) {
        Reflect.defineMetadata(INJECTABLE_METADATA_KEY, true, constructor);
        return constructor;
    };
}

// Inject 装饰器
function Inject(token: any) {
    return function(target: any, propertyKey: string | symbol, parameterIndex: number) {
        const existingInjectedParams = Reflect.getOwnMetadata(INJECT_METADATA_KEY, target) || [];
        existingInjectedParams.push({ index: parameterIndex, token });
        Reflect.defineMetadata(INJECT_METADATA_KEY, existingInjectedParams, target);
    };
}

// 依赖注入容器
class Container {
    private services = new Map<any, any>();
    
    register<T>(token: any, instance: T): void {
        this.services.set(token, instance);
    }
    
    resolve<T>(token: any): T {
        // 检查是否已注册
        if (this.services.has(token)) {
            return this.services.get(token);
        }
        
        // 检查是否是可注入的类
        const isInjectable = Reflect.getMetadata(INJECTABLE_METADATA_KEY, token);
        if (!isInjectable) {
            throw new Error(`${token.name} is not injectable`);
        }
        
        // 获取构造函数参数的依赖
        const injections = Reflect.getMetadata(INJECT_METADATA_KEY, token) || [];
        
        // 解析依赖
        const dependencies = injections
            .sort((a: any, b: any) => a.index - b.index)
            .map((injection: any) => this.resolve(injection.token));
        
        // 创建实例
        const instance = new token(...dependencies);
        this.services.set(token, instance);
        
        return instance;
    }
}

// 使用示例
@Injectable()
class Database {
    connect() {
        console.log("Database connected");
    }
    
    query(sql: string) {
        console.log("Executing query:", sql);
        return [];
    }
}

@Injectable()
class Logger {
    log(message: string) {
        console.log(`[LOG] ${message}`);
    }
    
    error(message: string) {
        console.error(`[ERROR] ${message}`);
    }
}

@Injectable()
class UserRepository {
    constructor(
        @Inject(Database) private db: Database,
        @Inject(Logger) private logger: Logger
    ) {}
    
    findAll() {
        this.logger.log("Finding all users");
        return this.db.query("SELECT * FROM users");
    }
    
    findById(id: number) {
        this.logger.log(`Finding user ${id}`);
        return this.db.query(`SELECT * FROM users WHERE id = ${id}`);
    }
}

@Injectable()
class UserService {
    constructor(
        @Inject(UserRepository) private userRepo: UserRepository,
        @Inject(Logger) private logger: Logger
    ) {}
    
    getUsers() {
        this.logger.log("Getting users");
        return this.userRepo.findAll();
    }
    
    getUser(id: number) {
        this.logger.log(`Getting user ${id}`);
        return this.userRepo.findById(id);
    }
}

// 使用容器
const container = new Container();
const userService = container.resolve<UserService>(UserService);
userService.getUsers();
userService.getUser(1);
```

## 实践示例：路由装饰器

```typescript
// src/router.ts

/**
 * Express 风格的路由装饰器
 */

type HTTPMethod = "GET" | "POST" | "PUT" | "DELETE";

interface RouteDefinition {
    path: string;
    method: HTTPMethod;
    methodName: string;
}

const routes: RouteDefinition[] = [];

// 路由装饰器
function Route(method: HTTPMethod, path: string) {
    return function(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
        routes.push({
            path,
            method,
            methodName: propertyKey
        });
    };
}

// 便捷方法装饰器
function Get(path: string) {
    return Route("GET", path);
}

function Post(path: string) {
    return Route("POST", path);
}

function Put(path: string) {
    return Route("PUT", path);
}

function Delete(path: string) {
    return Route("DELETE", path);
}

// Controller 装饰器
function Controller(basePath: string) {
    return function<T extends { new(...args: any[]): {} }>(constructor: T) {
        // 为该控制器的所有路由添加基础路径
        routes.forEach(route => {
            if (route.path.startsWith(basePath)) {
                return;
            }
            route.path = basePath + route.path;
        });
        
        return constructor;
    };
}

// 使用示例
@Controller("/api/users")
class UserController {
    @Get("/")
    getAllUsers() {
        return { users: [] };
    }
    
    @Get("/:id")
    getUserById(id: string) {
        return { user: { id } };
    }
    
    @Post("/")
    createUser(body: any) {
        return { created: true };
    }
    
    @Put("/:id")
    updateUser(id: string, body: any) {
        return { updated: true };
    }
    
    @Delete("/:id")
    deleteUser(id: string) {
        return { deleted: true };
    }
}

// 打印所有路由
console.log("Registered routes:");
routes.forEach(route => {
    console.log(`${route.method} ${route.path} -> ${route.methodName}`);
});
```
