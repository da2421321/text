# 12 - 声明文件

声明文件（`.d.ts`）用于为 JavaScript 代码提供类型信息，使得 TypeScript 可以对其进行类型检查。

## 什么是声明文件

声明文件只包含类型声明，不包含实现代码。它们以 `.d.ts` 为扩展名。

```typescript
// example.d.ts
declare function greet(name: string): string;
declare const version: string;
declare class User {
    constructor(name: string);
    name: string;
    sayHello(): void;
}
```

## 全局声明

```typescript
// global.d.ts
declare global {
    // 全局变量
    const APP_VERSION: string;
    const API_URL: string;
    
    // 全局函数
    function log(message: string): void;
    
    // 扩展全局接口
    interface Window {
        myCustomProperty: string;
    }
    
    // 全局命名空间
    namespace MyApp {
        interface Config {
            theme: string;
            lang: string;
        }
    }
}

export {};
```

## 模块声明

```typescript
// lodash.d.ts
declare module "lodash" {
    export function chunk<T>(array: T[], size: number): T[][];
    export function debounce<T extends (...args: any[]) => any>(
        func: T,
        wait: number
    ): T;
    export function get(object: any, path: string, defaultValue?: any): any;
}

// 使用通配符
declare module "*.vue" {
    import { DefineComponent } from "vue";
    const component: DefineComponent<{}, {}, any>;
    export default component;
}

declare module "*.json" {
    const value: any;
    export default value;
}

declare module "*.png" {
    const value: string;
    export default value;
}
```

## 第三方库声明

```typescript
// jquery.d.ts
declare namespace JQuery {
    interface AjaxSettings {
        url?: string;
        method?: string;
        data?: any;
        success?: (data: any) => void;
        error?: (error: any) => void;
    }
}

declare interface JQuery {
    text(): string;
    text(value: string): JQuery;
    html(): string;
    html(value: string): JQuery;
    addClass(className: string): JQuery;
    removeClass(className: string): JQuery;
    on(event: string, handler: (e: Event) => void): JQuery;
    ajax(settings: JQuery.AjaxSettings): void;
}

declare function $(selector: string): JQuery;
declare namespace $ {
    function ajax(settings: JQuery.AjaxSettings): void;
}

// 使用
$("#app").text("Hello");
$.ajax({ url: "/api/data" });
```

## UMD 库声明

```typescript
// math-lib.d.ts
export as namespace MathLib;

export function add(a: number, b: number): number;
export function subtract(a: number, b: number): number;

export interface Calculator {
    add(a: number, b: number): number;
    subtract(a: number, b: number): number;

// 可以作为全局变量使用
// const result = MathLib.add(1, 2);

```

## 声明合并

```typescript
// 接口合并
interface User {
    name: string;
}

interface User {
    age: number;
}

// 合并后的接口
const user: User = {
    name: "Alice",
    age: 25
};

// 命名空间合并
namespace Animals {
    export class Dog {}
}

namespace Animals {
    export class Cat {}
}

// 使用
const dog = new Animals.Dog();
const cat = new Animals.Cat();

// 类与命名空间合并
class Album {
    label: string;
}

namespace Album {
    export class AlbumLabel {}
}

// 函数与命名空间合并
function buildLabel(name: string): string {
    return buildLabel.prefix + name + buildLabel.suffix;
}

namespace buildLabel {
    export let prefix = "[";
    export let suffix = "]";
}
```

## 实践示例：为 JavaScript 库编写声明文件

```typescript
// src/types/my-library.d.ts

/**
 * 为自定义 JavaScript 库编写声明文件
 */

// 假设有一个 JavaScript 库 my-library.js
declare module "my-library" {
    // 配置选项
    export interface Config {
        apiKey: string;
        timeout?: number;
        debug?: boolean;
    }
    
    // 用户接口
    export interface User {
        id: number;
        username: string;
        email: string;
        role: "admin" | "user";
    }
    
    // API 响应
    export interface ApiResponse<T> {
        success: boolean;
        data?: T;
        error?: string;
    }
    
    // 主类
    export class MyLibrary {
        constructor(config: Config);
        
        // 用户相关方法
        getUser(id: number): Promise<ApiResponse<User>>;
        getUsers(): Promise<ApiResponse<User[]>>;
        createUser(user: Omit<User, "id">): Promise<ApiResponse<User>>;
        updateUser(id: number, data: Partial<User>): Promise<ApiResponse<User>>;
        deleteUser(id: number): Promise<ApiResponse<void>>;
        
        // 工具方法
        setApiKey(key: string): void;
        setTimeout(ms: number): void;
        enableDebug(enabled: boolean): void;
    }
    
    // 工具函数
    export function formatDate(date: Date, format: string): string;
    export function validateEmail(email: string): boolean;
    export function generateId(): string;
    
    // 常量
    export const VERSION: string;
    export const DEFAULT_TIMEOUT: number;
    
    // 默认导出
    const instance: MyLibrary;
    export default instance;
}

// 扩展现有类型
declare module "my-library" {
    interface MyLibrary {
        // 添加新方法
        customMethod(): void;
    }
}
```

## 环境声明

```typescript
// env.d.ts - 声明环境变量
declare namespace NodeJS {
    interface ProcessEnv {
        NODE_ENV: "development" | "production" | "test";
        API_URL: string;
        API_KEY: string;
        PORT: string;
        DATABASE_URL: string;
    }
}

// 使用
const apiUrl = process.env.API_URL;
const nodeEnv = process.env.NODE_ENV;
```
