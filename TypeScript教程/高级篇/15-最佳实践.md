# 15 - 最佳实践与性能优化

本章总结 TypeScript 开发中的最佳实践和性能优化技巧。

## 类型定义最佳实践

### 1. 优先使用 interface 定义对象类型

```typescript
// ✅ 推荐：使用 interface
interface User {
    id: number;
    name: string;
    email: string;
}

// ❌ 不推荐：对简单对象使用 type
type User = {
    id: number;
    name: string;
    email: string;
};

// ✅ type 适用场景：联合类型、工具类型
type Status = "active" | "inactive" | "pending";
type ID = string | number;
```

### 2. 避免使用 any

```typescript
// ❌ 不好
function processData(data: any): any {
    return data.value;
}

// ✅ 好：使用 unknown 或泛型
function processData<T>(data: T): T {
    return data;
}

// ✅ 好：使用 unknown 进行类型守卫
function processUnknown(data: unknown): void {
    if (typeof data === "object" && data !== null && "value" in data) {
        console.log((data as { value: any }).value);
    }
}
```

### 3. 使用 const 断言

```typescript
// ❌ 类型过于宽泛
const config = {
    api: "https://api.example.com",
    timeout: 5000
}; // { api: string; timeout: number }

// ✅ 使用 const 断言获得更精确的类型
const config = {
    api: "https://api.example.com",
    timeout: 5000
} as const; // { readonly api: "https://api.example.com"; readonly timeout: 5000 }

// ✅ 枚举替代方案
const STATUS = {
    Active: "active",
    Inactive: "inactive"
} as const;

type Status = typeof STATUS[keyof typeof STATUS];
```

### 4. 合理使用可选属性

```typescript
// ❌ 不好：所有属性都可选
interface User {
    id?: number;
    name?: string;
    email?: string;
}

// ✅ 好：只有真正可选的属性才标记为可选
interface User {
    id: number;
    name: string;
    email: string;
    phone?: string;
    address?: string;
}

// ✅ 使用工具类型处理更新场景
function updateUser(id: number, updates: Partial<User>): void {
    // ...
}
```

### 5. 避免循环依赖

```typescript
// ❌ 不好：循环依赖
// a.ts
import { B } from "./b";
export class A {
    b: B;
}

// b.ts
import { A } from "./a";
export class B {
    a: A;
}

// ✅ 好：提取共同依赖
// types.ts
export interface IClassA {
    b: IClassB;
}
export interface IClassB {
    a: IClassA;
}

// a.ts
import { IClassA, IClassB } from "./types";
export class A implements IClassA {
    b: IClassB;
}
```

## 类型安全实践

### 1. 使用类型守卫

```typescript
// ✅ 好：自定义类型守卫
interface Dog {
    type: "dog";
    bark(): void;
}

interface Cat {
    type: "cat";
    meow(): void;
}

type Pet = Dog | Cat;

function isDog(pet: Pet): pet is Dog {
    return pet.type === "dog";
}

function handlePet(pet: Pet): void {
    if (isDog(pet)) {
        pet.bark(); // 类型安全
    } else {
        pet.meow(); // 类型安全
    }
}
```

### 2. 使用可辨识联合

```typescript
// ✅ 好：使用标签字段
type Result<T> =
    | { success: true; data: T }
    | { success: false; error: string };

function handleResult<T>(result: Result<T>): void {
    if (result.success) {
        console.log(result.data); // 类型安全
    } else {
        console.error(result.error); // 类型安全
    }
}
```

### 3. 使用 strictNullChecks

```json
{
    "compilerOptions": {
        "strictNullChecks": true
    }
}
```

```typescript
// 启用 strictNullChecks 后
function getName(user: User | null): string {
    // ❌ 错误：需要检查 null
    // return user.name;
    
    // ✅ 正确
    if (user === null) {
        return "Unknown";
    }
    return user.name;
}
```

## 错误处理最佳实践

### 1. 使用 Result 类型

```typescript
type Result<T, E = Error> =
    | { ok: true; value: T }
    | { ok: false; error: E };

function divide(a: number, b: number): Result<number> {
    if (b === 0) {
        return {
            ok: false,
            error: new Error("Division by zero")
        };
    }
    
    return {
        ok: true,
        value: a / b
    };
}

const result = divide(10, 2);
if (result.ok) {
    console.log(result.value);
} else {
    console.error(result.error);
}
```

### 2. 自定义错误类型

```typescript
class ValidationError extends Error {
    constructor(
        message: string,
        public field: string,
        public value: any
    ) {
        super(message);
        this.name = "ValidationError";
    }
}

class NotFoundError extends Error {
    constructor(
        public resourceType: string,
        public id: string | number
    ) {
        super(`${resourceType} with id ${id} not found`);
        this.name = "NotFoundError";
    }
}

function handleError(error: unknown): void {
    if (error instanceof ValidationError) {
        console.log(`Validation failed for ${error.field}`);
    } else if (error instanceof NotFoundError) {
        console.log(`Resource not found: ${error.resourceType}`);
    } else if (error instanceof Error) {
        console.log(error.message);
    } else {
        console.log("Unknown error");
    }
}
```

## 实用工具函数

```typescript
// utils/type-guards.ts

/**
 * 检查值是否为字符串
 */
export function isString(value: unknown): value is string {
    return typeof value === "string";
}

/**
 * 检查值是否为数字
 */
export function isNumber(value: unknown): value is number {
    return typeof value === "number" && !isNaN(value);
}

/**
 * 检查值是否为数组
 */
export function isArray<T>(value: unknown): value is T[] {
    return Array.isArray(value);
}

/**
 * 检查对象是否有指定属性
 */
export function hasProperty<K extends string>(
    obj: unknown,
    key: K
): obj is Record<K, unknown> {
    return typeof obj === "object" && obj !== null && key in obj;
}

/**
 * 安全的 JSON 解析
 */
export function parseJSON<T>(json: string): T | null {
    try {
        return JSON.parse(json) as T;
    } catch {
        return null;
    }
}

/**
 * 类型安全的键值获取
 */
export function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
    return obj[key];
}

/**
 * 过滤假值
 */
export function filterTruthy<T>(array: (T | null | undefined | false | "" | 0)[]): T[] {
    return array.filter(Boolean) as T[];
}
```

## 性能监控

```typescript
// utils/performance.ts

/**
 * 性能监控装饰器
 */
export function measurePerformance(
    target: any,
    propertyKey: string,
    descriptor: PropertyDescriptor
): PropertyDescriptor {
    const originalMethod = descriptor.value;
    
    descriptor.value = async function(...args: any[]) {
        const start = performance.now();
        const result = await originalMethod.apply(this, args);
        const end = performance.now();
        
        console.log(`${propertyKey} 执行时间: ${(end - start).toFixed(2)}ms`);
        
        return result;
    };
    
    return descriptor;
}

/**
 * 性能统计类
 */
export class PerformanceMonitor {
    private metrics: Map<string, number[]> = new Map();
    
    record(name: string, duration: number): void {
        if (!this.metrics.has(name)) {
            this.metrics.set(name, []);
        }
        this.metrics.get(name)!.push(duration);
    }
    
    getStats(name: string): { avg: number; min: number; max: number } | null {
        const durations = this.metrics.get(name);
        if (!durations || durations.length === 0) {
            return null;
        }
        
        return {
            avg: durations.reduce((a, b) => a + b, 0) / durations.length,
            min: Math.min(...durations),
            max: Math.max(...durations)
        };
    }
}
```
