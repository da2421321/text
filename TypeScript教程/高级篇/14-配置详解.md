# 14 - TypeScript 配置详解

`tsconfig.json` 是 TypeScript 项目的配置文件，控制编译器的行为。

## 基本配置

```json
{
    "compilerOptions": {
        "target": "ES2020",
        "module": "commonjs",
        "lib": ["ES2020"],
        "outDir": "./dist",
        "rootDir": "./src",
        "strict": true
    },
    "include": ["src/**/*"],
    "exclude": ["node_modules", "**/*.spec.ts"]
}
```

## compilerOptions 详解

### 类型检查选项

```json
{
    "compilerOptions": {
        // 启用所有严格类型检查选项
        "strict": true,
        
        // 以下选项在 strict 为 true 时自动启用
        "noImplicitAny": true,              // 不允许隐式 any
        "strictNullChecks": true,           // 严格的 null 检查
        "strictFunctionTypes": true,        // 严格的函数类型检查
        "strictBindCallApply": true,        // 严格的 bind/call/apply 检查
        "strictPropertyInitialization": true, // 严格的属性初始化检查
        "noImplicitThis": true,             // 不允许隐式 this
        "alwaysStrict": true,               // 使用严格模式
        
        // 额外的检查
        "noUnusedLocals": true,             // 检查未使用的局部变量
        "noUnusedParameters": true,         // 检查未使用的参数
        "noImplicitReturns": true,          // 检查函数是否所有路径都有返回值
        "noFallthroughCasesInSwitch": true, // 检查 switch 语句的 fallthrough
        "noUncheckedIndexedAccess": true,   // 检查索引访问
        "noImplicitOverride": true,         // 要求 override 关键字
        "noPropertyAccessFromIndexSignature": true // 索引签名必须用索引访问
    }
}
```

### 模块解析选项

```json
{
    "compilerOptions": {
        "module": "commonjs",               // 模块系统
        "moduleResolution": "node",         // 模块解析策略
        "baseUrl": "./",                    // 基础路径
        "paths": {                          // 路径映射
            "@/*": ["src/*"],
            "@models/*": ["src/models/*"],
            "@utils/*": ["src/utils/*"]
        },
        "rootDirs": ["src", "generated"],   // 虚拟目录
        "typeRoots": ["./types", "./node_modules/@types"], // 类型定义目录
        "types": ["node", "jest"],          // 包含的类型定义
        "allowSyntheticDefaultImports": true, // 允许合成默认导入
        "esModuleInterop": true,            // ES 模块互操作
        "resolveJsonModule": true           // 允许导入 JSON
    }
}
```

### 输出选项

```json
{
    "compilerOptions": {
        "outDir": "./dist",                 // 输出目录
        "outFile": "./dist/bundle.js",      // 输出单文件（仅 AMD/System）
        "removeComments": true,             // 移除注释
        "noEmit": true,                     // 不生成输出
        "noEmitOnError": true,              // 有错误时不输出
        "declaration": true,                // 生成 .d.ts 文件
        "declarationDir": "./types",        // .d.ts 文件输出目录
        "declarationMap": true,             // 生成 .d.ts.map
        "sourceMap": true,                  // 生成 .map 文件
        "inlineSourceMap": true,            // 内联 source map
        "inlineSources": true               // 在 source map 中包含源码
    }
}
```

### 语言特性选项

```json
{
    "compilerOptions": {
        "target": "ES2020",                 // 目标 ECMAScript 版本
        "lib": ["ES2020", "DOM"],           // 包含的库文件
        "jsx": "react",                     // JSX 编译选项
        "experimentalDecorators": true,     // 启用装饰器
        "emitDecoratorMetadata": true,      // 为装饰器生成元数据
        "downlevelIteration": true,         // 降级迭代器
        "allowJs": true,                    // 允许编译 JS 文件
        "checkJs": true                     // 检查 JS 文件
    }
}
```

### 其他选项

```json
{
    "compilerOptions": {
        "incremental": true,                // 增量编译
        "tsBuildInfoFile": "./.tsbuildinfo", // 增量编译信息文件
        "skipLibCheck": true,               // 跳过声明文件检查
        "forceConsistentCasingInFileNames": true, // 强制文件名大小写一致
        "isolatedModules": true,            // 每个文件作为独立模块
        "importHelpers": true,              // 从 tslib 导入辅助函数
        "importsNotUsedAsValues": "remove"  // 未使用的导入处理
    }
}
```

## include 和 exclude

```json
{
    "include": [
        "src/**/*",
        "tests/**/*"
    ],
    "exclude": [
        "node_modules",
        "**/*.spec.ts",
        "dist",
        "build"
    ]
}
```

## files

```json
{
    "files": [
        "src/index.ts",
        "src/app.ts",
        "src/types.d.ts"
    ]
}
```

## extends

```json
// tsconfig.json
{
    "extends": "./tsconfig.base.json",
    "compilerOptions": {
        "outDir": "./dist"
    }
}

// tsconfig.base.json
{
    "compilerOptions": {
        "target": "ES2020",
        "strict": true
    }
}
```

## 项目引用

```json
// tsconfig.json
{
    "references": [
        { "path": "./packages/core" },
        { "path": "./packages/utils" }
    ]
}

// packages/core/tsconfig.json
{
    "compilerOptions": {
        "composite": true,
        "declaration": true,
        "outDir": "./dist"
    },
    "include": ["src/**/*"]
}
```

## 实用配置示例

```json
// tsconfig.json - 生产环境
{
    "extends": "./tsconfig.base.json",
    "compilerOptions": {
        "sourceMap": false,
        "removeComments": true,
        "noEmitOnError": true
    }
}

// tsconfig.dev.json - 开发环境
{
    "extends": "./tsconfig.base.json",
    "compilerOptions": {
        "sourceMap": true,
        "removeComments": false,
        "noEmitOnError": false,
        "incremental": true
    }
}

// tsconfig.test.json - 测试环境
{
    "extends": "./tsconfig.base.json",
    "compilerOptions": {
        "types": ["jest", "node"]
    },
    "include": ["src/**/*", "tests/**/*"]
}
```

## 命令行选项

```bash
# 使用特定配置文件
tsc --project tsconfig.prod.json

# 覆盖配置选项
tsc --target ES2015 --outDir build

# 监视模式
tsc --watch

# 编译特定文件
tsc src/index.ts src/app.ts

# 生成配置文件
tsc --init

# 显示编译过程
tsc --listFiles

# 只生成声明文件
tsc --declaration --emitDeclarationOnly

# 增量编译
tsc --incremental
```
