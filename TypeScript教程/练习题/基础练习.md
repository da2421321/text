class SimplePromise<T> {
    private value: T | null = null;
    private resolved: boolean = false;
    private callbacks: Array<(value: T) => void> = [];
    
    constructor(executor: (resolve: (value: T) => void) => void) {
        executor(this.resolve.bind(this));
    }
    
    private resolve(value: T): void {
        this.value = value;
        this.resolved = true;
        this.callbacks.forEach(callback => callback(value));
    }
    
    then(callback: (value: T) => void): this {
        if (this.resolved && this.value !== null) {
            callback(this.value);
        } else {
            this.callbacks.push(callback);
        }
        return this;
    }
}

const promise = new SimplePromise<string>((resolve) => {
    setTimeout(() => resolve("Hello"), 1000);
});

promise.then((value) => console.log(value));
```
</details>

type EventMap = {
    click: { x: number; y: number };
    input: { value: string };
    change: { oldValue: any; newValue: any };
};

class EventEmitter<T extends Record<string, any>> {
    private listeners: { [K in keyof T]?: Array<(data: T[K]) => void> } = {};
    
    on<K extends keyof T>(event: K, callback: (data: T[K]) => void): void {
        if (!this.listeners[event]) {
            this.listeners[event] = [];
        }
        this.listeners[event]!.push(callback);
    }
    
    emit<K extends keyof T>(event: K, data: T[K]): void {
        const callbacks = this.listeners[event];
        if (callbacks) {
            callbacks.forEach(callback => callback(data));
        }
    }
}

const emitter = new EventEmitter<EventMap>();

emitter.on("click", (data) => {
    console.log(`Clicked at ${data.x}, ${data.y}`);
});

emitter.on("input", (data) => {
    console.log(`Input value: ${data.value}`);
});

emitter.emit("click", { x: 10, y: 20 });
emitter.emit("input", { value: "Hello" });
```
</details>
