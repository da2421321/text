# 10 - 类型守卫与类型断言

类型守卫和类型断言是 TypeScript 中处理复杂类型场景的重要工具。

## 类型断言（Type Assertion）

### 基本类型断言

```typescript
// 尖括号语法（不推荐在 JSX 中使用）
let someValue: any = "this is a string";
let strLength1: number = (<string>someValue).length;

// as 语法（推荐）
let strLength2: number = (someValue as string).length;

// DOM 元素断言
const inputElement = document.getElementById("username") as HTMLInputElement;
inputElement.value = "Alice";

// 常量断言
let x = "hello" as const;
// x = "world"; // Error: Type '"world"' is not assignable to type '"hello"'

let arr = [1, 2, 3] as const;
// arr[0] = 4; // Error: Cannot assign to '0' because it is a read-only property

let obj = {
    name: "Alice",
    age: 25
} as const;
// obj.name = "Bob"; // Error
```

### 双重断言

```typescript
// 有时需要先断言为 any 或 unknown，再断言为目标类型
interface User {
    name: string;
    age: number;
}

let data = '{"name":"Alice","age":25}';
let user = JSON.parse(data) as User; // 单次断言通常就够了

// 极端情况（不推荐）
let value: string = "hello";
// let num: number = value as number; // Error
let num: number = (value as any) as number; // 双重断言，强制转换
```

### 非空断言

```typescript
function processValue(value: string | null | undefined): void {
    // 使用 ! 断言值不为 null 或 undefined
    console.log(value!.toUpperCase());
}

// 更安全的方式
function processValueSafe(value: string | null | undefined): void {
    if (value != null) {
        console.log(value.toUpperCase());
    }
}

// 确定性赋值断言
class Component {
    // 告诉编译器这个属性会在构造函数外被赋值
    element!: HTMLElement;
    
    constructor() {
        this.initialize();
    }
    
    initialize(): void {
        this.element = document.createElement("div");
    }
}
```

## 类型守卫（Type Guards）

### typeof 类型守卫

```typescript
function padLeft(value: string, padding: string | number): string {
    if (typeof padding === "number") {
        // 在这个分支中，TypeScript 知道 padding 是 number
        return Array(padding + 1).join(" ") + value;
    }
    if (typeof padding === "string") {
        // 在这个分支中，TypeScript 知道 padding 是 string
        return padding + value;
    }
    throw new Error("Expected string or number");
}

// typeof 可以识别的类型
function printType(value: any): void {
    if (typeof value === "string") {
        console.log("String:", value.toUpperCase());
    } else if (typeof value === "number") {
        console.log("Number:", value.toFixed(2));
    } else if (typeof value === "boolean") {
        console.log("Boolean:", value);
    } else if (typeof value === "object") {
        console.log("Object:", value);
    } else if (typeof value === "function") {
        console.log("Function:", value.name);
    } else if (typeof value === "undefined") {
        console.log("Undefined");
    } else if (typeof value === "symbol") {
        console.log("Symbol:", value.toString());
    }
}
```

### instanceof 类型守卫

```typescript
class Dog {
    bark(): void {
        console.log("Woof!");
    }
}

class Cat {
    meow(): void {
        console.log("Meow!");
    }
}

function makeSound(animal: Dog | Cat): void {
    if (animal instanceof Dog) {
        animal.bark(); // TypeScript 知道是 Dog
    } else {
        animal.meow(); // TypeScript 知道是 Cat
    }
}

// 检查错误类型
function handleError(error: unknown): void {
    if (error instanceof Error) {
        console.log("Error:", error.message);
        console.log("Stack:", error.stack);
    } else {
        console.log("Unknown error:", error);
    }
}
```

### in 操作符

```typescript
interface Bird {
    fly(): void;
    layEggs(): void;
}

interface Fish {
    swim(): void;
    layEggs(): void;
}

function move(animal: Bird | Fish): void {
    if ("fly" in animal) {
        animal.fly(); // TypeScript 知道是 Bird
    } else {
        animal.swim(); // TypeScript 知道是 Fish
    }
}

// 检查对象属性
interface Response {
    data?: any;
    error?: string;
}

function handleResponse(response: Response): void {
    if ("error" in response && response.error) {
        console.error("Error:", response.error);
    } else if ("data" in response && response.data) {
        console.log("Data:", response.data);
    }
}
```

### 自定义类型守卫

```typescript
// 基本自定义类型守卫
interface User {
    type: "user";
    name: string;
    email: string;
}

interface Admin {
    type: "admin";
    name: string;
    permissions: string[];
}

type Person = User | Admin;

// 自定义类型守卫函数
function isAdmin(person: Person): person is Admin {
    return person.type === "admin";
}

function isUser(person: Person): person is User {
    return person.type === "user";
}

function greet(person: Person): void {
    if (isAdmin(person)) {
        console.log(`Admin ${person.name} with permissions:`, person.permissions);
    } else {
        console.log(`User ${person.name} (${person.email})`);
    }
}

// 更复杂的类型守卫
interface Success<T> {
    success: true;
    data: T;
}

interface Failure {
    success: false;
    error: string;
}

type Result<T> = Success<T> | Failure;

function isSuccess<T>(result: Result<T>): result is Success<T> {
    return result.success === true;
}

function isFailure<T>(result: Result<T>): result is Failure {
    return result.success === false;
}

function handleResult<T>(result: Result<T>): void {
    if (isSuccess(result)) {
        console.log("Success:", result.data);
    } else {
        console.error("Failure:", result.error);
    }
}

// 数组类型守卫
function isStringArray(value: any): value is string[] {
    return Array.isArray(value) && value.every(item => typeof item === "string");
}

function isNumberArray(value: any): value is number[] {
    return Array.isArray(value) && value.every(item => typeof item === "number");
}

function processArray(arr: unknown): void {
    if (isStringArray(arr)) {
        console.log("Strings:", arr.map(s => s.toUpperCase()));
    } else if (isNumberArray(arr)) {
        console.log("Numbers:", arr.map(n => n * 2));
    }
}
```

### 可辨识联合（Discriminated Unions）

```typescript
// 使用标签字段作为判别器
interface Circle {
    kind: "circle";
    radius: number;
}

interface Square {
    kind: "square";
    size: number;
}

interface Rectangle {
    kind: "rectangle";
    width: number;
    height: number;
}

type Shape = Circle | Square | Rectangle;

function getArea(shape: Shape): number {
    // TypeScript 可以通过 kind 字段进行类型收窄
    switch (shape.kind) {
        case "circle":
            return Math.PI * shape.radius ** 2;
        case "square":
            return shape.size ** 2;
        case "rectangle":
            return shape.width * shape.height;
    }
}

// 完整性检查
function assertNever(x: never): never {
    throw new Error("Unexpected object: " + x);
}

function getAreaWithCheck(shape: Shape): number {
    switch (shape.kind) {
        case "circle":
            return Math.PI * shape.radius ** 2;
        case "square":
            return shape.size ** 2;
        case "rectangle":
            return shape.width * shape.height;
        default:
            // 如果没有处理所有情况，这里会报错
            return assertNever(shape);
    }
}
```

## 类型收窄（Type Narrowing）

### 真值收窄

```typescript
function printLength(str: string | null | undefined): void {
    if (str) {
        // str 是 string（排除了 null 和 undefined）
        console.log(str.length);
    } else {
        console.log("No string");
    }
}

// 数组和对象检查
function processValue(value: string | string[] | null): void {
    if (value && typeof value === "object") {
        // value 是 string[]
        console.log("Array length:", value.length);
    } else if (value) {
        // value 是 string
        console.log("String:", value);
    } else {
        console.log("Null");
    }
}
```

### 等值收窄

```typescript
function example(x: string | number, y: string | boolean): void {
    if (x === y) {
        // x 和 y 都是 string
        x.toUpperCase();
        y.toUpperCase();
    } else {
        console.log(x); // string | number
        console.log(y); // string | boolean
    }
}

// null 检查
function printName(name: string | null): void {
    if (name !== null) {
        console.log(name.toUpperCase());
    }
}

function printName2(name: string | null): void {
    if (name != null) {
        // 同时排除 null 和 undefined
        console.log(name.toUpperCase());
    }
}
```

### 控制流分析

```typescript
function processInput(input: string | number): void {
    let result: string;
    
    if (typeof input === "string") {
        result = input.toUpperCase();
    } else {
        result = input.toFixed(2);
    }
    
    // TypeScript 知道 result 一定被赋值了
    console.log(result);
}

// 提前返回
function getValue(value: string | null): string {
    if (value === null) {
        return "default";
    }
    
    // 这里 value 一定是 string
    return value.toUpperCase();
}
```

## 实践示例：API 响应处理

```typescript
// src/api-handler.ts

/**
 * 使用类型守卫处理 API 响应
 */

// API 响应类型
interface ApiSuccess<T> {
    status: "success";
    data: T;
    timestamp: number;
}

interface ApiError {
    status: "error";
    errorCode: string;
    message: string;
    timestamp: number;
}

type ApiResponse<T> = ApiSuccess<T> | ApiError;

// 类型守卫
function isApiSuccess<T>(response: ApiResponse<T>): response is ApiSuccess<T> {
    return response.status === "success";
}

function isApiError<T>(response: ApiResponse<T>): response is ApiError {
    return response.status === "error";
}

// 用户数据
interface User {
    id: number;
    username: string;
    email: string;
    role: "admin" | "user";
}

// API 客户端
class ApiClient {
    private baseUrl: string;
    
    constructor(baseUrl: string) {
        this.baseUrl = baseUrl;
    }
    
    async fetchUser(userId: number): Promise<User> {
        const response = await this.makeRequest<User>(`/users/${userId}`);
        
        if (isApiSuccess(response)) {
            return response.data;
        } else {
            throw new Error(`API Error: ${response.message} (${response.errorCode})`);
        }
    }
    
    async fetchUsers(): Promise<User[]> {
        const response = await this.makeRequest<User[]>("/users");
        
        if (isApiSuccess(response)) {
            return response.data;
        } else {
            console.error("Failed to fetch users:", response.message);
            return [];
        }
    }
    
    private async makeRequest<T>(endpoint: string): Promise<ApiResponse<T>> {
        // 模拟 API 请求
        try {
            // 实际实现中应该是 fetch 或其他 HTTP 客户端
            if (Math.random() > 0.8) {
                return {
                    status: "error",
                    errorCode: "SERVER_ERROR",
                    message: "Internal server error",
                    timestamp: Date.now()
                };
            }
            
            return {
                status: "success",
                data: {} as T,
                timestamp: Date.now()
            };
        } catch (error) {
            return {
                status: "error",
                errorCode: "NETWORK_ERROR",
                message: error instanceof Error ? error.message : "Unknown error",
                timestamp: Date.now()
            };
        }
    }
}

// 数据验证
function isValidUser(data: unknown): data is User {
    if (typeof data !== "object" || data === null) {
        return false;
    }
    
    const obj = data as Record<string, unknown>;
    
    return (
        typeof obj.id === "number" &&
        typeof obj.username === "string" &&
        typeof obj.email === "string" &&
        (obj.role === "admin" || obj.role === "user")
    );
}

function isValidUserArray(data: unknown): data is User[] {
    return Array.isArray(data) && data.every(isValidUser);
}

// 安全的数据处理
class UserService {
    constructor(private apiClient: ApiClient) {}
    
    async getUser(userId: number): Promise<User | null> {
        try {
            const user = await this.apiClient.fetchUser(userId);
            
            if (isValidUser(user)) {
                return user;
            } else {
                console.error("Invalid user data:", user);
                return null;
            }
        } catch (error) {
            if (error instanceof Error) {
                console.error("Error fetching user:", error.message);
            }
            return null;
        }
    }
    
    async getUsers(): Promise<User[]> {
        try {
            const users = await this.apiClient.fetchUsers();
            
            if (isValidUserArray(users)) {
                return users;
            } else {
                console.error("Invalid users data");
                return [];
            }
        } catch (error) {
            console.error("Error fetching users:", error);
            return [];
        }
    }
    
    async getAdminUsers(): Promise<User[]> {
        const users = await this.getUsers();
        return users.filter(user => user.role === "admin");
    }
}

// 使用示例
async function main() {
    const apiClient = new ApiClient("https://api.example.com");
    const userService = new UserService(apiClient);
    
    // 获取单个用户
    const user = await userService.getUser(1);
    if (user) {
        console.log("User:", user.username);
    }
    
    // 获取所有用户
    const users = await userService.getUsers();
    console.log("Total users:", users.length);
    
    // 获取管理员用户
    const admins = await userService.getAdminUsers();
    console.log("Admin users:", admins.length);
}

// 表单验证示例
type ValidationResult = 
    | { valid: true }
    | { valid: false; errors: string[] };

function isValidationSuccess(result: ValidationResult): result is { valid: true } {
    return result.valid === true;
}

function isValidationFailure(result: ValidationResult): result is { valid: false; errors: string[] } {
    return result.valid === false;
}

class FormValidator {
    validate(data: Record<string, unknown>): ValidationResult {
        const errors: string[] = [];
        
        if (!data.email || typeof data.email !== "string") {
            errors.push("Email is required");
        }
        
        if (!data.password || typeof data.password !== "string" || data.password.length < 6) {
            errors.push("Password must be at least 6 characters");
        }
        
        if (errors.length > 0) {
            return { valid: false, errors };
        }
        
        return { valid: true };
    }
}

function handleFormSubmit(data: Record<string, unknown>): void {
    const validator = new FormValidator();
    const result = validator.validate(data);
    
    if (isValidationSuccess(result)) {
        console.log("Form is valid, submitting...");
    } else {
        console.error("Validation errors:", result.errors);
    }
}
```
