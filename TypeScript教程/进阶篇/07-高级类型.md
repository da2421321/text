# 07 - 高级类型

TypeScript 提供了许多高级类型功能，帮助我们构建更加灵活和类型安全的代码。

## 联合类型（Union Types）

```typescript
// 基本联合类型
type StringOrNumber = string | number;

let value: StringOrNumber;
value = "hello"; // OK
value = 42; // OK
// value = true; // Error

// 函数参数的联合类型
function formatValue(value: string | number): string {
    if (typeof value === "string") {
        return value.toUpperCase();
    } else {
        return value.toFixed(2);
    }
}

console.log(formatValue("hello")); // "HELLO"
console.log(formatValue(3.14159)); // "3.14"

// 联合类型数组
let items: (string | number)[] = ["apple", 1, "banana", 2];
```

## 交叉类型（Intersection Types）

```typescript
// 交叉类型：组合多个类型
interface Person {
    name: string;
    age: number;
}

interface Employee {
    employeeId: number;
    department: string;
}

type Staff = Person & Employee;

const staff: Staff = {
    name: "Alice",
    age: 30,
    employeeId: 1001,
    department: "IT"
};

// Mixin 模式
class Disposable {
    isDisposed: boolean = false;
    dispose() {
        this.isDisposed = true;
    }
}

class Activatable {
    isActive: boolean = false;
    activate() {
        this.isActive = true;
    }
    deactivate() {
        this.isActive = false;
    }
}

class SmartObject implements Disposable, Activatable {
    isDisposed: boolean = false;
    isActive: boolean = false;
    
    constructor() {
        setInterval(() => console.log(this.isActive + " : " + this.isDisposed), 500);
    }
    
    dispose: () => void;
    activate: () => void;
    deactivate: () => void;
}

// 混入实现
function applyMixins(derivedCtor: any, baseCtors: any[]) {
    baseCtors.forEach(baseCtor => {
        Object.getOwnPropertyNames(baseCtor.prototype).forEach(name => {
            derivedCtor.prototype[name] = baseCtor.prototype[name];
        });
    });
}

applyMixins(SmartObject, [Disposable, Activatable]);
```

## 类型守卫（Type Guards）

### typeof 类型守卫

```typescript
function padLeft(value: string, padding: string | number): string {
    // typeof 类型守卫
    if (typeof padding === "number") {
        return Array(padding + 1).join(" ") + value;
    }
    if (typeof padding === "string") {
        return padding + value;
    }
    throw new Error("Expected string or number");
}
```

### instanceof 类型守卫

```typescript
class Bird {
    fly() {
        console.log("Flying");
    }
    layEggs() {
        console.log("Laying eggs");
    }
}

class Fish {
    swim() {
        console.log("Swimming");
    }
    layEggs() {
        console.log("Laying eggs");
    }
}

function move(pet: Bird | Fish) {
    if (pet instanceof Bird) {
        pet.fly(); // TypeScript 知道这是 Bird
    } else {
        pet.swim(); // TypeScript 知道这是 Fish
    }
    pet.layEggs(); // 两者都有
}
```

### 自定义类型守卫

```typescript
interface Cat {
    type: "cat";
    meow(): void;
}

interface Dog {
    type: "dog";
    bark(): void;
}

type Pet = Cat | Dog;

// 自定义类型守卫函数
function isCat(pet: Pet): pet is Cat {
    return pet.type === "cat";
}

function isDog(pet: Pet): pet is Dog {
    return pet.type === "dog";
}

function makeSound(pet: Pet) {
    if (isCat(pet)) {
        pet.meow();
    } else if (isDog(pet)) {
        pet.bark();
    }
}
```

### in 操作符

```typescript
interface A {
    a: string;
}

interface B {
    b: string;
}

function test(x: A | B) {
    if ("a" in x) {
        console.log(x.a); // x 是 A 类型
    } else {
        console.log(x.b); // x 是 B 类型
    }
}
```

## 可辨识联合（Discriminated Unions）

```typescript
// 标签联合类型
interface Square {
    kind: "square";
    size: number;
}

interface Rectangle {
    kind: "rectangle";
    width: number;
    height: number;
}

interface Circle {
    kind: "circle";
    radius: number;
}

type Shape = Square | Rectangle | Circle;

function getArea(shape: Shape): number {
    switch (shape.kind) {
        case "square":
            return shape.size ** 2;
        case "rectangle":
            return shape.width * shape.height;
        case "circle":
            return Math.PI * shape.radius ** 2;
    }
}

const square: Square = { kind: "square", size: 10 };
const rectangle: Rectangle = { kind: "rectangle", width: 10, height: 5 };
const circle: Circle = { kind: "circle", radius: 7 };

console.log("正方形面积:", getArea(square));
console.log("长方形面积:", getArea(rectangle));
console.log("圆形面积:", getArea(circle).toFixed(2));

// 完整性检查
function assertNever(x: never): never {
    throw new Error("Unexpected object: " + x);
}

function getAreaWithCheck(shape: Shape): number {
    switch (shape.kind) {
        case "square":
            return shape.size ** 2;
        case "rectangle":
            return shape.width * shape.height;
        case "circle":
            return Math.PI * shape.radius ** 2;
        default:
            return assertNever(shape); // 确保处理了所有情况
    }
}
```

## 索引类型（Index Types）

```typescript
// keyof 操作符
interface Person {
    name: string;
    age: number;
    location: string;
}

type PersonKeys = keyof Person; // "name" | "age" | "location"

function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
    return obj[key];
}

const person: Person = {
    name: "Alice",
    age: 25,
    location: "Beijing"
};

const name = getProperty(person, "name"); // string
const age = getProperty(person, "age"); // number
// const invalid = getProperty(person, "invalid"); // Error

// 索引访问类型
type NameType = Person["name"]; // string
type AgeType = Person["age"]; // number
type PersonValues = Person[keyof Person]; // string | number
```

## 映射类型（Mapped Types）

```typescript
// 基本映射类型
type Readonly<T> = {
    readonly [P in keyof T]: T[P];
};

type Partial<T> = {
    [P in keyof T]?: T[P];
};

interface Todo {
    title: string;
    description: string;
    completed: boolean;
}

type ReadonlyTodo = Readonly<Todo>;
// {
//     readonly title: string;
//     readonly description: string;
//     readonly completed: boolean;
// }

type PartialTodo = Partial<Todo>;
// {
//     title?: string;
//     description?: string;
//     completed?: boolean;
// }

// 自定义映射类型
type Nullable<T> = {
    [P in keyof T]: T[P] | null;
};

type NullableTodo = Nullable<Todo>;
// {
//     title: string | null;
//     description: string | null;
//     completed: boolean | null;
// }

// 映射类型修饰符
type Required<T> = {
    [P in keyof T]-?: T[P]; // 移除可选修饰符
};

type Mutable<T> = {
    -readonly [P in keyof T]: T[P]; // 移除只读修饰符
};
```

## 条件类型（Conditional Types）

```typescript
// 基本条件类型
type IsString<T> = T extends string ? true : false;

type A = IsString<string>; // true
type B = IsString<number>; // false

// 更实用的条件类型
type NonNullable<T> = T extends null | undefined ? never : T;

type C = NonNullable<string | null | undefined>; // string
type D = NonNullable<number | null>; // number

// 分布式条件类型
type ToArray<T> = T extends any ? T[] : never;

type StrArrOrNumArr = ToArray<string | number>; // string[] | number[]

// Extract 和 Exclude
type Extract<T, U> = T extends U ? T : never;
type Exclude<T, U> = T extends U ? never : T;

type T1 = Extract<"a" | "b" | "c", "a" | "f">; // "a"
type T2 = Exclude<"a" | "b" | "c", "a" | "f">; // "b" | "c"

// infer 关键字
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : any;

function test() {
    return { a: 1, b: "hello" };
}

type TestReturn = ReturnType<typeof test>; // { a: number; b: string; }

// 获取 Promise 的值类型
type Unpromise<T> = T extends Promise<infer U> ? U : T;

type E = Unpromise<Promise<string>>; // string
type F = Unpromise<number>; // number
```

## 模板字面量类型（Template Literal Types）

```typescript

type World = "world";
type Greeting = `hello ${World}`; // "hello world"

type EmailLocaleIDs = "welcome_email" | "email_heading";
type FooterLocaleIDs = "footer_title" | "footer_sendoff";

// 实用工具：事件处理器类型
type PropEventSource<T> = {
    on<K extends string & keyof T>(
        eventName: `${K}Changed`,
        callback: (newValue: T[K]) => void
    ): void;
};

declare function makeWatchedObject<T>(obj: T): T & PropEventSource<T>;

const person2 = makeWatchedObject({
    firstName: "Alice",
    lastName: "Smith",
    age: 25
});

person2.on("firstNameChanged", newName => {
    console.log(`新名字: ${newName.toUpperCase()}`);
});
```

## 实践示例：状态机

```typescript
// src/state-machine.ts

/**
 * 使用高级类型实现状态机
 */

// 状态定义
type State = "idle" | "loading" | "success" | "error";

// 事件定义
type Event =
    | { type: "FETCH" }
    | { type: "SUCCESS"; data: any }
    | { type: "FAILURE"; error: string }
    | { type: "RESET" };

// 状态上下文
interface Context {
    data: any;
    error: string | null;
}

// 状态机类型
type StateMachine = {
    [S in State]: {
        on: {
            [E in Event["type"]]?: State;
        };
    };
};

// 状态机配置
const stateMachineConfig: StateMachine = {
    idle: {
        on: {
            FETCH: "loading"
        }
    },
    loading: {
        on: {
            SUCCESS: "success",
            FAILURE: "error"
        }
    },
    success: {
        on: {
            FETCH: "loading",
            RESET: "idle"
        }
    },
    error: {
        on: {
            FETCH: "loading",
            RESET: "idle"
        }
    }
};

// 状态机实现
class SimpleStateMachine {
    private state: State = "idle";
    private context: Context = {
        data: null,
        error: null
    };
    
    transition(event: Event): void {
        const currentStateConfig = stateMachineConfig[this.state];
        const nextState = currentStateConfig.on[event.type];
        
        if (nextState) {
            console.log(`状态转换: ${this.state} -> ${nextState}`);
            this.state = nextState;
            this.updateContext(event);
        } else {
            console.log(`无效的事件 ${event.type} 在状态 ${this.state}`);
        }
    }
    
    private updateContext(event: Event): void {
        switch (event.type) {
            case "SUCCESS":
                this.context.data = event.data;
                this.context.error = null;
                break;
            case "FAILURE":
                this.context.error = event.error;
                this.context.data = null;
                break;
            case "RESET":
                this.context.data = null;
                this.context.error = null;
                break;
        }
    }
    
    getState(): State {
        return this.state;
    }
    
    getContext(): Context {
        return { ...this.context };
    }
}

// 使用示例
const machine = new SimpleStateMachine();

console.log("初始状态:", machine.getState());

machine.transition({ type: "FETCH" });
console.log("当前状态:", machine.getState());

machine.transition({ type: "SUCCESS", data: { users: [] } });
console.log("当前状态:", machine.getState());
console.log("上下文:", machine.getContext());

machine.transition({ type: "FETCH" });
machine.transition({ type: "FAILURE", error: "网络错误" });
console.log("当前状态:", machine.getState());
console.log("上下文:", machine.getContext());

machine.transition({ type: "RESET" });
console.log("当前状态:", machine.getState());

// 类型安全的 API 客户端
type HTTPMethod = "GET" | "POST" | "PUT" | "DELETE";

type Endpoint = {
    "/users": {
        GET: { response: User[] };
        POST: { body: CreateUserDTO; response: User };
    };
    "/users/:id": {
        GET: { response: User };
        PUT: { body: UpdateUserDTO; response: User };
        DELETE: { response: void };
    };
};

interface User {
    id: number;
    name: string;
    email: string;
}

interface CreateUserDTO {
    name: string;
    email: string;
}

interface UpdateUserDTO {
    name?: string;
    email?: string;
}

class TypeSafeAPIClient {
    async request<
        Path extends keyof Endpoint,
        Method extends keyof Endpoint[Path],
        Config extends Endpoint[Path][Method]
    >(
        path: Path,
        method: Method,
        ...args: "body" extends keyof Config ? [Config["body"]] : []
    ): Promise<Config extends { response: infer R } ? R : never> {
        // 实际的 HTTP 请求逻辑
        console.log(`${String(method)} ${String(path)}`, args[0]);
        
        // 模拟响应
        return {} as any;
    }
}

// 使用示例
const api = new TypeSafeAPIClient();

// 类型安全的 API 调用
async function examples() {
    // GET /users
    const users = await api.request("/users", "GET");
    
    // POST /users
    const newUser = await api.request("/users", "POST", {

        email: "alice@example.com"
```
