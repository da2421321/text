# 09 - 模块与命名空间

模块是 TypeScript 中组织代码的主要方式，帮助我们构建可维护的大型应用。

## ES6 模块

### 导出（Export）

```typescript
// math.ts - 命名导出
export function add(x: number, y: number): number {
    return x + y;
}

export function subtract(x: number, y: number): number {
    return x - y;
}

export const PI = 3.14159;

export class Calculator {
    add(x: number, y: number): number {
        return x + y;
    }
}

// 也可以统一导出
function multiply(x: number, y: number): number {
    return x * y;
}

function divide(x: number, y: number): number {
    return x / y;
}

export { multiply, divide };

// 重命名导出
export { multiply as mul, divide as div };
```

### 默认导出（Default Export）

```typescript
// logger.ts - 默认导出
export default class Logger {
    log(message: string): void {
        console.log(`[LOG] ${message}`);
    }
    
    error(message: string): void {
        console.error(`[ERROR] ${message}`);
    }
}

// user.ts - 默认导出对象
export default {
    name: "Alice",
    age: 25
};

// config.ts - 默认导出函数
export default function getConfig() {
    return {
        apiUrl: "https://api.example.com",
        timeout: 5000
    };
}
```

### 导入（Import）

```typescript
// 命名导入
import { add, subtract, PI } from "./math";
import { Calculator } from "./math";

console.log(add(5, 3));
console.log(PI);

// 重命名导入
import { add as sum, subtract as diff } from "./math";

console.log(sum(10, 5));
console.log(diff(10, 5));

// 导入所有
import * as MathUtils from "./math";

console.log(MathUtils.add(1, 2));
console.log(MathUtils.PI);

// 默认导入
import Logger from "./logger";
import User from "./user";
import getConfig from "./config";

const logger = new Logger();
logger.log("Hello");

// 混合导入
// utils.ts
export const version = "1.0.0";
export default class Utils {
    static format(str: string): string {
        return str.toUpperCase();
    }
}

// app.ts
import Utils, { version } from "./utils";
console.log(Utils.format("hello"));
console.log(version);
```

### 重新导出（Re-export）

```typescript
// models/user.ts
export interface User {
    id: number;
    name: string;
}

// models/product.ts
export interface Product {
    id: number;
    title: string;
    price: number;

// models/index.ts - 汇总导出
export { User } from "./user";
export { Product } from "./product";

// 或者使用 * 导出所有
export * from "./user";
export * from "./product";

// 使用
import { User, Product } from "./models";
```

## 命名空间（Namespace）

命名空间是 TypeScript 早期组织代码的方式，现在主要用于兼容旧代码或特殊场景。

### 基本命名空间

```typescript
// validation.ts
namespace Validation {
    export interface StringValidator {
        isValid(s: string): boolean;
    }
    
    const lettersRegexp = /^[A-Za-z]+$/;
    const numberRegexp = /^[0-9]+$/;
    
    export class LettersOnlyValidator implements StringValidator {
        isValid(s: string): boolean {
            return lettersRegexp.test(s);
        }
    }
    
    export class NumbersOnlyValidator implements StringValidator {
        isValid(s: string): boolean {
            return numberRegexp.test(s);
        }
    }
}

// 使用命名空间
let validator = new Validation.LettersOnlyValidator();
console.log(validator.isValid("Hello")); // true
console.log(validator.isValid("Hello123")); // false
```

### 嵌套命名空间

```typescript
namespace Company {
    export namespace HR {
        export class Employee {
            constructor(public name: string, public id: number) {}
        }
        
        export function hire(name: string, id: number): Employee {
            return new Employee(name, id);
        }
    }
    
    export namespace Finance {
        export interface Budget {
            amount: number;
            year: number;
        }
        
        export function allocate(budget: Budget): void {
            console.log(`分配预算 ${budget.amount} for ${budget.year}`);
        }
    }
}

// 使用
const employee = Company.HR.hire("Alice", 1001);
Company.Finance.allocate({ amount: 100000, year: 2024 });

// 别名
import HR = Company.HR;
const emp = new HR.Employee("Bob", 1002);
```

### 跨文件命名空间

```typescript
// shapes.ts
namespace Shapes {
    export interface Shape {
        area(): number;
    }
}

// circle.ts
/// <reference path="shapes.ts" />
namespace Shapes {
    export class Circle implements Shape {
        constructor(public radius: number) {}
        
        area(): number {
            return Math.PI * this.radius ** 2;
        }
    }
}

// rectangle.ts
/// <reference path="shapes.ts" />
namespace Shapes {
    export class Rectangle implements Shape {
        constructor(public width: number, public height: number) {}
        
        area(): number {
            return this.width * this.height;
        }
    }
}

// 使用
const circle = new Shapes.Circle(5);
const rectangle = new Shapes.Rectangle(10, 5);
```

## 模块 vs 命名空间

```typescript
// ❌ 不推荐：使用命名空间
namespace Utils {
    export function formatDate(date: Date): string {
        return date.toISOString();
    }
    
    export function parseDate(str: string): Date {
        return new Date(str);
    }
}

// ✅ 推荐：使用模块
// date-utils.ts
export function formatDate(date: Date): string {
    return date.toISOString();
}

export function parseDate(str: string): Date {
    return new Date(str);
}

// 使用
import { formatDate, parseDate } from "./date-utils";
```

## 动态导入

```typescript
// 动态导入模块
async function loadModule() {
    if (condition) {
        const module = await import("./heavy-module");
        module.doSomething();
    }
}

// 条件导入
async function getFormatter(type: string) {
    if (type === "json") {
        const { JsonFormatter } = await import("./formatters/json");
        return new JsonFormatter();
    } else if (type === "xml") {
        const { XmlFormatter } = await import("./formatters/xml");
        return new XmlFormatter();
    }
}

// 懒加载
class Application {
    private chartModule: any;
    
    async loadChartModule() {
        if (!this.chartModule) {
            this.chartModule = await import("./charts");
        }
        return this.chartModule;
    }
}
```

## 模块增强（Module Augmentation）

```typescript
// observable.ts
export class Observable<T> {
    // ...implementation
}

// map-operator.ts
import { Observable } from "./observable";

// 声明模块增强
declare module "./observable" {
    interface Observable<T> {
        map<U>(fn: (value: T) => U): Observable<U>;
    }
}

// 实现增强
Observable.prototype.map = function<T, U>(this: Observable<T>, fn: (value: T) => U): Observable<U> {
    // ...implementation
    return new Observable<U>();
};

// 全局增强
declare global {
    interface Array<T> {
        first(): T | undefined;
        last(): T | undefined;
    }
}

Array.prototype.first = function<T>(this: T[]): T | undefined {
    return this[0];
};

Array.prototype.last = function<T>(this: T[]): T | undefined {
    return this[this.length - 1];
};

// 使用
const arr = [1, 2, 3];
console.log(arr.first()); // 1
console.log(arr.last()); // 3
```

## 实践示例：模块化应用架构

```typescript
// src/types/index.ts
export interface User {
    id: number;
    name: string;
    email: string;
}

export interface Product {
    id: number;
    title: string;
    price: number;
    category: string;
}

export interface Order {
    id: number;
    userId: number;
    products: Product[];
    total: number;
    status: OrderStatus;
}

export enum OrderStatus {
    Pending = "pending",
    Processing = "processing",
    Shipped = "shipped",
    Delivered = "delivered",
    Cancelled = "cancelled"
}

// src/services/base-service.ts
export abstract class BaseService<T> {
    protected items: Map<number, T> = new Map();
    
    abstract create(data: Omit<T, "id">): T;
    
    findById(id: number): T | undefined {
        return this.items.get(id);
    }
    
    findAll(): T[] {
        return Array.from(this.items.values());
    }
    
    update(id: number, data: Partial<T>): T | undefined {
        const item = this.items.get(id);
        if (item) {
            const updated = { ...item, ...data };
            this.items.set(id, updated);
            return updated;
        }
        return undefined;
    }
    
    delete(id: number): boolean {
        return this.items.delete(id);
    }
}

// src/services/user-service.ts
import { BaseService } from "./base-service";
import { User } from "../types";

export class UserService extends BaseService<User> {
    private idCounter = 1;
    
    create(data: Omit<User, "id">): User {
        const user: User = {
            id: this.idCounter++,
            ...data
        };
        this.items.set(user.id, user);
        return user;
    }
    
    findByEmail(email: string): User | undefined {
        return this.findAll().find(user => user.email === email);
    }
}

// src/services/product-service.ts
import { BaseService } from "./base-service";
import { Product } from "../types";

export class ProductService extends BaseService<Product> {
    private idCounter = 1;
    
    create(data: Omit<Product, "id">): Product {
        const product: Product = {
            id: this.idCounter++,
            ...data
        };
        this.items.set(product.id, product);
        return product;
    }
    
    findByCategory(category: string): Product[] {
        return this.findAll().filter(p => p.category === category);
    }
    
    searchByTitle(query: string): Product[] {
        return this.findAll().filter(p => 
            p.title.toLowerCase().includes(query.toLowerCase())
        );
    }
}

// src/services/order-service.ts
import { BaseService } from "./base-service";
import { Order, OrderStatus, Product } from "../types";
import { ProductService } from "./product-service";

export class OrderService extends BaseService<Order> {
    private idCounter = 1;
    
    constructor(private productService: ProductService) {
        super();
    }
    
    create(data: Omit<Order, "id">): Order {
        const order: Order = {
            id: this.idCounter++,
            ...data
        };
        this.items.set(order.id, order);
        return order;
    }
    
    createOrder(userId: number, productIds: number[]): Order | null {
        const products: Product[] = [];
        
        for (const id of productIds) {
            const product = this.productService.findById(id);
            if (!product) {
                return null;
            }
            products.push(product);
        }
        
        const total = products.reduce((sum, p) => sum + p.price, 0);
        
        return this.create({
            userId,
            products,
            total,
            status: OrderStatus.Pending
        });
    }
    
    updateStatus(orderId: number, status: OrderStatus): Order | undefined {
        return this.update(orderId, { status });
    }
    
    findByUser(userId: number): Order[] {
        return this.findAll().filter(order => order.userId === userId);
    }
}

// src/services/index.ts
export { UserService } from "./user-service";
export { ProductService } from "./product-service";
export { OrderService } from "./order-service";

// src/app.ts
import { UserService, ProductService, OrderService } from "./services";
import { OrderStatus } from "./types";

class Application {
    private userService: UserService;
    private productService: ProductService;
    private orderService: OrderService;
    
    constructor() {
        this.userService = new UserService();
        this.productService = new ProductService();
        this.orderService = new OrderService(this.productService);
    }
    
    initialize(): void {
        // 创建用户
        const user1 = this.userService.create({
            name: "Alice",
            email: "alice@example.com"
        });
        
        const user2 = this.userService.create({
            name: "Bob",
            email: "bob@example.com"
        });
        
        // 创建商品
        const product1 = this.productService.create({
            title: "Laptop",
            price: 1200,
            category: "Electronics"
        });
        
        const product2 = this.productService.create({
            title: "Mouse",
            price: 25,
            category: "Electronics"
        });
        
        const product3 = this.productService.create({
            title: "Desk",
            price: 300,
            category: "Furniture"
        });
        
        // 创建订单
        const order1 = this.orderService.createOrder(user1.id, [
            product1.id,
            product2.id
        ]);
        
        if (order1) {
            console.log("订单创建成功:", order1);
            
            // 更新订单状态
            this.orderService.updateStatus(order1.id, OrderStatus.Processing);
        }
        
        // 查询
        console.log("所有用户:", this.userService.findAll());
        console.log("电子产品:", this.productService.findByCategory("Electronics"));
        console.log("用户订单:", this.orderService.findByUser(user1.id));
    }
    
    run(): void {
        this.initialize();
    }
}

// 启动应用
const app = new Application();
app.run();
```
