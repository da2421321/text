# 08 - 枚举与字面量类型

枚举和字面量类型是 TypeScript 中定义一组常量值的重要方式。

## 数字枚举

### 基本数字枚举

```typescript
// 默认从 0 开始
enum Direction {
    Up,
    Down,
    Left,
    Right
}

console.log(Direction.Up); // 0
console.log(Direction.Down); // 1
console.log(Direction.Left); // 2
console.log(Direction.Right); // 3

// 反向映射
console.log(Direction[0]); // "Up"
console.log(Direction[1]); // "Down"

// 自定义起始值
enum Status {
    Pending = 1,
    Approved,
    Rejected
}

console.log(Status.Pending); // 1
console.log(Status.Approved); // 2
console.log(Status.Rejected); // 3

// 自定义所有值
enum HttpStatus {
    OK = 200,
    BadRequest = 400,
    Unauthorized = 401,
    Forbidden = 403,
    NotFound = 404,
    InternalServerError = 500
}
```

### 计算成员和常量成员

```typescript
enum FileAccess {
    // 常量成员
    None,
    Read = 1 << 1,
    Write = 1 << 2,
    ReadWrite = Read | Write,
    
    // 计算成员
    G = "123".length
}

console.log(FileAccess.Read); // 2
console.log(FileAccess.Write); // 4
console.log(FileAccess.ReadWrite); // 6
console.log(FileAccess.G); // 3
```

## 字符串枚举

```typescript
enum Color {
    Red = "RED",
    Green = "GREEN",
    Blue = "BLUE"
}

console.log(Color.Red); // "RED"
// 字符串枚举没有反向映射
// console.log(Color["RED"]); // undefined

// 实际应用示例
enum LogLevel {
    ERROR = "ERROR",
    WARN = "WARN",
    INFO = "INFO",
    DEBUG = "DEBUG"
}

function log(level: LogLevel, message: string): void {
    const timestamp = new Date().toISOString();
    console.log(`[${timestamp}] [${level}] ${message}`);
}

log(LogLevel.INFO, "应用程序启动");
log(LogLevel.ERROR, "发生错误");
```

## 常量枚举

```typescript
// 使用 const 声明的枚举会在编译时被内联
const enum Directions {
    Up,
    Down,
    Left,
    Right
}

let directions = [
    Directions.Up,
    Directions.Down,
    Directions.Left,
    Directions.Right
];

// 编译后的代码：
// let directions = [0, 1, 2, 3];

// 优点：性能更好，生成的代码更小
// 缺点：无法进行反向映射
```

## 枚举作为类型

```typescript
enum UserRole {
    Admin = "ADMIN",
    User = "USER",
    Guest = "GUEST"
}

interface User {
    id: number;
    name: string;
    role: UserRole;
}

function checkPermission(user: User): boolean {
    switch (user.role) {
        case UserRole.Admin:
            return true;
        case UserRole.User:
            return false;
        case UserRole.Guest:
            return false;
        default:
            // 完整性检查
            const _exhaustive: never = user.role;
            return _exhaustive;
    }
}

const admin: User = {
    id: 1,
    name: "Alice",
    role: UserRole.Admin
};

console.log(checkPermission(admin));
```

## 字面量类型

### 字符串字面量类型

```typescript
// 字符串字面量类型
type Easing = "ease-in" | "ease-out" | "ease-in-out";

function animate(easing: Easing): void {
    console.log(`使用 ${easing} 缓动`);
}

animate("ease-in"); // OK
// animate("linear"); // Error

// 结合函数重载
class UIElement {
    animate(dx: number, dy: number, easing: "ease-in"): void;
    animate(dx: number, dy: number, easing: "ease-out"): void;
    animate(dx: number, dy: number, easing: "ease-in-out"): void;
    animate(dx: number, dy: number, easing: string): void {
        console.log(`移动 (${dx}, ${dy}) 使用 ${easing}`);
    }
}
```

### 数字字面量类型

```typescript
type DiceRoll = 1 | 2 | 3 | 4 | 5 | 6;

function rollDice(): DiceRoll {
    return (Math.floor(Math.random() * 6) + 1) as DiceRoll;
}

// 配置选项
interface Config {
    width: 100 | 200 | 300;
    height: 100 | 200 | 300;
}

const config: Config = {
    width: 200,
    height: 300
};
```

### 布尔字面量类型

```typescript
type True = true;
type False = false;

// 实际应用：条件类型
type IsTrue<T> = T extends true ? "yes" : "no";

type A = IsTrue<true>; // "yes"
type B = IsTrue<false>; // "no"
```

## 字面量类型 vs 枚举

```typescript
// 使用枚举
enum Direction1 {
    Up = "UP",
    Down = "DOWN",
    Left = "LEFT",
    Right = "RIGHT"
}

function move1(direction: Direction1): void {
    console.log(`移动方向: ${direction}`);
}

// 使用字面量类型
type Direction2 = "UP" | "DOWN" | "LEFT" | "RIGHT";

function move2(direction: Direction2): void {
    console.log(`移动方向: ${direction}`);
}

// 使用 const 对象（作为命名空间）
const Direction3 = {
    Up: "UP",
    Down: "DOWN",
    Left: "LEFT",
    Right: "RIGHT"
} as const;

type Direction3 = typeof Direction3[keyof typeof Direction3];

function move3(direction: Direction3): void {
    console.log(`移动方向: ${direction}`);
}

// 调用方式比较
move1(Direction1.Up);
move2("UP");
move3(Direction3.Up);
```

## 实践示例：状态管理

```typescript
// src/state-management.ts

/**
 * 使用枚举和字面量类型实现状态管理
 */

// 请求状态枚举
enum RequestStatus {
    Idle = "idle",
    Loading = "loading",
    Success = "success",
    Error = "error"
}

// 错误类型字面量
type ErrorType = 
    | "NetworkError" 
    | "ValidationError" 
    | "AuthenticationError" 
    | "ServerError";

// 状态类型
type State<T> = 
    | { status: RequestStatus.Idle }
    | { status: RequestStatus.Loading }
    | { status: RequestStatus.Success; data: T }
    | { status: RequestStatus.Error; errorType: ErrorType; message: string };

// 状态管理类
class StateManager<T> {
    private state: State<T> = { status: RequestStatus.Idle };
    private listeners: Array<(state: State<T>) => void> = [];
    
    getState(): State<T> {
        return this.state;
    }
    
    setState(newState: State<T>): void {
        this.state = newState;
        this.notify();
    }
    
    subscribe(listener: (state: State<T>) => void): () => void {
        this.listeners.push(listener);
        
        // 返回取消订阅函数
        return () => {
            const index = this.listeners.indexOf(listener);
            if (index > -1) {
                this.listeners.splice(index, 1);
            }
        };
    }
    
    private notify(): void {
        this.listeners.forEach(listener => listener(this.state));
    }
    
    // 便捷方法
    setLoading(): void {
        this.setState({ status: RequestStatus.Loading });
    }
    
    setSuccess(data: T): void {
        this.setState({ status: RequestStatus.Success, data });
    }
    
    setError(errorType: ErrorType, message: string): void {
        this.setState({ status: RequestStatus.Error, errorType, message });
    }
    
    reset(): void {
        this.setState({ status: RequestStatus.Idle });
    }
}

// 使用示例
interface User {
    id: number;
    name: string;
    email: string;
}

const userManager = new StateManager<User>();

// 订阅状态变化
const unsubscribe = userManager.subscribe(state => {
    switch (state.status) {
        case RequestStatus.Idle:
            console.log("空闲状态");
            break;
        case RequestStatus.Loading:
            console.log("加载中...");
            break;
        case RequestStatus.Success:
            console.log("加载成功:", state.data);
            break;
        case RequestStatus.Error:
            console.log(`错误 [${state.errorType}]: ${state.message}`);
            break;
    }
});

// 模拟异步操作
async function fetchUser(id: number): Promise<void> {
    userManager.setLoading();
    
    try {
        // 模拟网络请求
        await new Promise(resolve => setTimeout(resolve, 1000));
        
        if (Math.random() > 0.7) {
            throw new Error("网络错误");
        }
        
        const user: User = {
            id,
            name: "Alice",
            email: "alice@example.com"
        };
        
        userManager.setSuccess(user);
    } catch (error) {
        userManager.setError("NetworkError", error.message);
    }
}

// 表单验证示例
const enum ValidationRule {
    Required = "REQUIRED",
    Email = "EMAIL",
    MinLength = "MIN_LENGTH",
    MaxLength = "MAX_LENGTH",
    Pattern = "PATTERN"
}

type ValidationError = {
    rule: ValidationRule;
    message: string;
};

class FormValidator {
    private errors: Map<string, ValidationError[]> = new Map();
    
    validate(field: string, value: string, rules: ValidationRule[]): boolean {
        const fieldErrors: ValidationError[] = [];
        
        for (const rule of rules) {
            switch (rule) {
                case ValidationRule.Required:
                    if (!value || value.trim() === "") {
                        fieldErrors.push({
                            rule,
                            message: "此字段不能为空"
                        });
                    }
                    break;
                    
                case ValidationRule.Email:
                    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
                    if (!emailRegex.test(value)) {
                        fieldErrors.push({
                            rule,
                            message: "请输入有效的邮箱地址"
                        });
                    }
                    break;
                    
                case ValidationRule.MinLength:
                    if (value.length < 6) {
                        fieldErrors.push({
                            rule,
                            message: "至少需要6个字符"
                        });
                    }
                    break;
            }
        }
        
        if (fieldErrors.length > 0) {
            this.errors.set(field, fieldErrors);
            return false;
        } else {
            this.errors.delete(field);
            return true;
        }
    }
    
    getErrors(field: string): ValidationError[] | undefined {
        return this.errors.get(field);
    }
    
    hasErrors(): boolean {
        return this.errors.size > 0;
    }
    
    clearErrors(): void {
        this.errors.clear();
    }
}

// 使用表单验证
const validator = new FormValidator();

validator.validate("email", "test@example.com", [
    ValidationRule.Required,
    ValidationRule.Email
]);

validator.validate("password", "123", [
    ValidationRule.Required,
    ValidationRule.MinLength
]);

console.log("邮箱错误:", validator.getErrors("email"));
console.log("密码错误:", validator.getErrors("password"));

// 游戏状态管理
const enum GameState {
    Menu = "MENU",
    Playing = "PLAYING",
    Paused = "PAUSED",
    GameOver = "GAME_OVER"
}

const enum PlayerAction {
    MoveUp = "MOVE_UP",
    MoveDown = "MOVE_DOWN",
    MoveLeft = "MOVE_LEFT",
    MoveRight = "MOVE_RIGHT",
    Jump = "JUMP",
    Attack = "ATTACK"
}

class GameController {
    private currentState: GameState = GameState.Menu;
    
    setState(newState: GameState): void {
        console.log(`游戏状态: ${this.currentState} -> ${newState}`);
        this.currentState = newState;
    }
    
    handleAction(action: PlayerAction): void {
        if (this.currentState !== GameState.Playing) {
            console.log("游戏未在运行状态");
            return;
        }
        
        switch (action) {
            case PlayerAction.MoveUp:
                console.log("向上移动");
                break;
            case PlayerAction.Jump:
                console.log("跳跃");
                break;
            case PlayerAction.Attack:
                console.log("攻击");
                break;
        }
    }
    
    getState(): GameState {
        return this.currentState;
    }
}

const game = new GameController();
game.setState(GameState.Playing);
game.handleAction(PlayerAction.Jump);
game.handleAction(PlayerAction.Attack);
game.setState(GameState.Paused);
```

## 枚举的最佳实践

```typescript
// 1. 使用字符串枚举提高可读性
enum ApiEndpoint {
    Users = "/api/users",
    Posts = "/api/posts",
    Comments = "/api/comments"
}

// 2. 使用 const enum 优化性能（如果不需要反向映射）
const enum Environment {
    Development = "development",
    Staging = "staging",
    Production = "production"
}

// 3. 结合命名空间组织相关枚举
namespace HTTP {
    export enum Method {
        GET = "GET",
        POST = "POST",
        PUT = "PUT",
        DELETE = "DELETE"
    }
    
    export enum StatusCode {
        OK = 200,
        Created = 201,
        BadRequest = 400,
        NotFound = 404,
        InternalServerError = 500
    }
}

// 使用
console.log(HTTP.Method.GET);
console.log(HTTP.StatusCode.OK);

// 4. 使用字面量类型替代简单枚举
type Theme = "light" | "dark" | "auto";

const THEMES = {
    Light: "light" as const,
    Dark: "dark" as const,
    Auto: "auto" as const
};
```
