> 类比 Vue 中的 axios + JSON 处理，Flutter 用 http 或 dio 包。

## 1. 方案对比

| 工具 | 类比 | 特点 |

## 2. 数据模型（Model）

Flutter 没有 JS 的动态类型，需要定义数据模型类：

```dart
// models/user.dart
class User {
  final int id;
  final String name;
  final String email;
  final String? avatar;

  const User({
    required this.id,
    required this.name,
    required this.email,
    this.avatar,
  });

  // 从 JSON 创建（类比 JS 直接用 response.data）
  factory User.fromJson(Map<String, dynamic> json) {
    return User(
      id: json['id'] as int,
      name: json['name'] as String,
      email: json['email'] as String,
      avatar: json['avatar'] as String?,
    );
  }

  // 转为 JSON（类比 JSON.stringify）
  Map<String, dynamic> toJson() {
    return {
      'id': id,
      'name': name,
      'email': email,
      if (avatar != null) 'avatar': avatar,
    };
  }

  // copyWith（类比 Pinia 的 $patch）
  User copyWith({String? name, String? email, String? avatar}) {
    return User(
      id: id,
      name: name ?? this.name,
      email: email ?? this.email,
      avatar: avatar ?? this.avatar,
    );
  }

  @override
  String toString() => 'User(id: $id, name: $name)';
}
```

## 3. http 包（基础）

```yaml
dependencies:

```

```dart
import 'dart:convert';
import 'package:http/http.dart' as http;

class ApiService {
  static const baseUrl = 'https://api.example.com';

  // GET 请求（类比 axios.get）
  static Future<List<User>> getUsers() async {
    final response = await http.get(
      Uri.parse('$baseUrl/users'),
      headers: {
        'Content-Type': 'application/json',
        'Authorization': 'Bearer $token',
      },
    );

    if (response.statusCode == 200) {
      final List<dynamic> data = jsonDecode(response.body);
      return data.map((json) => User.fromJson(json)).toList();
    } else {
      throw Exception('Failed to load users: ${response.statusCode}');
    }
  }

  // POST 请求（类比 axios.post）
  static Future<User> createUser(Map<String, dynamic> userData) async {
    final response = await http.post(
      Uri.parse('$baseUrl/users'),
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode(userData),
    );

    if (response.statusCode == 201) {
      return User.fromJson(jsonDecode(response.body));
    } else {
      throw Exception('Failed to create user');
    }
  }
}
```

### 基础配置

```dart
// services/http_client.dart
import 'package:dio/dio.dart';

class HttpClient {
  static final Dio _dio = Dio(BaseOptions(
    baseUrl: 'https://api.example.com',
    connectTimeout: Duration(seconds: 10),
    receiveTimeout: Duration(seconds: 10),
    headers: {
      'Content-Type': 'application/json',
    },
  ));

  static Dio get instance {
    _setupInterceptors();
    return _dio;
  }

  static void _setupInterceptors() {
    _dio.interceptors.clear();

    // 请求拦截器（类比 axios.interceptors.request）
    _dio.interceptors.add(
      InterceptorsWrapper(
        onRequest: (options, handler) {
          // 自动添加 token
          final token = AuthService.getToken();
          if (token != null) {
            options.headers['Authorization'] = 'Bearer $token';
          }
          print('请求: ${options.method} ${options.path}');
          handler.next(options);
        },

        // 响应拦截器（类比 axios.interceptors.response）
        onResponse: (response, handler) {
          print('响应: ${response.statusCode}');
          handler.next(response);
        },

        // 错误拦截器
        onError: (error, handler) async {
          if (error.response?.statusCode == 401) {
            // token 过期，刷新 token
            try {
              await AuthService.refreshToken();
              // 重试请求
              final response = await _dio.fetch(error.requestOptions);
              handler.resolve(response);
            } catch (e) {
              // 刷新失败，跳转登录
              AuthService.logout();
              handler.next(error);
            }
          } else {
            handler.next(error);
          }
        },
      ),
    );
  }
}
```

### API 封装

```dart
// services/api_service.dart
class ApiService {
  static final _dio = HttpClient.instance;

  // GET
  static Future<T> get<T>(
    String path, {
    Map<String, dynamic>? params,
    T Function(dynamic)? fromJson,
  }) async {
    try {
      final response = await _dio.get(path, queryParameters: params);
      if (fromJson != null) {
        return fromJson(response.data);
      }
      return response.data as T;
    } on DioException catch (e) {
      throw _handleError(e);
    }
  }

  // POST
  static Future<T> post<T>(
    String path, {
    dynamic data,
    T Function(dynamic)? fromJson,
  }) async {
    try {
      final response = await _dio.post(path, data: data);
      if (fromJson != null) {
        return fromJson(response.data);
      }
      return response.data as T;
    } on DioException catch (e) {
      throw _handleError(e);
    }
  }

  // 错误处理
  static Exception _handleError(DioException e) {
    switch (e.type) {
      case DioExceptionType.connectionTimeout:
        return Exception('连接超时');
      case DioExceptionType.receiveTimeout:
        return Exception('响应超时');
      case DioExceptionType.badResponse:
        final status = e.response?.statusCode;
        final message = e.response?.data?['message'] ?? '请求失败';
        return Exception('[$status] $message');
      default:
        return Exception('网络错误: ${e.message}');
    }
  }
}
```

### 具体 API 模块

```dart
// services/product_api.dart
class ProductApi {
  static Future<List<Product>> getProducts({
    int page = 1,
    int pageSize = 20,
    String? category,
  }) async {
    final data = await ApiService.get<Map<String, dynamic>>(
      '/products',
      params: {
        'page': page,
        'page_size': pageSize,
        if (category != null) 'category': category,
      },
    );
    final list = data['list'] as List;
    return list.map((json) => Product.fromJson(json)).toList();
  }

  static Future<Product> getProductDetail(int id) async {
    final data = await ApiService.get<Map<String, dynamic>>('/products/$id');
    return Product.fromJson(data);
  }

  static Future<void> addToCart(int productId, int quantity) async {
    await ApiService.post('/cart', data: {
      'product_id': productId,
      'quantity': quantity,
    });
  }
}
```

## 5. 在 UI 中使用

### FutureBuilder（类比 Vue 的 async setup + v-if loading）

```dart
class ProductListPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('商品列表')),
      body: FutureBuilder<List<Product>>(
        future: ProductApi.getProducts(),
        builder: (context, snapshot) {
          // 加载中
          if (snapshot.connectionState == ConnectionState.waiting) {
            return Center(child: CircularProgressIndicator());
          }

          // 错误
          if (snapshot.hasError) {
            return Center(
              child: Column(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  Icon(Icons.error_outline, size: 64, color: Colors.red),
                  SizedBox(height: 16),
                  Text('加载失败: ${snapshot.error}'),
                  ElevatedButton(
                    onPressed: () => setState(() {}),  // 重新加载
                    child: Text('重试'),
                  ),
                ],
              ),
            );
          }

          // 成功
          final products = snapshot.data!;
          return ListView.builder(
            itemCount: products.length,
            itemBuilder: (context, index) => ProductCard(product: products[index]),
          );
        },
      ),
    );
  }
}
```

### 结合 Provider 管理异步状态（推荐）

```dart
// stores/product_store.dart
enum LoadingState { idle, loading, success, error }

class ProductStore extends ChangeNotifier {
  List<Product> _products = [];
  LoadingState _state = LoadingState.idle;
  String? _errorMessage;
  int _page = 1;
  bool _hasMore = true;

  List<Product> get products => _products;
  LoadingState get state => _state;
  String? get errorMessage => _errorMessage;
  bool get isLoading => _state == LoadingState.loading;
  bool get hasMore => _hasMore;

  Future<void> loadProducts({bool refresh = false}) async {
    if (refresh) {
      _page = 1;
      _products = [];
      _hasMore = true;
    }

    if (!_hasMore || _state == LoadingState.loading) return;

    _state = LoadingState.loading;
    notifyListeners();

    try {
      final newProducts = await ProductApi.getProducts(page: _page);

      if (newProducts.isEmpty) {
        _hasMore = false;
      } else {
        _products.addAll(newProducts);
        _page++;
      }

      _state = LoadingState.success;
    } catch (e) {
      _state = LoadingState.error;
      _errorMessage = e.toString();
    }

    notifyListeners();
  }
}
```

```dart
// 在页面中使用
class ProductListPage extends StatefulWidget {
  @override
  State<ProductListPage> createState() => _ProductListPageState();
}

class _ProductListPageState extends State<ProductListPage> {
  final _scrollController = ScrollController();

  @override
  void initState() {
    super.initState();
    // 初始加载
    WidgetsBinding.instance.addPostFrameCallback((_) {
      context.read<ProductStore>().loadProducts();
    });

    // 滚动到底部加载更多
    _scrollController.addListener(() {
      if (_scrollController.position.pixels >=
          _scrollController.position.maxScrollExtent - 200) {
        context.read<ProductStore>().loadProducts();
      }
    });
  }

  @override
  void dispose() {
    _scrollController.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('商品列表')),
      body: Consumer<ProductStore>(
        builder: (context, store, child) {
          if (store.state == LoadingState.loading && store.products.isEmpty) {
            return Center(child: CircularProgressIndicator());
          }

          if (store.state == LoadingState.error && store.products.isEmpty) {
            return Center(
              child: Column(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  Text(store.errorMessage ?? '加载失败'),
                  ElevatedButton(
                    onPressed: () => store.loadProducts(refresh: true),
                    child: Text('重试'),
                  ),
                ],
              ),
            );
          }

          return RefreshIndicator(
            onRefresh: () => store.loadProducts(refresh: true),
            child: ListView.builder(
              controller: _scrollController,
              itemCount: store.products.length + (store.hasMore ? 1 : 0),
              itemBuilder: (context, index) {
                if (index == store.products.length) {
                  return Center(
                    child: Padding(
                      padding: EdgeInsets.all(16),
                      child: store.isLoading
                          ? CircularProgressIndicator()
                          : Text('没有更多了'),
                    ),
                  );
                }
                return ProductCard(product: store.products[index]);
              },
            ),
          );
        },
      ),
    );
  }
}
```

## 7. 和 Vue/axios 对比

| axios | dio |
|-------|-----|
| `axios.create({baseURL})` | `Dio(BaseOptions(baseUrl: ...))` |
| `axios.get('/path', {params})` | `dio.get('/path', queryParameters: params)` |
| `axios.post('/path', data)` | `dio.post('/path', data: data)` |
| `axios.interceptors.request` | `InterceptorsWrapper.onRequest` |
| `axios.interceptors.response` | `InterceptorsWrapper.onResponse` |
| `response.data` | `response.data` |
| `FormData` | `FormData.fromMap(...)` |
| `CancelToken` | `CancelToken()` |
