## 1. 本地存储

### shared_preferences（键值存储）

类比 UniApp 的 `uni.setStorageSync` / `localStorage`：

```yaml
dependencies:
  shared_preferences: ^2.2.0
```

```dart
import 'package:shared_preferences/shared_preferences.dart';

class StorageService {
  static SharedPreferences? _prefs;

  // 初始化（在 main.dart 中调用）
  static Future<void> init() async {
    _prefs = await SharedPreferences.getInstance();
  }

  // 存储
  static Future<void> setString(String key, String value) async {
    await _prefs!.setString(key, value);
  }

  static Future<void> setInt(String key, int value) async {
    await _prefs!.setInt(key, value);
  }

  static Future<void> setBool(String key, bool value) async {
    await _prefs!.setBool(key, value);
  }

  // 存储对象（序列化为 JSON）
  static Future<void> setObject(String key, Map<String, dynamic> value) async {
    await _prefs!.setString(key, jsonEncode(value));
  }

  // 读取
  static String? getString(String key) => _prefs!.getString(key);
  static int? getInt(String key) => _prefs!.getInt(key);
  static bool? getBool(String key) => _prefs!.getBool(key);

  static Map<String, dynamic>? getObject(String key) {
    final str = _prefs!.getString(key);
    if (str == null) return null;
    return jsonDecode(str) as Map<String, dynamic>;
  }

  // 删除
  static Future<void> remove(String key) async {
    await _prefs!.remove(key);
  }

  // 清空
  static Future<void> clear() async {
    await _prefs!.clear();
  }
}

// 使用示例
void main() async {
  WidgetsFlutterBinding.ensureInitialized();
  await StorageService.init();
  runApp(MyApp());
}

// 存储 token
await StorageService.setString('token', 'Bearer xxx');

// 读取 token
final token = StorageService.getString('token');

// 存储用户信息
await StorageService.setObject('user', {'id': 1, 'name': 'Alice'});
final user = StorageService.getObject('user');
```

### sqflite（SQLite 数据库）

类比 UniApp 的 plus.sqlite，适合复杂数据：

```yaml
dependencies:
  sqflite: ^2.3.0
  path: ^1.8.0
```

```dart
import 'package:sqflite/sqflite.dart';
import 'package:path/path.dart';

class DatabaseService {
  static Database? _db;

  static Future<Database> get database async {
    _db ??= await _initDatabase();
    return _db!;
  }

  static Future<Database> _initDatabase() async {
    final dbPath = await getDatabasesPath();
    final path = join(dbPath, 'app.db');

    return openDatabase(
      path,
      version: 1,
      onCreate: (db, version) async {
        // 创建表
        await db.execute('''
          CREATE TABLE users (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL,
            email TEXT UNIQUE NOT NULL,
            created_at TEXT DEFAULT CURRENT_TIMESTAMP
          )
        ''');
      },
    );
  }

  // 插入
  static Future<int> insertUser(Map<String, dynamic> user) async {
    final db = await database;
    return db.insert('users', user);
  }

  // 查询
  static Future<List<Map<String, dynamic>>> getUsers() async {
    final db = await database;
    return db.query('users', orderBy: 'created_at DESC');
  }

  // 条件查询
  static Future<Map<String, dynamic>?> getUserById(int id) async {
    final db = await database;
    final results = await db.query(
      'users',
      where: 'id = ?',
      whereArgs: [id],
    );
    return results.isNotEmpty ? results.first : null;
  }

  // 更新
  static Future<int> updateUser(int id, Map<String, dynamic> data) async {
    final db = await database;
    return db.update('users', data, where: 'id = ?', whereArgs: [id]);
  }

  // 删除
  static Future<int> deleteUser(int id) async {
    final db = await database;
    return db.delete('users', where: 'id = ?', whereArgs: [id]);
  }
}
```

## 2. 定位（GPS）

```yaml
dependencies:
  geolocator: ^10.1.0
  geocoding: ^2.1.0  # 地址解析
```

```dart
import 'package:geolocator/geolocator.dart';

class LocationService {
  // 获取当前位置（类比 uni.getLocation）
  static Future<Position?> getCurrentLocation() async {
    // 检查权限
    LocationPermission permission = await Geolocator.checkPermission();
    if (permission == LocationPermission.denied) {
      permission = await Geolocator.requestPermission();
      if (permission == LocationPermission.denied) {
        return null;
      }
    }

    if (permission == LocationPermission.deniedForever) {
      // 引导用户去设置页开启权限
      await Geolocator.openAppSettings();
      return null;
    }

    return await Geolocator.getCurrentPosition(
      desiredAccuracy: LocationAccuracy.high,
    );
  }

  // 计算两点距离
  static double calculateDistance(
    double lat1, double lon1,
    double lat2, double lon2,
  ) {
    return Geolocator.distanceBetween(lat1, lon1, lat2, lon2);
  }
}

// 使用
void _getLocation() async {
  final position = await LocationService.getCurrentLocation();
  if (position != null) {
    print('纬度: ${position.latitude}');
    print('经度: ${position.longitude}');
  }
}
```

## 3. 相机与相册

```yaml
dependencies:
  image_picker: ^1.0.0
  camera: ^0.10.0  # 自定义相机
```

```dart
import 'package:image_picker/image_picker.dart';

class MediaService {
  static final _picker = ImagePicker();

  // 从相册选择图片（类比 uni.chooseImage）
  static Future<List<XFile>> pickImages({int maxCount = 9}) async {
    final images = await _picker.pickMultiImage(
      maxWidth: 1080,
      imageQuality: 80,
    );
    return images.take(maxCount).toList();
  }

  // 拍照
  static Future<XFile?> takePhoto() async {
    return _picker.pickImage(
      source: ImageSource.camera,
      maxWidth: 1080,
      imageQuality: 80,
    );
  }

  // 选择视频
  static Future<XFile?> pickVideo() async {
    return _picker.pickVideo(source: ImageSource.gallery);
  }
}
```

## 4. 推送通知

```yaml
dependencies:
  flutter_local_notifications: ^16.1.0
  firebase_messaging: ^14.7.0  # FCM 推送
```

```dart
import 'package:flutter_local_notifications/flutter_local_notifications.dart';

class NotificationService {
  static final _notifications = FlutterLocalNotificationsPlugin();

  static Future<void> init() async {
    const androidSettings = AndroidInitializationSettings('@mipmap/ic_launcher');
    const iosSettings = DarwinInitializationSettings(
      requestAlertPermission: true,
      requestBadgePermission: true,
      requestSoundPermission: true,
    );

    await _notifications.initialize(
      InitializationSettings(android: androidSettings, iOS: iosSettings),
      onDidReceiveNotificationResponse: (details) {
        // 处理通知点击
        print('通知被点击: ${details.payload}');
      },
    );
  }

  // 发送本地通知（类比 uni.showToast 但更强大）
  static Future<void> showNotification({
    required String title,
    required String body,
    String? payload,
  }) async {
    const androidDetails = AndroidNotificationDetails(
      'default_channel',
      '默认通知',
      importance: Importance.high,
      priority: Priority.high,
    );

    await _notifications.show(
      DateTime.now().millisecondsSinceEpoch ~/ 1000,
      title,
      body,
      NotificationDetails(android: androidDetails),
      payload: payload,
    );
  }

  // 定时通知
  static Future<void> scheduleNotification({
    required String title,
    required String body,
    required DateTime scheduledTime,
  }) async {
    await _notifications.zonedSchedule(
      0,
      title,
      body,
      tz.TZDateTime.from(scheduledTime, tz.local),
      NotificationDetails(
        android: AndroidNotificationDetails('scheduled', '定时通知'),
      ),
      androidScheduleMode: AndroidScheduleMode.exactAllowWhileIdle,
      uiLocalNotificationDateInterpretation:
          UILocalNotificationDateInterpretation.absoluteTime,
    );
  }
}
```

## 6. 权限管理

```yaml
dependencies:
  permission_handler: ^11.0.0
```

```dart
import 'package:permission_handler/permission_handler.dart';

class PermissionService {
  // 请求权限（类比 uni.authorize）
  static Future<bool> requestPermission(Permission permission) async {
    final status = await permission.request();
    return status.isGranted;
  }

  // 检查并请求相机权限
  static Future<bool> checkCameraPermission() async {
    final status = await Permission.camera.status;
    if (status.isGranted) return true;
    if (status.isPermanentlyDenied) {
      // 引导去设置
      openAppSettings();
      return false;
    }
    return requestPermission(Permission.camera);
  }

  // 常用权限
  // Permission.camera          相机
  // Permission.photos          相册
  // Permission.location        位置
  // Permission.microphone      麦克风
  // Permission.notification    通知
  // Permission.storage         存储
}
```

```yaml
dependencies:
```

```dart
import 'package:device_info_plus/device_info_plus.dart';
import 'package:package_info_plus/package_info_plus.dart';

// 获取设备信息（类比 uni.getSystemInfo）
Future<void> getDeviceInfo() async {
  final deviceInfo = DeviceInfoPlugin();

  if (Platform.isAndroid) {
    final info = await deviceInfo.androidInfo;
    print('设备型号: ${info.model}');
    print('Android 版本: ${info.version.release}');
    print('设备 ID: ${info.id}');
  } else if (Platform.isIOS) {
    final info = await deviceInfo.iosInfo;
    print('设备型号: ${info.model}');
    print('iOS 版本: ${info.systemVersion}');
  }
}

// 获取 App 信息（类比 uni.getAppBaseInfo）
Future<void> getAppInfo() async {
  final info = await PackageInfo.fromPlatform();
  print('App 名称: ${info.appName}');
  print('包名: ${info.packageName}');
  print('版本号: ${info.version}');
  print('构建号: ${info.buildNumber}');
}
```

## 8. 网络状态

```yaml
dependencies:
  connectivity_plus: ^5.0.0
```

```dart
import 'package:connectivity_plus/connectivity_plus.dart';

class NetworkService {
  static final _connectivity = Connectivity();

  // 检查网络状态（类比 uni.getNetworkType）
  static Future<bool> isConnected() async {
    final result = await _connectivity.checkConnectivity();
    return result != ConnectivityResult.none;
  }

  // 监听网络变化
  static Stream<ConnectivityResult> get onConnectivityChanged =>
      _connectivity.onConnectivityChanged;
}

// 在 App 中监听网络变化
class _AppState extends State<App> {
  late StreamSubscription _subscription;

  @override
  void initState() {
    super.initState();
    _subscription = NetworkService.onConnectivityChanged.listen((result) {
      if (result == ConnectivityResult.none) {
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(content: Text('网络已断开')),
        );
      }
    });
  }

  @override
  void dispose() {
    _subscription.cancel();
    super.dispose();
  }
}
```
