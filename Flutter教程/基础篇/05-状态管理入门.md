# 05 - 状态管理

> 类比 Vue3 的 Pinia，Flutter 主流方案是 Provider 和 Riverpod。

---

## 1. 状态管理方案对比

| 方案 | 类比 Vue | 适用场景 |
|------|---------|---------|
| `setState` | `ref` 局部状态 | 单组件内的简单状态 |
| `Provider` | Pinia（简单版） | 中小型项目，官方推荐 |
| `Riverpod` | Pinia（增强版） | 中大型项目，更安全 |
| `GetX` | Pinia + Vue Router | 快速开发，功能全 |
| `BLoC` | Vuex | 大型项目，严格分层 |

**推荐学习路径：** setState → Provider → Riverpod

---

## 2. setState（局部状态）

适合单个组件内的状态，类比 Vue 的 `ref`：

```dart
class ShoppingCart extends StatefulWidget {
  @override
  State<ShoppingCart> createState() => _ShoppingCartState();
}

class _ShoppingCartState extends State<ShoppingCart> {
  List<String> _items = [];
  bool _isLoading = false;

  void _addItem(String item) {
    setState(() {
      _items.add(item);
    });
  }

  void _removeItem(int index) {
    setState(() {
      _items.removeAt(index);
    });
  }

  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        Text('购物车 (${_items.length})'),
        ..._items.asMap().entries.map((entry) =>
          ListTile(
            title: Text(entry.value),
            trailing: IconButton(
              icon: Icon(Icons.delete),
              onPressed: () => _removeItem(entry.key),
            ),
          )
        ),
      ],
    );
  }
}
```

## 3. Provider（全局状态）

### 创建 Store（类比 Pinia defineStore）

```dart
// stores/user_store.dart
import 'package:flutter/foundation.dart';

// ChangeNotifier 类比 Pinia 的 store
class UserStore extends ChangeNotifier {
  // 状态（类比 Pinia 的 state）
  String _name = '';
  bool _isLoggedIn = false;
  Map<String, dynamic>? _userInfo;

  // Getter（类比 Pinia 的 getters）
  String get name => _name;
  bool get isLoggedIn => _isLoggedIn;
  Map<String, dynamic>? get userInfo => _userInfo;
  String get displayName => _name.isEmpty ? '未登录' : _name;

  // Action（类比 Pinia 的 actions）
  void login(String name, Map<String, dynamic> info) {
    _name = name;
    _isLoggedIn = true;
    _userInfo = info;
    notifyListeners();  // 类比 Pinia 自动触发更新
  }

  void logout() {
    _name = '';
    _isLoggedIn = false;
    _userInfo = null;
    notifyListeners();
  }

  void updateName(String newName) {
    _name = newName;
    notifyListeners();
  }
}
```

```dart
// stores/cart_store.dart
class CartItem {
  final String id;
  final String name;
  final double price;
  int quantity;

  CartItem({
    required this.id,
    required this.name,
    required this.price,
    this.quantity = 1,
  });
}

class CartStore extends ChangeNotifier {
  final List<CartItem> _items = [];

  List<CartItem> get items => List.unmodifiable(_items);
  int get count => _items.fold(0, (sum, item) => sum + item.quantity);
  double get total => _items.fold(0, (sum, item) => sum + item.price * item.quantity);

  void addItem(CartItem item) {
    final existing = _items.firstWhere(
      (i) => i.id == item.id,
      orElse: () => CartItem(id: '', name: '', price: 0),
    );

    if (existing.id.isNotEmpty) {
      existing.quantity++;
    } else {
      _items.add(item);
    }
    notifyListeners();
  }

  void removeItem(String id) {
    _items.removeWhere((item) => item.id == id);
    notifyListeners();
  }

  void clear() {
    _items.clear();
    notifyListeners();
  }
}
```

### 注册 Provider（类比 Pinia 注册到 app）

```dart
// main.dart
import 'package:provider/provider.dart';

void main() {
  runApp(
    // MultiProvider 类比 Vue 的 app.use(pinia)
    MultiProvider(
      providers: [
        ChangeNotifierProvider(create: (_) => UserStore()),
        ChangeNotifierProvider(create: (_) => CartStore()),
      ],
      child: MyApp(),
    ),
  );
}
```

### 在组件中使用（类比 useStore）

```dart
// 方式1：context.watch（类比 storeToRefs，响应式）
class CartIcon extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    // watch 会在状态变化时重建组件
    final cart = context.watch<CartStore>();

    return Badge(
      label: Text('${cart.count}'),
      child: Icon(Icons.shopping_cart),
    );
  }
}

// 方式2：context.read（类比 store.action，只读取不监听）
class AddToCartButton extends StatelessWidget {
  final CartItem item;
  const AddToCartButton({required this.item});

  @override
  Widget build(BuildContext context) {
    return ElevatedButton(
      onPressed: () {
        // read 不会触发重建，适合在事件处理中使用
        context.read<CartStore>().addItem(item);
      },
      child: Text('加入购物车'),
    );
  }
}

// 方式3：Consumer（精细控制重建范围）
class ProductCard extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        Text('商品信息（不会因购物车变化重建）'),
        // 只有这部分会在购物车变化时重建
        Consumer<CartStore>(
          builder: (context, cart, child) {
            return Text('购物车数量: ${cart.count}');
          },
        ),
      ],
    );
  }
}

// 方式4：Selector（只监听特定字段，性能最优）
Selector<CartStore, int>(
  selector: (context, cart) => cart.count,  // 只监听 count
  builder: (context, count, child) {
    return Text('$count 件商品');
  },
)
```

### 完整示例

```dart
// 登录页面
class LoginPage extends StatefulWidget {
  @override
  State<LoginPage> createState() => _LoginPageState();
}

class _LoginPageState extends State<LoginPage> {
  final _nameController = TextEditingController();

  @override
  void dispose() {
    _nameController.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('登录')),
      body: Padding(
        padding: EdgeInsets.all(16),
        child: Column(
          children: [
            TextField(
              controller: _nameController,
              decoration: InputDecoration(labelText: '用户名'),
            ),
            SizedBox(height: 16),
            ElevatedButton(
              onPressed: () {
                final name = _nameController.text;
                if (name.isNotEmpty) {
                  // 调用 store action
                  context.read<UserStore>().login(name, {'role': 'user'});
                  Navigator.pop(context);
                }
              },
              child: Text('登录'),
            ),
          ],
        ),
      ),
    );
  }
}

// 首页显示用户信息
class HomePage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    final user = context.watch<UserStore>();

    return Scaffold(
      appBar: AppBar(
        title: Text('首页'),
        actions: [
          if (user.isLoggedIn)
            TextButton(
              onPressed: () => context.read<UserStore>().logout(),
              child: Text('退出', style: TextStyle(color: Colors.white)),
            ),
        ],
      ),
      body: Center(
        child: Text(
          user.isLoggedIn ? '欢迎，${user.displayName}！' : '请先登录',
          style: TextStyle(fontSize: 24),
        ),
      ),
    );
  }
}
```

## 4. Riverpod（推荐，更现代）

### 安装

```yaml
dependencies:
  flutter_riverpod: ^2.4.0
```

### 基本用法

```dart
// main.dart
import 'package:flutter_riverpod/flutter_riverpod.dart';

void main() {
  runApp(
    ProviderScope(  // 类比 Provider 的 MultiProvider
      child: MyApp(),
    ),
  );
}
```

```dart
// providers/user_provider.dart
import 'package:flutter_riverpod/flutter_riverpod.dart';

// 简单状态（类比 Pinia 的 ref state）
final counterProvider = StateProvider<int>((ref) => 0);

// 复杂状态（类比 Pinia store）
class UserNotifier extends StateNotifier<UserState> {
  UserNotifier() : super(UserState.initial());

  void login(String name) {
    state = state.copyWith(name: name, isLoggedIn: true);
  }

  void logout() {
    state = UserState.initial();
  }
}

// 定义 Provider
final userProvider = StateNotifierProvider<UserNotifier, UserState>(
  (ref) => UserNotifier(),
);

// 状态类
class UserState {
  final String name;
  final bool isLoggedIn;

  const UserState({required this.name, required this.isLoggedIn});

  factory UserState.initial() => UserState(name: '', isLoggedIn: false);

  UserState copyWith({String? name, bool? isLoggedIn}) {
    return UserState(
      name: name ?? this.name,
      isLoggedIn: isLoggedIn ?? this.isLoggedIn,
    );
  }
}

// 异步 Provider（类比 Pinia 的 async action）
final userListProvider = FutureProvider<List<User>>((ref) async {
  final response = await http.get(Uri.parse('/api/users'));
  return parseUsers(response.body);
});
```

```dart
// 在组件中使用（组件继承 ConsumerWidget）
class UserPage extends ConsumerWidget {
  @override
  Widget build(BuildContext context, WidgetRef ref) {
    // 监听状态（类比 storeToRefs）
    final user = ref.watch(userProvider);
    final counter = ref.watch(counterProvider);

    return Column(
      children: [
        Text('用户: ${user.name}'),
        Text('计数: $counter'),
        ElevatedButton(
          onPressed: () {
            // 调用 action
            ref.read(userProvider.notifier).login('Alice');
            // 修改简单状态
            ref.read(counterProvider.notifier).state++;
          },
          child: Text('登录'),
        ),
      ],
    );
  }
}

// 异步数据展示
class UserListPage extends ConsumerWidget {
  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final usersAsync = ref.watch(userListProvider);

    return usersAsync.when(
      loading: () => CircularProgressIndicator(),
      error: (err, stack) => Text('错误: $err'),
      data: (users) => ListView.builder(
        itemCount: users.length,
        itemBuilder: (context, index) => ListTile(
          title: Text(users[index].name),
        ),
      ),
    );
  }
}
```

## 5. 状态管理选型建议

```
项目规模     推荐方案
─────────────────────────────────────
小型项目     setState（够用就行）
中型项目     Provider（官方推荐，生态好）
中大型项目   Riverpod（类型安全，更现代）
快速开发     GetX（功能全，但耦合高）
```

### 和 Vue/Pinia 的完整对比

| Pinia | Provider | Riverpod |
|-------|---------|---------|
| `defineStore` | `ChangeNotifier` 类 | `StateNotifier` 类 |
| `state` | 私有字段 + getter | `state` 属性 |
| `getters` | getter 方法 | 计算 Provider |
| `actions` | 普通方法 + `notifyListeners()` | 修改 `state` |
| `useStore()` | `context.watch<T>()` | `ref.watch(provider)` |
| `store.action()` | `context.read<T>().method()` | `ref.read(provider.notifier).method()` |
| `storeToRefs` | `Consumer` / `Selector` | `ref.watch` |
| `$patch` | 直接修改 + `notifyListeners()` | `state = state.copyWith(...)` |
