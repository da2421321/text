# 部署与监控

## 容器化部署

### Docker基础

Docker是一个开源的应用容器引擎，让开发者可以打包他们的应用以及依赖包到一个可移植的容器中。

#### Dockerfile编写

```dockerfile
# Dockerfile
FROM golang:1.19-alpine AS builder

# 设置工作目录
WORKDIR /app

# 复制go mod和sum文件
COPY go.mod go.sum ./

# 下载依赖
RUN go mod download

# 复制源代码
COPY . .

# 构建应用
RUN CGO_ENABLED=0 GOOS=linux go build -a -installsuffix cgo -o main .

# 使用更小的基础镜像
FROM alpine:latest

# 安装ca证书（用于HTTPS请求）
RUN apk --no-cache add ca-certificates

# 创建非root用户
RUN adduser -D -s /bin/sh -u 1000 appuser

# 设置工作目录
WORKDIR /app

# 从builder阶段复制二进制文件
COPY --from=builder /app/main .

# 更改文件所有者
RUN chown -R appuser:appuser /app

# 切换到非root用户
USER appuser

# 暴露端口
EXPOSE 8080

# 健康检查
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD curl -f http://localhost:8080/health || exit 1

# 启动命令
CMD ["./main"]
```

#### 多阶段构建优化

```dockerfile
# optimized.Dockerfile
FROM golang:1.19-alpine AS builder

# 安装构建工具
RUN apk add --no-cache git

# 设置工作目录
WORKDIR /build

# 复制go mod和sum文件
COPY go.mod go.sum ./

# 下载依赖（利用Docker缓存）
RUN go mod download

# 复制源代码
COPY . .

# 构建应用
RUN CGO_ENABLED=0 GOOS=linux go build -a -installsuffix cgo -o app .

# 最终阶段
FROM gcr.io/distroless/static:nonroot

# 从builder阶段复制二进制文件
COPY --from=builder /build/app /

# 暴露端口
EXPOSE 8080

# 设置用户
USER nonroot:nonroot

# 启动命令
ENTRYPOINT ["/app"]
```

## Kubernetes部署

### 基础概念

Kubernetes是一个开源的容器编排平台，用于自动化部署、扩展和管理容器化应用。

#### Deployment配置

```yaml

  name: go-app

    app: go-app

  replicas: 3

      app: go-app

      labels:
        app: go-app
    spec:
      containers:
      - name: go-app
```

### Prometheus集成

```go
// metrics.go
package main

import (
    "net/http"
    "time"
    
    "github.com/gin-gonic/gin"
    "github.com/prometheus/client_golang/prometheus"
    "github.com/prometheus/client_golang/prometheus/promhttp"
)

// 定义指标
var (
    httpRequestTotal = prometheus.NewCounterVec(
        prometheus.CounterOpts{
            Name: "http_requests_total",
            Help: "Total number of HTTP requests",
        },
        []string{"method", "endpoint", "status"},
    )
    
    httpRequestDuration = prometheus.NewHistogramVec(
        prometheus.HistogramOpts{
            Name:    "http_request_duration_seconds",
            Help:    "HTTP request duration in seconds",
            Buckets: prometheus.DefBuckets,
        },
        []string{"method", "endpoint"},
    )
    
    activeUsers = prometheus.NewGauge(
        prometheus.GaugeOpts{
            Name: "active_users",
            Help: "Number of active users",
        },
    )
)

func init() {
    // 注册指标
    prometheus.MustRegister(httpRequestTotal)
    prometheus.MustRegister(httpRequestDuration)
    prometheus.MustRegister(activeUsers)
}

// 监控中间件
func MetricsMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        start := time.Now()
        
        c.Next()
        
        // 记录指标
        status := c.Writer.Status()
        method := c.Request.Method
        endpoint := c.FullPath()
        
        httpRequestTotal.WithLabelValues(method, endpoint, string(rune(status))).Inc()
        httpRequestDuration.WithLabelValues(method, endpoint).Observe(time.Since(start).Seconds())
    }
}

func main() {
    r := gin.Default()
    
    // 使用监控中间件
    r.Use(MetricsMiddleware())
    
    // 暴露Prometheus指标端点
    r.GET("/metrics", gin.WrapH(promhttp.Handler()))
    
    // 应用路由
    r.GET("/api/users", getUsers)
    r.GET("/api/users/:id", getUser)
    r.POST("/api/users", createUser)
    
    r.Run(":8080")
}

func getUsers(c *gin.Context) {
    // 模拟活跃用户增加
    activeUsers.Inc()
    defer activeUsers.Dec()
    
    // 模拟处理时间
    time.Sleep(100 * time.Millisecond)
    
    c.JSON(http.StatusOK, gin.H{"users": []string{"user1", "user2"}})
}

func getUser(c *gin.Context) {
    activeUsers.Inc()
    defer activeUsers.Dec()
    
    time.Sleep(50 * time.Millisecond)
    
    id := c.Param("id")
    c.JSON(http.StatusOK, gin.H{"user": "user" + id})
}

func createUser(c *gin.Context) {
    activeUsers.Inc()
    defer activeUsers.Dec()
    
    time.Sleep(200 * time.Millisecond)
    
    c.JSON(http.StatusCreated, gin.H{"message": "User created"})
}
```

## 日志管理

### 结构化日志

```go
// logging.go
package main

import (
    "fmt"
    "time"
    
    "github.com/gin-gonic/gin"
    "github.com/sirupsen/logrus"
)

var log = logrus.New()

func init() {
    // 设置日志格式为JSON
    log.SetFormatter(&logrus.JSONFormatter{})
    
    // 设置日志级别
    log.SetLevel(logrus.InfoLevel)
}

// 日志中间件
func LoggingMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        start := time.Now()
        
        // 处理请求
        c.Next()
        
        // 记录访问日志
        log.WithFields(logrus.Fields{
            "method":     c.Request.Method,
            "path":       c.Request.URL.Path,
            "status":     c.Writer.Status(),
            "client_ip":  c.ClientIP(),
            "user_agent": c.Request.UserAgent(),
            "duration":   time.Since(start).Milliseconds(),
            "timestamp":  time.Now().Format(time.RFC3339),
        }).Info("HTTP request")
    }
}

// 错误日志中间件
func ErrorLoggingMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        c.Next()
        
        // 记录错误
        if len(c.Errors) > 0 {
            for _, err := range c.Errors {
                log.WithFields(logrus.Fields{
                    "error": err.Error(),
                    "path":  c.Request.URL.Path,
                }).Error("Request error")
            }
        }
    }
}

func main() {
    r := gin.Default()
    
    // 使用日志中间件
    r.Use(LoggingMiddleware())
    r.Use(ErrorLoggingMiddleware())
    
    r.GET("/api/users", getUsers)
    r.GET("/api/users/:id", getUser)
    r.POST("/api/users", createUser)
    
    r.Run(":8080")
}

func getUsers(c *gin.Context) {
    log.WithFields(logrus.Fields{
        "function": "getUsers",
    }).Info("Fetching users")
    
    // 模拟业务逻辑
    users := []string{"user1", "user2", "user3"}
    
    c.JSON(200, gin.H{"users": users})
}

func getUser(c *gin.Context) {
    id := c.Param("id")
    
    log.WithFields(logrus.Fields{
        "function": "getUser",
        "user_id":  id,
    }).Info("Fetching user")
    
    if id == "0" {
        err := fmt.Errorf("user not found: %s", id)
        log.WithFields(logrus.Fields{
            "error":   err.Error(),
            "user_id": id,
        }).Error("User not found")
        c.JSON(404, gin.H{"error": err.Error()})
        return
    }
    
    c.JSON(200, gin.H{"user": "user" + id})
}

func createUser(c *gin.Context) {
    log.WithFields(logrus.Fields{
        "function": "createUser",
    }).Info("Creating user")
    
    // 模拟创建用户
    c.JSON(201, gin.H{"message": "User created successfully"})
}
```

## 安全最佳实践

### HTTPS配置

```go
// https.go
package main

import (
    "crypto/tls"
    "net/http"
    
    "github.com/gin-gonic/gin"
)

func main() {
    r := gin.Default()
    
    // API路由
    r.GET("/api/health", func(c *gin.Context) {
        c.JSON(http.StatusOK, gin.H{"status": "healthy"})
    })
    
    // HTTPS服务器配置
    server := &http.Server{
        Addr:    ":8443",
        Handler: r,
        TLSConfig: &tls.Config{
            MinVersion: tls.VersionTLS12,
            CipherSuites: []uint16{
                tls.TLS_ECDHE_RSA_WITH_AES_256_GCM_SHA384,
                tls.TLS_ECDHE_RSA_WITH_CHACHA20_POLY1305,
                tls.TLS_ECDHE_RSA_WITH_AES_128_GCM_SHA256,
            },
        },
    }
    
    // 启动HTTPS服务器
    if err := server.ListenAndServeTLS("cert.pem", "key.pem"); err != nil {
        panic(err)
    }
}
```

```go
package main

import (
    "net/http"

func main() {
    r := gin.Default()
    
    // 配置CORS中间件
    corsConfig := cors.New(cors.Options{
        AllowedOrigins: []string{
            "https://yourdomain.com",
            "https://www.yourdomain.com",
        },
        AllowedMethods: []string{
            http.MethodGet,
            http.MethodPost,
            http.MethodPut,
            http.MethodDelete,
            http.MethodOptions,
        },
        AllowedHeaders: []string{
            "Origin",
            "Content-Type",
            "Accept",
            "Authorization",
        },
        ExposedHeaders: []string{
            "Content-Length",
        },
        AllowCredentials: true,
        MaxAge:          3600, // 1小时
    })
    
    // 应用CORS中间件
    r.Use(func(c *gin.Context) {
        corsConfig.HandlerFunc(c.Writer, c.Request)
        if c.Request.Method == "OPTIONS" {
            c.AbortWithStatus(204)
            return
        }
        c.Next()
    })
    
    r.GET("/api/users", getUsers)
    
    r.Run(":8080")
}
```

## 代码示例

```go
// complete-deployment-example/main.go
package main

import (
    "context"
    "fmt"
    "net/http"
    "os"
    "os/signal"
    "syscall"
    ""time"
    
    "github.com/gin-gonic/gin"
    "github.com/prometheus/client_golang/prometheus"
    "github.com/prometheus/client_golang/prometheus/promhttp"
    "github.com/sirupsen/logrus"
)

var (
    log = logrus.New()
    
    // Prometheus指标
    httpRequestTotal = prometheus.NewCounterVec(
        prometheus.CounterOpts{
            Name: "http_requests_total",
            Help: "Total number of HTTP requests",
        },
        []string{"method", "endpoint", "status"},
    )
    
    httpRequestDuration = prometheus.NewHistogramVec(
        prometheus.HistogramOpts{
            Name:    "http_request_duration_seconds",
            Help:    "HTTP request duration in seconds",
            Buckets: prometheus.DefBuckets,
        },
        []string{"method", "endpoint"},
    )
)

func init() {
    // 配置日志
    log.SetFormatter(&logrus.JSONFormatter{})
    log.SetLevel(logrus.InfoLevel)
    
    // 注册Prometheus指标
    prometheus.MustRegister(httpRequestTotal)
    prometheus.MustRegister(httpRequestDuration)
}

// 应用配置
type Config struct {
    Port         string
    LogLevel     string
    DatabaseURL  string
    RedisURL     string
}

func loadConfig() Config {
    return Config{
        Port:        getEnv("PORT", "8080"),
        LogLevel:    getEnv("LOG_LEVEL", "info"),
        DatabaseURL: getEnv("DATABASE_URL", "localhost:5432"),
        RedisURL:    getEnv("REDIS_URL", "localhost:6379"),
    }
}

func getEnv(key, defaultValue string) string {
    if value := os.Getenv(key); value != "" {
        return value
    }
    return defaultValue
}

// 健康检查处理器
func healthCheck(c *gin.Context) {
    c.JSON(http.StatusOK, gin.H{
        "status":    "healthy",
        "timestamp": time.Now().Format(time.RFC3339),
    })
}

// 就绪检查处理器
func readyCheck(c *gin.Context) {
    // 这里可以检查数据库连接、缓存连接等
    c.JSON(http.StatusOK, gin.H{
        "status": "ready",
    })
}

// 监控中间件
func metricsMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        start := time.Now()
        
        c.Next()
        
        status := c.Writer.Status()
        method := c.Request.Method
        endpoint := c.FullPath()
        
        httpRequestTotal.WithLabelValues(method, endpoint, fmt.Sprintf("%d", status)).Inc()
        httpRequestDuration.WithLabelValues(method, endpoint).Observe(time.Since(start).Seconds())
    }
}

// 日志中间件
func loggingMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        start := time.Now()
        
        c.Next()
        
        log.WithFields(logrus.Fields{
            "method":     c.Request.Method,
            "path":       c.Request.URL.Path,
            "status":     c.Writer.Status(),
            "client_ip":  c.ClientIP(),
            "user_agent": c.Request.UserAgent(),
            "duration_ms": time.Since(start).Milliseconds(),
            "timestamp":  time.Now().Format(time.RFC3339),
        }).Info("HTTP request processed")
    }
}

func main() {
    config := loadConfig()
    
    // 设置日志级别
    if level, err := logrus.ParseLevel(config.LogLevel); err == nil {
        log.SetLevel(level)
    }
    
    log.WithFields(logrus.Fields{
        "port": config.Port,
        "log_level": config.LogLevel,
    }).Info("Starting application")
    
    // 设置Gin模式
    if config.LogLevel == "debug" {
        gin.SetMode(gin.DebugMode)
    } else {
        gin.SetMode(gin.ReleaseMode)
    }
    
    r := gin.New()
    
    // 全局中间件
    r.Use(loggingMiddleware())
    r.Use(metricsMiddleware())
    r.Use(gin.Recovery())
    
    // 路由
    r.GET("/health", healthCheck)
    r.GET("/ready", readyCheck)
    r.GET("/metrics", gin.WrapH(promhttp.Handler()))
    
    // API路由
    api := r.Group("/api/v1")
    {
        api.GET("/users", func(c *gin.Context) {
            c.JSON(http.StatusOK, gin.H{"users": []string{"user1", "user2"}})
        })
        
        api.GET("/users/:id", func(c *gin.Context) {
            id := c.Param("id")
            c.JSON(http.StatusOK, gin.H{"user": "user" + id})
        })
    }
    
    // 创建HTTP服务器
    srv := &http.Server{
        Addr:    ":" + config.Port,
        Handler: r,
    }
    
    // 在goroutine中启动服务器
    go func() {
        log.Infof("Server starting on port %s", config.Port)
        if err := srv.ListenAndServe(); err != nil && err != http.ErrServerClosed {
            log.Fatalf("Server failed to start: %v", err)
        }
    }()
    
    // 等待中断信号以优雅地关闭服务器
    quit := make(chan os.Signal, 1)
    signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
    <-quit
    log.Info("Shutting down server...")
    
    // 上下文用于通知服务器它有5秒的时间来完成当前请求
    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    defer cancel()
    
    if err := srv.Shutdown(ctx); err != nil {
        log.Fatalf("Server forced to shutdown: %v", err)
    }
    
    log.Info("Server exited")
}
```

```bash
#!/bin/bash
# blue-green-deploy.sh - 蓝绿部署脚本

set -e

APP_NAME="go-app"

NAMESPACE="production"

    echo "请指定新版本 (blue|green)"

# 构建并推送新版本镜像
echo "构建新版本镜像..."

docker push your-registry/$APP_NAME:$NEW_VERSION-latest

# 扩容新版本
echo "扩容新版本..."
kubectl scale deployment/$APP_NAME-$NEW_VERSION --replicas=3 -n $NAMESPACE

# 等待新版本就绪
echo "等待新版本就绪..."
kubectl rollout status deployment/$APP_NAME-$NEW_VERSION -n $NAMESPACE

# 测试新版本
echo "测试新版本..."
# 这里可以添加实际的测试逻辑

# 切换流量到新版本

kubectl patch service/$APP_NAME-service -p '{"spec":{"selector":{"version":"'$NEW_VERSION'"}}}' -n $NAMESPACE
```
