# 数据库操作

## 数据库驱动和 ORM

Go 语言提供了多种数据库操作方式，包括原生数据库驱动和 ORM 框架。

### 数据库驱动

#### MySQL 驱动

首先安装 MySQL 驱动：

```bash
go get -u github.com/go-sql-driver/mysql
```

```go
// mysql_example.go
package main

import (
    "database/sql"
    "fmt"
    "log"
    "time"

    _ "github.com/go-sql-driver/mysql"
)

func main() {
    // 连接数据库
    db, err := sql.Open("mysql", "user:password@tcp(127.0.0.1:3306)/testdb")
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()

    // 测试连接
    if err := db.Ping(); err != nil {
        log.Fatal(err)
    }

    fmt.Println("数据库连接成功")

    // 创建表
    createTable(db)

    // 插入数据
    insertUser(db, "张三", "zhangsan@example.com", 25)
    insertUser(db, "李四", "lisi@example.com", 30)

    // 查询数据
    queryUsers(db)

    // 更新数据
    updateUser(db, 1, "张三丰", "zhangsanfeng@example.com", 35)

    // 删除数据
    deleteUser(db, 2)

    // 查询更新后的数据
    queryUsers(db)
}

func createTable(db *sql.DB) {
    query := `
    CREATE TABLE IF NOT EXISTS users (
        id INT AUTO_INCREMENT PRIMARY KEY,
        name VARCHAR(100) NOT NULL,
        email VARCHAR(100) UNIQUE NOT NULL,
        age INT,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )`

    _, err := db.Exec(query)
    if err != nil {
        log.Fatal(err)
    }

    fmt.Println("表创建成功")
}

func insertUser(db *sql.DB, name, email string, age int) {
    query := "INSERT INTO users (name, email, age) VALUES (?, ?, ?)"
    result, err := db.Exec(query, name, email, age)
    if err != nil {
        log.Fatal(err)
    }

    id, _ := result.LastInsertId()
    fmt.Printf("插入用户成功，ID: %d\n", id)
}

func queryUsers(db *sql.DB) {
    query := "SELECT id, name, email, age, created_at FROM users"
    rows, err := db.Query(query)
    if err != nil {
        log.Fatal(err)
    }
    defer rows.Close()

    fmt.Println("用户列表:")
    for rows.Next() {
        var id int
        var name, email string
        var age int
        var createdAt time.Time

        err := rows.Scan(&id, &name, &email, &age, &createdAt)
        if err != nil {
            log.Fatal(err)
        }

        fmt.Printf("ID: %d, 姓名: %s, 邮箱: %s, 年龄: %d, 创建时间: %s\n",
            id, name, email, age, createdAt.Format("2006-01-02 15:04:05"))
    }
}

func updateUser(db *sql.DB, id int, name, email string, age int) {
    query := "UPDATE users SET name=?, email=?, age=? WHERE id=?"
    _, err := db.Exec(query, name, email, age, id)
    if err != nil {
        log.Fatal(err)
    }

    fmt.Printf("更新用户%d成功\n", id)
}

func deleteUser(db *sql.DB, id int) {
    query := "DELETE FROM users WHERE id=?"
    _, err := db.Exec(query, id)
    if err != nil {
        log.Fatal(err)
    }

    fmt.Printf("删除用户%d成功\n", id)
}
```

#### PostgreSQL 驱动

安装 PostgreSQL 驱动：

```bash
go get -u github.com/lib/pq
```

```go
// postgres_example.go
package main

import (
    "database/sql"
    "fmt"
    "log"

    _ "github.com/lib/pq"
)

func main() {
    // 连接PostgreSQL数据库
    connStr := "user=username dbname=testdb sslmode=disable"
    db, err := sql.Open("postgres", connStr)
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()

    // 测试连接
    if err := db.Ping(); err != nil {
        log.Fatal(err)
    }

    fmt.Println("PostgreSQL数据库连接成功")

    // 创建表
    createTable(db)

    // 插入数据
    insertProduct(db, "笔记本电脑", 5999.99, "高性能笔记本")
    insertProduct(db, "无线鼠标", 99.99, "人体工学设计")

    // 查询数据
    queryProducts(db)
}

func createTable(db *sql.DB) {
    query := `
    CREATE TABLE IF NOT EXISTS products (
        id SERIAL PRIMARY KEY,
        name VARCHAR(100) NOT NULL,
        price DECIMAL(10,2) NOT NULL,
        description TEXT,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )`

    _, err := db.Exec(query)
    if err != nil {
        log.Fatal(err)
    }

    fmt.Println("产品表创建成功")
}

func insertProduct(db *sql.DB, name string, price float64, desc string) {
    query := "INSERT INTO products (name, price, description) VALUES ($1, $2, $3)"
    result, err := db.Exec(query, name, price, desc)
    if err != nil {
        log.Fatal(err)
    }

    rowsAffected, _ := result.RowsAffected()
    fmt.Printf("插入产品成功，影响行数: %d\n", rowsAffected)
}

func queryProducts(db *sql.DB) {
    query := "SELECT id, name, price, description, created_at FROM products"
    rows, err := db.Query(query)
    if err != nil {
        log.Fatal(err)
    }
    defer rows.Close()

    fmt.Println("产品列表:")
    for rows.Next() {
        var id int
        var name, description string
        var price float64
        var createdAt string

        err := rows.Scan(&id, &name, &price, &description, &createdAt)
        if err != nil {
            log.Fatal(err)
        }

        fmt.Printf("ID: %d, 名称: %s, 价格: %.2f, 描述: %s, 创建时间: %s\n",
            id, name, price, description, createdAt)
    }
}
```

## ORM 框架 - GORM

GORM 是 Go 语言中最流行的 ORM 框架。

### 安装 GORM

```bash
go get -u gorm.io/gorm
go get -u gorm.io/driver/mysql
# 或者 PostgreSQL
# go get -u gorm.io/driver/postgres
```

### 基本使用

```go
// gorm_example.go
package main

import (
    "fmt"
    "log"

    "gorm.io/driver/mysql"
    "gorm.io/gorm"
)

// 定义模型
// 📌 GORM 表名映射规则：
// 结构体名 → 数据库表名（自动转换）
// User     → users      (小写 + 复数)
// Product  → products   (小写 + 复数)
// Order    → orders     (小写 + 复数)
// UserProfile → user_profiles (驼峰转下划线 + 复数)

type User struct {
    ID    uint   `gorm:"primaryKey"`
    Name  string `gorm:"not null"`
    Email string `gorm:"uniqueIndex;not null"`
    Age   int
}
// 对应表名: users

type Product struct {
    ID          uint    `gorm:"primaryKey"`
    Name        string  `gorm:"not null"`
    Price       float64 `gorm:"not null"`
    Description string
}
// 对应表名: products

type Order struct {
    ID       uint    `gorm:"primaryKey"`
    UserID   uint
    User     User    `gorm:"foreignKey:UserID"`
    ProductID uint
    Product  Product `gorm:"foreignKey:ProductID"`
    Quantity int     `gorm:"not null"`
    Total    float64 `gorm:"not null"`
}
// 对应表名: orders

func main() {
    // 连接数据库
    dsn := "user:password@tcp(127.0.0.1:3306)/testdb?charset=utf8mb4&parseTime=True&loc=Local"
    db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{})
    if err != nil {
        log.Fatal("连接数据库失败:", err)
    }

    fmt.Println("数据库连接成功")

    // 自动迁移模式
    db.AutoMigrate(&User{}, &Product{}, &Order{})
    fmt.Println("数据库表创建成功")

    // 创建用户
    user := User{Name: "张三", Email: "zhangsan@example.com", Age: 25}
    result := db.Create(&user)
    if result.Error != nil {
        log.Fatal("创建用户失败:", result.Error)
    }
    fmt.Printf("创建用户成功，ID: %d\n", user.ID)

    // 创建产品
    product := Product{Name: "笔记本电脑", Price: 5999.99, Description: "高性能笔记本"}
    result = db.Create(&product)
    if result.Error != nil {
        log.Fatal("创建产品失败:", result.Error)
    }
    fmt.Printf("创建产品成功，ID: %d\n", product.ID)

    // 查询用户
    var users []User
    db.Find(&users)
    // GORM 生成的 MySQL SQL（MySQL 方言，使用反引号包裹表名/列名）：
    // SELECT * FROM `users`;
    // 常见变体：
    //   Where 条件：db.Where("age > ?", 18).Find(&users) -> SELECT * FROM `users` WHERE `age` > 18;
    //   排序：     db.Order("id DESC").Find(&users)      -> SELECT * FROM `users` ORDER BY `id` DESC;
    //   分页：     db.Limit(10).Offset(20).Find(&users)  -> SELECT * FROM `users` LIMIT 10 OFFSET 20;
    //   调试输出： db.Debug().Find(&users)               -> 在日志中打印对应 SQL
    fmt.Println("用户列表:")
    for _, u := range users {
        fmt.Printf("ID: %d, 姓名: %s, 邮箱: %s, 年龄: %d\n", u.ID, u.Name, u.Email, u.Age)
    }

    // 根据条件查询
    var userByEmail User
    db.Where("email = ?", "zhangsan@example.com").First(&userByEmail)
    fmt.Printf("通过邮箱查询用户: %+v\n", userByEmail)

    // 更新用户
    db.Model(&userByEmail).Update("age", 30)
    // MySQL 生成的 SQL：
    // UPDATE `users` SET `age` = ? WHERE `id` = ?;
    // 若模型包含时间戳（如 `UpdatedAt`），则为：
    // UPDATE `users` SET `age` = ?, `updated_at` = ? WHERE `id` = ?;
    // 查看实际 SQL：db.Debug().Model(&userByEmail).Update("age", 30)
    fmt.Println("用户年龄更新成功")

    // 删除用户
    db.Delete(&User{}, 2) // 删除ID为2的用户
    fmt.Println("用户删除成功")

    // 关联查询
    createOrder(db, user.ID, product.ID, 2)
    queryOrdersWithRelations(db)
}

func createOrder(db *gorm.DB, userID, productID uint, quantity int) {
    // 1️⃣ 声明一个空的 Product 结构体变量
    var product Product

    // 2️⃣ GORM 查询：根据主键查询第一条记录
    // db.First(&product, productID) 等价于 SQL:
    //   SELECT * FROM products WHERE id = productID LIMIT 1
    //
    // 详细说明：
    //   - db: GORM 数据库连接对象
    //   - First(): GORM 方法，查询第一条记录
    //   - &product: 传入指针，GORM 会将查询结果填充到这个变量中
    //   - productID: 主键值，GORM 自动识别为 WHERE id = productID
    db.First(&product, productID)

    order := Order{
        UserID:   userID,
        ProductID: productID,
        Quantity: quantity,
        Total:    product.Price * float64(quantity),  // 使用查询到的 product.Price
    }

    result := db.Create(&order)
    // MySQL 生成的 SQL（GORM 方言，使用反引号）：
    // INSERT INTO `orders` (`user_id`,`product_id`,`quantity`,`total`) VALUES (?,?,?,?);
    // 若模型包含时间戳字段（如 `CreatedAt`,`UpdatedAt`），对应列也会包含在 INSERT 中。
    // 调试查看实际 SQL：db.Debug().Create(&order)
    if result.Error != nil {
        log.Fatal("创建订单失败:", result.Error)
    }
    fmt.Printf("创建订单成功，ID: %d\n", order.ID)
}

func queryOrdersWithRelations(db *gorm.DB) {
    var orders []Order

    // 📌 预加载（Preload）关联数据
    // 作用：在查询订单的同时，自动加载关联的 User 和 Product 数据
    //
    // ⚠️ 重要：Preload 不是执行 JOIN，而是执行多条独立的 SELECT 查询！
    //
    // 执行的 SQL（GORM 自动按顺序执行 3 条查询）：
    // 
    // 【SQL 1】查询所有订单
    // SELECT * FROM orders
    // 
    // 假设查到 3 条订单：
    // | id | user_id | product_id | quantity | total    |
    // |----|---------|------------|----------|----------|
    // | 1  | 1       | 5          | 2        | 11999.98 |
    // | 2  | 1       | 6          | 1        | 99.99    |
    // | 3  | 2       | 5          | 1        | 5999.99  |
    //
    // 【SQL 2】根据订单中的 user_id，批量查询用户
    // SELECT * FROM users WHERE id IN (1, 2)
    // 
    // 查询结果：
    // | id | name | email                |
    // |----|------|----------------------|
    // | 1  | 张三 | zhangsan@example.com |
    // | 2  | 李四 | lisi@example.com     |
    //
    // 【SQL 3】根据订单中的 product_id，批量查询产品
    // SELECT * FROM products WHERE id IN (5, 6)
    // 
    // 查询结果：
    // | id | name       | price   |
    // |----|------------|---------|
    // | 5  | 笔记本电脑 | 5999.99 |
    // | 6  | 无线鼠标   | 99.99   |
    //
    // GORM 自动将查询结果组装到 Order 结构体中：
    // orders[0].User.Name = "张三"
    // orders[0].Product.Name = "笔记本电脑"
    // orders[1].User.Name = "张三"
    // orders[1].Product.Name = "无线鼠标"
    // orders[2].User.Name = "李四"
    // orders[2].Product.Name = "笔记本电脑"
    //
    // Preload 的优势：
    //   ✓ 避免 N+1 查询问题（不会为每个订单单独查询用户和产品）
    //   ✓ 批量加载关联数据，提高效率
    //   ✓ 自动去重（相同的 user_id 或 product_id 只查询一次）
    db.Preload("User").Preload("Product").Find(&orders)

    fmt.Println("订单列表:")
    for _, order := range orders {
        // 可以直接访问 order.User.Name 和 order.Product.Name
        // 因为 Preload 已经提前加载了这些数据
        fmt.Printf("订单ID: %d, 用户: %s, 产品: %s, 数量: %d, 总价: %.2f\n",
            order.ID, order.User.Name, order.Product.Name, order.Quantity, order.Total)
    }
}
```

### 高级查询

```go
// gorm_advanced.go
package main

import (
    "fmt"
    "log"

    "gorm.io/driver/mysql"
    "gorm.io/gorm"
)

type Article struct {
    ID      uint   `gorm:"primaryKey"`
    Title   string `gorm:"not null"`
    Content string `gorm:"type:text"`
    Author  string `gorm:"not null"`
    Status  string `gorm:"default:'draft'"`
    Tags    string `gorm:"type:json"` // 存储标签数组的JSON
}

func main() {
    dsn := "user:password@tcp(127.0.0.1:3306)/testdb?charset=utf8mb4&parseTime=True&loc=Local"
    db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{})
    if err != nil {
        log.Fatal("连接数据库失败:", err)
    }

    db.AutoMigrate(&Article{})

    // 批量插入数据
    articles := []Article{
        {Title: "Go语言入门", Content: "Go语言是一门...", Author: "张三", Status: "published"},
        {Title: "Gin框架教程", Content: "Gin是一个...", Author: "李四", Status: "published"},
        {Title: "数据库操作", Content: "数据库操作...", Author: "张三", Status: "draft"},
        {Title: "Web开发", Content: "Web开发技术...", Author: "王五", Status: "published"},
    }

    db.Create(&articles)
    // MySQL 批量插入生成的 SQL（GORM 会使用多值 INSERT）：
    // INSERT INTO `articles` (`title`,`content`,`author`,`status`,`tags`) VALUES
    //   (?,?,?,?,?), (?,?,?,?,?), (?,?,?,?,?), (?,?,?,?,?);
    // 含时间戳字段（如 `CreatedAt`,`UpdatedAt`）时，上述列会一并包含。
    // 查看实际 SQL：db.Debug().Create(&articles)
    fmt.Println("批量插入文章成功")

    // 条件查询
    fmt.Println("\n=== 条件查询 ===")

    // 查询特定作者的文章
    var authorArticles []Article
    db.Where("author = ?", "张三").Find(&authorArticles)
    fmt.Printf("张三的文章数量: %d\n", len(authorArticles))

    // 查询已发布的文章
    var publishedArticles []Article
    db.Where("status = ?", "published").Find(&publishedArticles)
    fmt.Printf("已发布的文章数量: %d\n", len(publishedArticles))

    // 复合条件查询
    var complexArticles []Article
    db.Where("author = ? AND status = ?", "张三", "published").Find(&complexArticles)
    fmt.Printf("张三已发布的文章数量: %d\n", len(complexArticles))

    // 排序和限制
    fmt.Println("\n=== 排序和限制 ===")
    var sortedArticles []Article

    // 📌 Order + Limit 组合
    // db.Order("id desc"): 按 id 降序排序（desc = descending，从大到小）
    // Limit(3):            限制返回 3 条记录
    // Find(&sortedArticles): 查询所有符合条件的文章
    //
    // 生成的 SQL：
    // SELECT * FROM articles ORDER BY id DESC LIMIT 3
    //
    // 执行顺序：查询 → 排序 → 取前3条 → 返回结果
    db.Order("id desc").Limit(3).Find(&sortedArticles)

    fmt.Println("最新的3篇文章:")
    for _, article := range sortedArticles {
        fmt.Printf("  - %s (作者: %s)\n", article.Title, article.Author)
    }

    // 分页查询
    fmt.Println("\n=== 分页查询 ===")
    var paginatedArticles []Article
    page := 1      // 当前页码（第1页）
    pageSize := 2  // 每页显示2条记录

    // 📌 Offset + Limit 实现分页
    // Offset((page - 1) * pageSize): 跳过前面的记录
    //   - 第1页：跳过 (1-1)*2 = 0 条，从第1条开始
    //   - 第2页：跳过 (2-1)*2 = 2 条，从第3条开始
    //   - 第3页：跳过 (3-1)*2 = 4 条，从第5条开始
    // Limit(pageSize): 限制返回 pageSize 条记录
    //
    // 生成的 SQL（第1页）：
    // SELECT * FROM articles LIMIT 2 OFFSET 0
    //
    // 生成的 SQL（第2页）：
    // SELECT * FROM articles LIMIT 2 OFFSET 2
    //
    // 执行顺序：查询 → 跳过 Offset 条 → 取 Limit 条 → 返回结果
    db.Offset((page - 1) * pageSize).Limit(pageSize).Find(&paginatedArticles)

    fmt.Printf("第%d页的文章:\n", page)
    for _, article := range paginatedArticles {
        fmt.Printf("  - %s\n", article.Title)
    }

    // 聚合查询
    fmt.Println("\n=== 聚合查询 ===")
    var count int64
    db.Model(&Article{}).Where("status = ?", "published").Count(&count)
    // MySQL 生成的 SQL：
    // SELECT count(*) FROM `articles` WHERE `status` = ?;
    // 若模型包含软删除字段（`deleted_at`），将自动追加：AND `deleted_at` IS NULL
    // 查看实际 SQL：db.Debug().Model(&Article{}).Where("status = ?", "published").Count(&count)
    fmt.Printf("已发布文章总数: %d\n", count)

    // 分组查询
    // 定义结果结构体（用于接收聚合查询结果）
    type AuthorCount struct {
        Author string  // 作者名
        Count  int     // 文章数量
    }

    var authorCounts []AuthorCount

    // 📌 分组聚合查询：统计每个作者的文章数量
    db.Model(&Article{}).                      // 1️⃣ 指定查询的表（articles）
        Select("author, count(*) as count").   // 2️⃣ 选择字段并聚合计数
        Group("author").                       // 3️⃣ 按作者分组
        Scan(&authorCounts)                    // 4️⃣ 将结果扫描到结构体切片

    // 生成的 SQL：
    // SELECT author, count(*) as count
    // FROM articles
    // GROUP BY author
    //
    // 执行过程：
    // 1. 按 author 字段分组
    // 2. 统计每组的记录数
    // 3. 返回 author 和对应的 count

    fmt.Println("各作者文章数量:")
    for _, ac := range authorCounts {
        fmt.Printf("  %s: %d篇\n", ac.Author, ac.Count)
    }

    // 更新操作
    fmt.Println("\n=== 更新操作 ===")

    // 批量更新
    db.Model(&Article{}).Where("status = ?", "draft").Update("status", "published")
    fmt.Println("所有草稿文章已发布")

    // 条件更新
    db.Model(&Article{}).Where("author = ?", "张三").Update("tags", `["Go", "教程"]`)
    fmt.Println("张三的文章已添加标签")

    // 删除操作
    fmt.Println("\n=== 删除操作 ===")

    // 软删除（GORM默认行为）
    db.Where("title = ?", "Web开发").Delete(&Article{})
    fmt.Println("删除'Web开发'文章")

    // 查询包括被软删除的记录
    var allArticles []Article
    db.Unscoped().Find(&allArticles)
    fmt.Printf("包括软删除在内的文章总数: %d\n", len(allArticles))
}
```

```go
// transaction_example.go
package main

import (
    "fmt"
    "log"

    "gorm.io/driver/mysql"
    "gorm.io/gorm"
)

type Account struct {
    ID      uint    `gorm:"primaryKey"`
    Name    string  `gorm:"not null"`
    Balance float64 `gorm:"not null"`
}

type TransferRecord struct {
    ID            uint    `gorm:"primaryKey"`
    FromAccountID uint    `gorm:"not null"`
    ToAccountID   uint    `gorm:"not null"`
    Amount        float64 `gorm:"not null"`
    Status        string  `gorm:"default:'pending'"`
}

func main() {
    dsn := "user:password@tcp(127.0.0.1:3306)/testdb?charset=utf8mb4&parseTime=True&loc=Local"
    db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{})
    if err != nil {
        log.Fatal("连接数据库失败:", err)
    }

    db.AutoMigrate(&Account{}, &TransferRecord{})

    // 初始化账户
    account1 := Account{Name: "张三", Balance: 1000.00}
    account2 := Account{Name: "李四", Balance: 500.00}

    db.Create(&account1)
    db.Create(&account2)

    fmt.Printf("转账前 - 张三余额: %.2f, 李四余额: %.2f\n", account1.Balance, account2.Balance)

    // 执行转账事务
    err = transferMoney(db, account1.ID, account2.ID, 200.00)
    if err != nil {
        fmt.Printf("转账失败: %v\n", err)
    } else {
        fmt.Println("转账成功")
    }

    // 查询转账后余额
    db.First(&account1, account1.ID)
    db.First(&account2, account2.ID)
    fmt.Printf("转账后 - 张三余额: %.2f, 李四余额: %.2f\n", account1.Balance, account2.Balance)
}

func transferMoney(db *gorm.DB, fromID, toID uint, amount float64) error {
    return db.Transaction(func(tx *gorm.DB) error {
        var fromAccount, toAccount Account

        // 查询转出账户
        if err := tx.First(&fromAccount, fromID).Error; err != nil {
            return fmt.Errorf("查询转出账户失败: %v", err)
        }

        // 检查余额
        if fromAccount.Balance < amount {
            return fmt.Errorf("余额不足")
        }

        // 查询转入账户
        if err := tx.First(&toAccount, toID).Error; err != nil {
            return fmt.Errorf("查询转入账户失败: %v", err)
        }

        // 更新转出账户余额
        if err := tx.Model(&fromAccount).Update("balance", fromAccount.Balance-amount).Error; err != nil {
            return fmt.Errorf("更新转出账户失败: %v", err)
        }

        // 更新转入账户余额
        if err := tx.Model(&toAccount).Update("balance", toAccount.Balance+amount).Error; err != nil {
            return fmt.Errorf("更新转入账户失败: %v", err)
        }

        // 创建转账记录
        transferRecord := TransferRecord{
            FromAccountID: fromID,
            ToAccountID:   toID,
            Amount:        amount,
            Status:        "completed",
        }

        if err := tx.Create(&transferRecord).Error; err != nil {
            return fmt.Errorf("创建转账记录失败: %v", err)
        }

        return nil
    })
}
```

```go
// 用户模型
type User struct {

// 查询时 GORM 自动添加：WHERE ... AND deleted_at IS NULL

func complexQueryExample() {
    fmt.Println("\n=== 复杂查询示例 ===")

    // 预加载关联数据
    var orders []Order
    DB.Preload("User").Preload("Product").Find(&orders)
    fmt.Printf("查询到%d个订单\n", len(orders))
    for _, order := range orders {
        fmt.Printf("订单%d: %s 购买了 %s x %d，总价: %.2f\n",
            order.ID, order.User.Name, order.Product.Name, order.Quantity, order.Total)
    }

    // 条件查询with关联
    var userOrders []Order

    // 📌 Joins 查询：使用 JOIN 连接关联表
    // Joins("User"):    JOIN users 表（通过 Order.UserID = User.ID）
    // Joins("Product"): JOIN products 表（通过 Order.ProductID = Product.ID）
    // Where("User.name = ?", "张三"): 在关联表上添加条件
    // Find(&userOrders): 查询订单并自动填充 User 和 Product 字段
    //
    // 生成的 SQL（单条查询）：
    // SELECT orders.*, users.*, products.*
    // FROM orders
    // INNER JOIN users ON orders.user_id = users.id
    // INNER JOIN products ON orders.product_id = products.id
    // WHERE users.name = '张三'
    //
    // 与 Preload 的区别：
    // - Preload: 执行 3 条 SQL（先查订单，再查用户，再查产品）
    // - Joins:   执行 1 条 SQL（一次性 JOIN 查询）
    // - Joins 可以在 WHERE 中使用关联表的字段
    DB.Joins("User").Joins("Product").Where("User.name = ?", "张三").Find(&userOrders)

    fmt.Printf("张三的订单数量: %d\n", len(userOrders))

    // 聚合查询
    var totalSales float64

    // 📌 聚合函数查询：计算总销售额
    // Model(&Order{}):            指定查询 orders 表
    // Where("status = ?", "completed"): 条件：只统计已完成的订单
    // Select("SUM(total)"):       聚合函数：对 total 字段求和
    // Scan(&totalSales):          将结果扫描到 totalSales 变量
    //
    // 生成的 SQL：
    // SELECT SUM(total) FROM orders WHERE status = 'completed'
    //
    // 执行过程：
    // 1. 筛选 status = 'completed' 的订单
    // 2. 对这些订单的 total 字段求和
    // 3. 返回单个数值结果
    DB.Model(&Order{}).Where("status = ?", "completed").Select("SUM(total)").Scan(&totalSales)

    fmt.Printf("总销售额: %.2f\n", totalSales)

    // 分组统计
    // 定义结果结构体
    type CategorySales struct {
        Category string   // 产品分类
        Count    int64    // 订单数量
        Total    float64  // 总销售额
    }

    var categorySales []CategorySales

    // 📌 JOIN + 分组聚合查询：统计每个产品分类的销售情况
    DB.Model(&Order{}).                                                         // 1️⃣ 从 orders 表查询
        Joins("JOIN products ON orders.product_id = products.id").              // 2️⃣ JOIN products 表
        Select("products.category, COUNT(*) as count, SUM(orders.total) as total"). // 3️⃣ 选择分类、统计订单数、求和销售额
        Group("products.category").                                             // 4️⃣ 按产品分类分组
        Scan(&categorySales)                                                    // 5️⃣ 扫描结果到结构体切片

    // 生成的 SQL：
    // SELECT products.category, COUNT(*) as count, SUM(orders.total) as total
    // FROM orders
    // JOIN products ON orders.product_id = products.id
    // GROUP BY products.category
    //
    // 执行过程：
    // 1. 将 orders 表和 products 表通过 product_id 关联
    // 2. 按 products.category 分组
    // 3. 统计每组的订单数量（COUNT）和销售总额（SUM）
    // 4. 返回每个分类及其统计数据

    fmt.Println("各分类销售统计:")
    for _, cs := range categorySales {
        fmt.Printf("  %s: %d笔订单，总金额: %.2f\n", cs.Category, cs.Count, cs.Total)
    }
}

func updateExample() {
    fmt.Println("\n=== 更新示例 ===")

    // 📌 批量更新 - 使用 SQL 表达式更新价格
    // DB.Model(&Product{}):              指定操作的模型（对应 products 表）
    // Where("category = ?", "电脑配件"): 筛选条件：category 字段等于 "电脑配件"
    // Update("price", gorm.Expr("price * 0.9")): 更新 price 字段
    //
    // 关键：gorm.Expr("price * 0.9")
    //   - gorm.Expr() 告诉 GORM 这是一个 SQL 表达式，而不是固定值
    //   - "price * 0.9" 表示将当前价格乘以 0.9（即打9折，降价10%）
    //   - 如果不用 gorm.Expr，直接写 Update("price", 0.9) 会把所有价格都设为 0.9
    //
    // 生成的 SQL：
    // UPDATE products SET price = price * 0.9 WHERE category = '电脑配件'
    //
    // 执行过程：
    // 1. 筛选出所有 category = '电脑配件' 的产品
    // 2. 对每个产品，将其 price 字段更新为 price * 0.9
    // 3. 这是在数据库层面执行计算，效率高且原子性强
    DB.Model(&Product{}).Where("category = ?", "电脑配件").Update("price", gorm.Expr("price * 0.9"))
    fmt.Println("电脑配件类商品降价10%")

    // 📌 批量更新多个字段
    // DB.Model(&Order{}):              指定操作的模型（对应 orders 表）
    // Where("status = ?", "pending"):  筛选条件：status 字段等于 "pending"（待处理）
    // Updates(map[string]interface{}): 批量更新多个字段
    //
    // 关键：Update vs Updates 的区别
    //   - Update("field", value):  更新单个字段
    //   - Updates(map):            更新多个字段（使用 map 指定字段和值）
    //
    // 为什么使用 map[string]interface{}？
    //   - string: 字段名（数据库列名）
    //   - interface{}: 字段值（可以是任意类型：string、int、time.Time等）
    //   - 这样可以灵活地更新不同类型的字段
    //
    // 生成的 SQL：
    // UPDATE orders 
    // SET status = 'processing', updated_at = '2024-10-22 15:30:45'
    // WHERE status = 'pending'
    //
    // 执行过程：
    // 1. 筛选出所有 status = 'pending' 的订单
    // 2. 同时更新这些订单的 status 和 updated_at 两个字段
    // 3. 一条 SQL 完成批量多字段更新，效率高
    DB.Model(&Order{}).Where("status = ?", "pending").Updates(map[string]interface{}{
        "status": "processing",
        "updated_at": time.Now(),
    })
    fmt.Println("所有待处理订单状态更新为处理中")

    // 使用结构体更新（只会更新非零值字段）
    var product Product
    DB.First(&product, "name = ?", "笔记本电脑")
    product.Price = 5499.99
    DB.Save(&product)
    fmt.Printf("更新产品价格: %s -> %.2f\n", product.Name, product.Price)
}

func deleteExample() {
    fmt.Println("\n=== 删除示例 ===")

    // 📌 软删除（GORM 默认行为）
    // 什么是软删除？
    //   - 不是真正从数据库删除记录，而是给记录打上"已删除"标记
    //   - 通过设置 deleted_at 字段为当前时间来标记删除
    //   - 后续的普通查询会自动过滤掉这些记录
    //
    // 生成的 SQL：
    // UPDATE products SET deleted_at = '2024-10-22 15:30:45' WHERE name = '机械键盘'
    //
    // 注意：这不是 DELETE，而是 UPDATE！
    DB.Where("name = ?", "机械键盘").Delete(&Product{})
    fmt.Println("删除机械键盘（软删除）")

    // 📌 Unscoped().Find() - 查询包括软删除的记录
    // DB.Unscoped():                    忽略软删除标记，查询所有记录（包括已软删除的）
    // Where("name = ?", "机械键盘"):    筛选条件
    // Find(&deletedProducts):           查询结果
    //
    // 普通查询 vs Unscoped 查询：
    //   - DB.Find(&products)            → 只查询 deleted_at IS NULL 的记录
    //   - DB.Unscoped().Find(&products) → 查询所有记录，包括 deleted_at 不为 NULL 的
    //
    // 生成的 SQL：
    // SELECT * FROM products WHERE name = '机械键盘'
    // （注意：没有 deleted_at IS NULL 条件）
    //
    // 适用场景：
    //   - 恢复被删除的数据前先查看
    //   - 统计包括已删除在内的所有数据
    //   - 数据审计和日志追溯
    var deletedProducts []Product
    DB.Unscoped().Where("name = ?", "机械键盘").Find(&deletedProducts)
    fmt.Printf("找到%d个被软删除的机械键盘\n", len(deletedProducts))

    // 📌 Unscoped().Delete() - 永久删除（物理删除）
    // DB.Unscoped():                    忽略软删除机制
    // Where("name = ?", "机械键盘"):    筛选条件
    // Delete(&Product{}):               删除操作
    //
    // 软删除 vs 永久删除：
    //   - DB.Delete(&product)            → 软删除：UPDATE products SET deleted_at = NOW()
    //   - DB.Unscoped().Delete(&product) → 永久删除：DELETE FROM products
    //
    // 生成的 SQL：
    // DELETE FROM products WHERE name = '机械键盘'
    // （这是真正的 DELETE，数据将从数据库中彻底删除，无法恢复！）
    //
    // 适用场景：
    //   - 清理测试数据
    //   - 彻底删除敏感信息
    //   - 释放数据库存储空间
    //
    // ⚠️ 警告：永久删除的数据无法恢复，请谨慎使用！
    DB.Unscoped().Where("name = ?", "机械键盘").Delete(&Product{})
```

```go
    fmt.Println("\n原因: Student 结构体没有 DeletedAt 字段")
    fmt.Println("      GORM 不会添加软删除相关条件")

    fmt.Println("\n" + strings.Repeat("=", 70))
    fmt.Println("【对比总结】")
    fmt.Println(strings.Repeat("=", 70))

    fmt.Println("\n📊 结构体定义对比：\n")
    
    fmt.Println("✅ 支持软删除的 User:")
    fmt.Println("type User struct {")
    fmt.Println("    ID        uint")
    fmt.Println("    Name      string")
    fmt.Println("    DeletedAt gorm.DeletedAt  ← 关键！")
    fmt.Println("}")
    
    fmt.Println("\n❌ 不支持软删除的 Student:")
    fmt.Println("type Student struct {")
    fmt.Println("    ID        uint")
    fmt.Println("    Name      string")
    fmt.Println("    // 没有 DeletedAt 字段")
    fmt.Println("}")

    fmt.Println("\n📝 查询 SQL 对比：\n")
    
    fmt.Println("| 操作                | User (有 DeletedAt)                          | Student (无 DeletedAt)               |")
    fmt.Println("|---------------------|----------------------------------------------|--------------------------------------|")
    fmt.Println("| Find 查询           | WHERE ... AND deleted_at IS NULL             | WHERE ...                            |")
    fmt.Println("| Delete 删除         | UPDATE users SET deleted_at = NOW()          | DELETE FROM students                 |")
    fmt.Println("| Unscoped().Find     | WHERE ... (包括软删除的记录)                 | WHERE ... (与普通查询一样)           |")
    fmt.Println("| Unscoped().Delete   | DELETE FROM users (永久删除)                 | DELETE FROM students (与普通删除一样)|")

    fmt.Println("\n💡 最佳实践：\n")
    fmt.Println("  1️⃣  重要数据（用户、订单等）→ 使用软删除")
    fmt.Println("     type User struct {")
    fmt.Println("         DeletedAt gorm.DeletedAt `gorm:\"index\"`")
    fmt.Println("     }")
    fmt.Println("\n  2️⃣  临时数据或不重要数据 → 不使用软删除，直接物理删除")
    fmt.Println("     type Log struct {")
    fmt.Println("         // 不需要 DeletedAt 字段")
    fmt.Println("     }")
    fmt.Println("\n  3️⃣  需要查询包括已删除数据 → 使用 Unscoped()")
    fmt.Println("     DB.Unscoped().Find(&users)")
    fmt.Println("\n  4️⃣  需要永久删除 → 使用 Unscoped().Delete()")
    fmt.Println("     DB.Unscoped().Delete(&user)")
}
```
