# 微服务架构

## 微服务架构概述

微服务架构是一种将单一应用程序开发为一组小型服务的方法，每个服务运行在自己的进程中，并通过轻量级机制（通常是HTTP资源API）进行通信。

### 微服务的优势

1. **技术多样性**：每个服务可以使用不同的技术栈
2. **独立部署**：服务可以独立开发、测试、部署和扩展
3. **故障隔离**：单个服务的故障不会影响整个系统
4. **团队自治**：不同团队可以独立开发和维护不同的服务
5. **可扩展性**：可以根据需求独立扩展特定服务

### 微服务的挑战

1. **分布式系统复杂性**：网络延迟、故障处理、数据一致性
2. **运维复杂性**：需要管理多个服务实例
3. **数据管理**：分布式事务、数据一致性
4. **测试复杂性**：需要进行集成测试和端到端测试

## 微服务设计原则

### 单一职责原则

每个微服务应该只负责一个业务领域或功能。

### 高内聚低耦合

服务内部功能高度相关，服务之间依赖关系最小化。

### 无状态设计

服务不保存客户端状态，便于水平扩展。

```go
// service-registry/main.go
package main

import (
    "encoding/json"
    "net/http"
    "sync"
    "time"
    
    "github.com/gin-gonic/gin"
)

type Service struct {
    Name      string    `json:"name"`
    Address   string    `json:"address"`
    Port      int       `json:"port"`
    LastHeartbeat time.Time `json:"last_heartbeat"`
}

type ServiceRegistry struct {
    services map[string]Service
    mutex    sync.RWMutex
}

func NewServiceRegistry() *ServiceRegistry {
    return &ServiceRegistry{
        services: make(map[string]Service),
    }
}

func (sr *ServiceRegistry) Register(service Service) {
    sr.mutex.Lock()
    defer sr.mutex.Unlock()
    
    service.LastHeartbeat = time.Now()
    key := service.Name + ":" + service.Address + ":" + string(rune(service.Port))
    sr.services[key] = service
}

func (sr *ServiceRegistry) GetService(name string) (Service, bool) {
    sr.mutex.RLock()
    defer sr.mutex.RUnlock()
    
    // 返回第一个匹配的服务
    for _, service := range sr.services {
        if service.Name == name && time.Since(service.LastHeartbeat) < time.Minute*5 {
            return service, true
        }
    }
    
    return Service{}, false
}

func (sr *ServiceRegistry) GetAllServices() []Service {
    sr.mutex.RLock()
    defer sr.mutex.RUnlock()
    
    var services []Service
    for _, service := range sr.services {
        if time.Since(service.LastHeartbeat) < time.Minute*5 {
            services = append(services, service)
        }
    }
    
    return services
}

func (sr *ServiceRegistry) Heartbeat(service Service) {
    sr.mutex.Lock()
    defer sr.mutex.Unlock()
    
    key := service.Name + ":" + service.Address + ":" + string(rune(service.Port))
    if existing, exists := sr.services[key]; exists {
        existing.LastHeartbeat = time.Now()
        sr.services[key] = existing
    }
}

var registry = NewServiceRegistry()

func registerService(c *gin.Context) {
    var service Service
    if err := c.ShouldBindJSON(&service); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }
    
    registry.Register(service)
    c.JSON(http.StatusOK, gin.H{"message": "服务注册成功"})
}

func getService(c *gin.Context) {
    name := c.Param("name")
    service, exists := registry.GetService(name)
    if !exists {
        c.JSON(http.StatusNotFound, gin.H{"error": "服务不存在"})
        return
    }
    
    c.JSON(http.StatusOK, service)
}

func getAllServices(c *gin.Context) {
    services := registry.GetAllServices()
    c.JSON(http.StatusOK, services)
}

func heartbeat(c *gin.Context) {
    var service Service
    if err := c.ShouldBindJSON(&service); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }
    
    registry.Heartbeat(service)
    c.JSON(http.StatusOK, gin.H{"message": "心跳更新成功"})
}

func main() {
    r := gin.Default()
    
    r.POST("/register", registerService)
    r.GET("/service/:name", getService)
    r.GET("/services", getAllServices)
    r.POST("/heartbeat", heartbeat)
    
    r.Run(":8080")
}
```

```go
// config-manager/main.go
package main

import (
    "fmt"
    "log"
    
    "github.com/hashicorp/consul/api"
)

type ConfigManager struct {
    client *api.Client
}

func NewConfigManager() *ConfigManager {
    config := api.DefaultConfig()
    config.Address = "localhost:8500"
    
    client, err := api.NewClient(config)
    if err != nil {
        log.Fatal(err)
    }
    
    return &ConfigManager{client: client}
}

func (cm *ConfigManager) SetConfig(key, value string) error {
    kv := cm.client.KV()
    
    pair := &api.KVPair{
        Key:   key,
        Value: []byte(value),
    }
    
    _, err := kv.Put(pair, nil)
    return err
}

func (cm *ConfigManager) GetConfig(key string) (string, error) {
    kv := cm.client.KV()
    
    pair, _, err := kv.Get(key, nil)
    if err != nil {
        return "", err
    }
    
    if pair == nil {
        return "", fmt.Errorf("配置项 %s 不存在", key)
    }
    
    return string(pair.Value), nil
}

func (cm *ConfigManager) WatchConfig(key string, callback func(string)) error {
    kv := cm.client.KV()
    
    go func() {
        for {
            pair, meta, err := kv.Get(key, &api.QueryOptions{
                WaitIndex: 0,
            })
            if err != nil {
                log.Printf("监听配置失败: %v", err)
                continue
            }
            
            if pair != nil {
                callback(string(pair.Value))
            }
            
            // 更新等待索引
            meta.WaitIndex++
        }
    }()
    
    return nil
}

func main() {
    cm := NewConfigManager()
    
    // 设置配置
    cm.SetConfig("database/host", "localhost")
    cm.SetConfig("database/port", "5432")
    cm.SetConfig("database/name", "myapp")
    
    // 获取配置
    host, _ := cm.GetConfig("database/host")
    port, _ := cm.GetConfig("database/port")
    name, _ := cm.GetConfig("database/name")
    
    fmt.Printf("数据库配置: %s:%s/%s\n", host, port, name)
    
    // 监听配置变化
    cm.WatchConfig("database/host", func(value string) {
        fmt.Printf("数据库主机地址更新为: %s\n", value)
    })
    
    // 保持程序运行
    select {}
}
```

## 熔断器模式

```go
// circuit-breaker/main.go
package main

import (
    "errors"
    "fmt"
    "math/rand"
    "sync"
    "time"
)

type State int

const (
    Closed State = iota
    Open
    HalfOpen
)

type CircuitBreaker struct {
    mu           sync.Mutex
    state        State
    failureCount int
    failureThreshold int
    successCount int
    successThreshold int
    timeout      time.Duration
    lastFailure  time.Time
}

func NewCircuitBreaker(failureThreshold, successThreshold int, timeout time.Duration) *CircuitBreaker {
    return &CircuitBreaker{
        state:            Closed,
        failureThreshold: failureThreshold,
        successThreshold: successThreshold,
        timeout:          timeout,
    }
}

func (cb *CircuitBreaker) Call(fn func() error) error {
    cb.mu.Lock()
    
    switch cb.state {
    case Open:
        if time.Since(cb.lastFailure) < cb.timeout {
            cb.mu.Unlock()
            return errors.New("熔断器开启，拒绝请求")
        }
        cb.state = HalfOpen
        fmt.Println("熔断器进入半开启状态")
    case HalfOpen:
        // 在半开启状态下，只允许一个请求通过
    case Closed:
        // 正常状态下允许请求通过
    }
    
    cb.mu.Unlock()
    
    // 执行实际调用
    err := fn()
    
    cb.mu.Lock()
    defer cb.mu.Unlock()
    
    if err != nil {
        cb.handleFailure()
        return err
    }
    
    cb.handleSuccess()
    return nil
}

func (cb *CircuitBreaker) handleFailure() {
    cb.failureCount++
    cb.lastFailure = time.Now()
    
    if cb.state == HalfOpen || cb.failureCount >= cb.failureThreshold {
        cb.state = Open
        fmt.Println("熔断器开启")
    }
}

func (cb *CircuitBreaker) handleSuccess() {
    cb.failureCount = 0
    
    if cb.state == HalfOpen {
        cb.successCount++
        if cb.successCount >= cb.successThreshold {
            cb.state = Closed
            cb.successCount = 0
            fmt.Println("熔断器关闭")
        }
    }
}

// 模拟不稳定的外部服务
func unstableService() error {
    // 70%的概率失败
    if rand.Intn(100) < 70 {
        return errors.New("服务暂时不可用")
    }
    return nil
}

func main() {
    cb := NewCircuitBreaker(5, 3, time.Second*10)
    
    // 模拟连续调用
    for i := 0; i < 20; i++ {
        fmt.Printf("第%d次调用: ", i+1)
        err := cb.Call(func() error {
            return unstableService()
        })
        
        if err != nil {
            fmt.Printf("失败 - %v\n", err)
        } else {
            fmt.Println("成功")
        }
        
        time.Sleep(time.Millisecond * 500)
    }
}
```

```go
    if err := c.ShouldBindJSON(&order); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }
    
    // 验证用户是否存在
    _, err := getUserInfo(order.UserID)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "用户不存在"})
        return
    }
    
    mutex.Lock()
    order.ID = nextID
    nextID++
    order.Status = "pending"
    order.CreatedAt = time.Now()
    orders[order.ID] = order
    mutex.Unlock()
    
    c.JSON(http.StatusCreated, order)
}

    }

    if resp.StatusCode != http.StatusOK {

    }
    
    var user User
    if err := json.NewDecoder(resp.Body).Decode(&user); err != nil {
        return User{}, err
    }
    
    return user, nil
}
```

```go
// api-gateway/main.go
package main

import (
    "fmt"
    "net/http"
    "net/http/httputil"
    "net/url"
    
    "github.com/gin-gonic/gin"
    "github.com/hashicorp/consul/api"
)

type Gateway struct {
    userServiceURL    string
    orderServiceURL   string
    consulClient      *api.Client
}

func NewGateway() *Gateway {
    config := api.DefaultConfig()
    config.Address = "localhost:8500"
    
    client, err := api.NewClient(config)
    if err != nil {
        panic(err)
    }
    
    return &Gateway{
        consulClient: client,
    }
}

func (g *Gateway) discoverServices() error {
    // 发现用户服务
    userServices, _, err := g.consulClient.Health().Service("user-service", "", true, nil)
    if err != nil {
        return err
    }
    
    if len(userServices) > 0 {
        service := userServices[0].Service
        g.userServiceURL = fmt.Sprintf("http://%s:%d", service.Address, service.Port)
        fmt.Printf("发现用户服务: %s\n", g.userServiceURL)
    }
    
    // 发现订单服务
    orderServices, _, err := g.consulClient.Health().Service("order-service", "", true, nil)
    if err != nil {
        return err
    }
    
    if len(orderServices) > 0 {
        service := orderServices[0].Service
        g.orderServiceURL = fmt.Sprintf("http://%s:%d", service.Address, service.Port)
        fmt.Printf("发现订单服务: %s\n", g.orderServiceURL)
    }
    
    return nil
}

func (g *Gateway) setupRoutes(r *gin.Engine) {
    // 用户服务路由
    userProxy := g.createUserProxy()
    r.GET("/api/users/:id", gin.WrapH(userProxy))
    r.POST("/api/users", gin.WrapH(userProxy))
    
    // 订单服务路由
    orderProxy := g.createOrderProxy()
    r.GET("/api/orders/:id", gin.WrapH(orderProxy))
    r.POST("/api/orders", gin.WrapH(orderProxy))
    
    // 健康检查
    r.GET("/health", func(c *gin.Context) {
        c.JSON(http.StatusOK, gin.H{
            "status": "healthy",
            "services": map[string]string{
                "user-service":  g.userServiceURL,
                "order-service": g.orderServiceURL,
            },
        })
    })
}

func (g *Gateway) createUserProxy() *httputil.ReverseProxy {
    target, _ := url.Parse(g.userServiceURL)
    return httputil.NewSingleHostReverseProxy(target)
}

func (g *Gateway) createOrderProxy() *httputil.ReverseProxy {
    target, _ := url.Parse(g.orderServiceURL)
    return httputil.NewSingleHostReverseProxy(target)
}

func main() {
    gateway := NewGateway()
    
    // 发现服务
    if err := gateway.discoverServices(); err != nil {
        panic(err)
    }
    
    r := gin.Default()
    gateway.setupRoutes(r)
    
    fmt.Println("API网关启动在 :8080")
    r.Run(":8080")
}
```
