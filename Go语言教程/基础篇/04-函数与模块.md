# 函数与模块

## 函数定义

在Go语言中，函数使用`func`关键字定义：

```go
func functionName(parameterList) returnType {
    // 函数体
    return returnValue
}
```

### 基本函数示例

```go
package main

import "fmt"

// 无参数无返回值函数
func sayHello() {
    fmt.Println("Hello, World!")
}

// 有参数无返回值函数
func greet(name string) {
    fmt.Printf("Hello, %s!\n", name)
}

// 有参数有返回值函数
func add(a, b int) int {
    return a + b
}

// 多返回值函数
func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, fmt.Errorf("除数不能为0")
    }
    return a / b, nil
}

func main() {
    sayHello()
    greet("张三")
    
    sum := add(3, 5)
    fmt.Printf("3 + 5 = %d\n", sum)
    
    result, err := divide(10, 2)
    if err != nil {
        fmt.Println("错误:", err)
    } else {
        fmt.Printf("10 / 2 = %.2f\n", result)
    }
    
    _, err = divide(10, 0)
    if err != nil {
        fmt.Println("错误:", err)
    }
}
```

### 参数类型简写

当多个参数类型相同时，可以只在最后一个参数后指定类型：

```go
func add(a, b, c int) int {
    return a + b + c
}

func concat(str1, str2, str3 string) string {
    return str1 + str2 + str3
}
```

### 可变参数函数

使用`...`定义可变参数函数：

```go
func sum(nums ...int) int {
    total := 0
    for _, num := range nums {
        total += num
    }
    return total
}

func main() {
    fmt.Println(sum(1, 2, 3))        // 6
    fmt.Println(sum(1, 2, 3, 4, 5))  // 15
    
    numbers := []int{1, 2, 3, 4, 5}
    fmt.Println(sum(numbers...))     // 15 (展开切片)
}
```

## 匿名函数和闭包

### 匿名函数

```go
func main() {
    // 直接定义并调用匿名函数
    func() {
        fmt.Println("这是一个匿名函数")
    }()
    
    // 将匿名函数赋值给变量
    add := func(a, b int) int {
        return a + b
    }
    
    fmt.Println(add(3, 4)) // 7
    
    // 作为参数传递 高阶函数
    operate := func(a, b int, op func(int, int) int) int {
        return op(a, b) //调用传入的函数op处理参数
    }
    
    result := operate(5, 3, func(x, y int) int {
        return x * y
    })
    
    fmt.Println("5 * 3 =", result) // 15
}
```

### 闭包

闭包是一个函数值，它引用了其外部作用域中的变量：

```go
func main() {
    // 闭包示例
    counter := func() func() int {
        count := 0
        return func() int {
            count++
            return count
        }
    }
    
    c1 := counter()
    fmt.Println(c1()) // 1
    fmt.Println(c1()) // 2
    fmt.Println(c1()) // 3
    
    c2 := counter()
    fmt.Println(c2()) // 1 (独立的闭包)
    fmt.Println(c1()) // 4 (c1不受影响)
}
```

## defer语句

`defer`用于延迟执行函数，通常用于资源清理：

```go
func main() {
    defer fmt.Println("最后执行")
    fmt.Println("首先执行")
     
    // 多个defer按LIFO顺序执行
    defer fmt.Println("第一个defer")
    defer fmt.Println("第二个defer")
    defer fmt.Println("第三个defer")
    
    // defer常用于文件关闭
    file, err := os.Open("test.txt")
    if err != nil {
        fmt.Println("打开文件失败:", err)
        return
    }
    defer file.Close() // 确保文件在函数结束时关闭
    
    // 文件操作...
    fmt.Println("文件操作完成")
}
```

## init函数

每个包可以包含任意多个`init`函数，它们在包被导入时自动执行：

```go
package main

import "fmt"

var globalVar int

func init() {
    fmt.Println("第一个init函数")
    globalVar = 10
}

func init() {
    fmt.Println("第二个init函数")
    globalVar += 5
}

func main() {
    fmt.Printf("globalVar = %d\n", globalVar) // 15
}
```

## 模块和包

### 创建自定义包

创建一个`utils`包：

```go
// utils/math.go
package utils

// Add 两个整数相加
func Add(a, b int) int {
    return a + b
}

// Sub 两个整数相减
func Sub(a, b int) int {
    return a - b
}

// Max 返回两个整数中的较大值
func Max(a, b int) int {
    if a > b {
        return a
    }
    return b
}
```

```go
// utils/string.go
package utils

import "strings"

// Reverse 字符串反转
func Reverse(s string) string {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}

// IsPalindrome 判断字符串是否为回文
func IsPalindrome(s string) bool {
    s = strings.ToLower(s)
    return s == Reverse(s)
}
```

### 使用自定义包

```go
// main.go
package main

import (
    "fmt"
    "your-module-name/utils" // 替换为实际的模块名
)

func main() {
    fmt.Println(utils.Add(3, 5))        // 8
    fmt.Println(utils.Sub(10, 3))       // 7
    fmt.Println(utils.Max(5, 8))        // 8
    fmt.Println(utils.Reverse("hello")) // olleh
    fmt.Println(utils.IsPalindrome("level")) // true
}
```

## 代码示例

```go
package main

import (
    "fmt"
    "time"
)

// 计算函数执行时间的装饰器函数
func timeTracker(fn func()) func() {
    return func() {
        start := time.Now()
        fn()
        elapsed := time.Since(start)
        fmt.Printf("函数执行耗时: %v\n", elapsed)
    }
}

// 斐波那契数列（递归实现）
func fibonacci(n int) int {
    if n <= 1 {
        return n
    }
    return fibonacci(n-1) + fibonacci(n-2)
}

// 斐波那契数列（优化实现）
func fibonacciOptimized(n int) int {
    if n <= 1 {
        return n
    }
    
    a, b := 0, 1
    for i := 2; i <= n; i++ {
        a, b = b, a+b
    }
    return b
}

// 计算阶乘
func factorial(n int) int {
    if n <= 1 {
        return 1
    }
    return n * factorial(n-1)
}

func main() {
    // 使用装饰器测量函数执行时间
    timedFib := timeTracker(func() {
        result := fibonacci(30)
        fmt.Printf("fibonacci(30) = %d\n", result)
    })
    
    timedFibOpt := timeTracker(func() {
        result := fibonacciOptimized(30)
        fmt.Printf("fibonacciOptimized(30) = %d\n", result)
    })
    
    fmt.Println("递归实现:")
    timedFib()
    
    fmt.Println("优化实现:")
    timedFibOpt()
    
    // 计算阶乘
    fmt.Printf("5! = %d\n", factorial(5)) // 120
    
    // defer示例
    fmt.Println("defer示例:")
    defer fmt.Println("程序结束")
    fmt.Println("程序开始")
    
    // 多个defer
    for i := 0; i < 3; i++ {
        defer fmt.Printf("defer %d\n", i)
    }
}
```

## 练习题

1. 编写一个函数，计算数组中所有元素的平均值
2. 编写一个函数，实现字符串的凯撒密码加密
3. 编写一个函数，使用递归计算最大公约数

```go
// 练习题参考答案

package main

import (
    "fmt"
    "strings"
)

// 练习题1: 计算数组平均值
func average(nums []float64) float64 {
    if len(nums) == 0 {
        return 0
    }
    
    sum := 0.0
    for _, num := range nums {
        sum += num
    }
    return sum / float64(len(nums))
}

// 练习题2: 凯撒密码加密
func caesarCipher(text string, shift int) string {
    var result strings.Builder
    shift = shift % 26 // 确保shift在0-25范围内
    
    for _, char := range text {
        if char >= 'a' && char <= 'z' {
            // 处理小写字母
            shifted := ((int(char) - 'a' + shift) % 26) + 'a'
            result.WriteRune(rune(shifted))
        } else if char >= 'A' && char <= 'Z' {
            // 处理大写字母
            shifted := ((int(char) - 'A' + shift) % 26) + 'A'
            result.WriteRune(rune(shifted))
        } else {
            // 非字母字符保持不变
            result.WriteRune(char)
        }
    }
    
    return result.String()
}

// 练习题3: 递归计算最大公约数（欧几里得算法）
func gcd(a, b int) int {
    if b == 0 {
        return a
    }
    return gcd(b, a%b)
}

func main() {
    // 测试平均值函数
    numbers := []float64{1, 2, 3, 4, 5}
    fmt.Printf("数组%v的平均值是: %.2f\n", numbers, average(numbers))
    
    // 测试凯撒密码
    text := "Hello, World!"
    encrypted := caesarCipher(text, 3)
    fmt.Printf("原文: %s\n", text)
    fmt.Printf("加密后: %s\n", encrypted)
    decrypted := caesarCipher(encrypted, -3)
    fmt.Printf("解密后: %s\n", decrypted)
    
    // 测试最大公约数
    a, b := 48, 18
    fmt.Printf("%d和%d的最大公约数是: %d\n", a, b, gcd(a, b))
}
```
