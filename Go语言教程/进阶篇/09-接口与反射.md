# 接口与反射

## 接口进阶

### 接口的组合

```go
package main

import "fmt"

// 基础接口
type Reader interface {
    Read() string
}

type Writer interface {
    Write(data string)
}

// 组合接口
type ReadWriter interface {
    Reader
    Writer
}

// 实现组合接口的结构体
type File struct {
    Name    string
    Content string
}

func (f *File) Read() string {
    return f.Content
}

func (f *File) Write(data string) {
    f.Content = data
}

func ProcessReadWriter(rw ReadWriter) {
    rw.Write("Hello, World!")
    fmt.Printf("读取内容: %s\n", rw.Read())
}

func main() {
    file := &File{Name: "test.txt"}
    ProcessReadWriter(file)
}
```

### 空接口与类型断言

```go
package main

import "fmt"

// 空接口可以表示任何类型
func PrintTypeAndValue(v interface{}) {
    fmt.Printf("值: %v, 类型: %T\n", v, v)
}

func main() {
    // 空接口可以存储任何类型的值
    values := []interface{}{
        42,
        "Hello",
        true,
        3.14,
        []int{1, 2, 3},
    }
    
    for _, v := range values {
        PrintTypeAndValue(v)
    }
    
    // 类型断言
    var i interface{} = "Hello, World!"
    
    // 安全类型断言
    if str, ok := i.(string); ok {
        fmt.Printf("字符串值: %s, 长度: %d\n", str, len(str))
    }
    
    // 类型switch
    switch v := i.(type) {
    case string:
        fmt.Printf("字符串类型: %s\n", v)
    case int:
        fmt.Printf("整数类型: %d\n", v)
    case bool:
        fmt.Printf("布尔类型: %t\n", v)
    default:
        fmt.Printf("其他类型: %T\n", v)
    }
}
```

### 接口的比较

```go
package main

import "fmt"

type Person struct {
    Name string
    Age  int
}

func main() {
    // 接口值的比较
    var i1, i2 interface{}
    
    i1 = 42
    i2 = 42
    fmt.Printf("i1 == i2: %t\n", i1 == i2) // true
    
    i1 = Person{Name: "张三", Age: 25}
    i2 = Person{Name: "张三", Age: 25}
    fmt.Printf("i1 == i2: %t\n", i1 == i2) // true
    
    // 包含不可比较类型的接口比较会panic
    // i1 = []int{1, 2, 3}
    // i2 = []int{1, 2, 3}
    // fmt.Printf("i1 == i2: %t\n", i1 == i2) // panic: runtime error
}
```

## 反射（Reflection）

反射是在运行时检查类型和操作对象的能力。

### 常用反射方法详解

#### reflect.Type 类型相关方法

| 方法 | 作用 | 示例 |
|------|------|------|
| `reflect.TypeOf(v)` | 获取变量的类型信息，返回 `reflect.Type` | `t := reflect.TypeOf(123)` |
| `Kind()` | 获取类型的底层种类（如 Int、String、Struct 等） | `t.Kind() == reflect.Int` |
| `Name()` | 获取类型的名称（对于命名类型） | `t.Name() // "User"` |
| `String()` | 返回类型的字符串表示 | `t.String() // "main.User"` |
| `Size()` | 返回类型占用的字节数 | `t.Size() // 8` |
| `Align()` | 返回类型的对齐字节数 | `t.Align() // 4` |
| `Comparable()` | 判断类型是否可比较 | `t.Comparable() // true/false` |

#### 结构体类型方法（仅适用于 struct 类型）

| 方法 | 作用 | 示例 |
|------|------|------|
| `NumField()` | 返回结构体字段的数量 | `t.NumField() // 3` |
| `Field(i int)` | 返回第 i 个字段的 `StructField` 信息 | `field := t.Field(0)` |
| `FieldByName(name)` | 根据名称获取字段信息 | `field, ok := t.FieldByName("Name")` |
| `FieldByIndex([]int)` | 根据索引路径获取字段（支持嵌套） | `field := t.FieldByIndex([]int{0, 1})` |

#### 方法类型方法

| 方法 | 作用 | 示例 |
|------|------|------|
| `NumMethod()` | 返回类型的方法数量 | `t.NumMethod() // 2` |
| `Method(i int)` | 返回第 i 个方法的 `Method` 信息 | `method := t.Method(0)` |
| `MethodByName(name)` | 根据名称获取方法信息 | `method, ok := t.MethodByName("Add")` |

#### 函数类型方法（仅适用于 func 类型）

| 方法 | 作用 | 示例 |

#### 使用注意事项

1. **性能考虑**：反射比直接操作慢得多，应谨慎使用
2. **类型安全**：反射绕过了编译期类型检查，容易出现运行时错误
3. **可设置性**：只有可寻址的值才能被修改，需要传递指针
4. **导出字段**：只有导出的字段（首字母大写）才能被反射访问和修改
5. **panic 风险**：不当使用反射方法可能导致 panic，需要做好错误检查

#### 常见使用模式

```go
// 模式1: 检查和转换
v := reflect.ValueOf(obj)
if v.Kind() == reflect.Ptr {
    v = v.Elem()
}

// 模式2: 安全设置值
if v.CanSet() {
    v.Set(reflect.ValueOf(newValue))
}

// 模式3: 遍历结构体字段
for i := 0; i < v.NumField(); i++ {
    field := v.Field(i)
    fieldType := v.Type().Field(i)
    // 处理字段
}

// 模式4: 调用方法
method := v.MethodByName("MethodName")
if method.IsValid() {
    args := []reflect.Value{reflect.ValueOf(arg1)}
    results := method.Call(args)
}
```

### 基本反射操作

```go
package main

import (
    "fmt"
    "reflect"
)

func InspectType(v interface{}) {
    t := reflect.TypeOf(v)
    fmt.Printf("类型: %v\n", t)
    fmt.Printf("种类: %v\n", t.Kind())
    
    // 获取值
    rv := reflect.ValueOf(v)
    fmt.Printf("值: %v\n", rv)
    
    // 根据种类进行不同处理
    switch t.Kind() {
    case reflect.Int:
        fmt.Printf("整数值: %d\n", rv.Int())
    case reflect.String:
        fmt.Printf("字符串值: %s\n", rv.String())
    case reflect.Slice:
        fmt.Printf("切片长度: %d\n", rv.Len())
        for i := 0; i < rv.Len(); i++ {
            fmt.Printf("  元素%d: %v\n", i, rv.Index(i))
        }
    }
}

func main() {
    InspectType(42)
    fmt.Println("---")
    InspectType("Hello")
    fmt.Println("---")
    InspectType([]int{1, 2, 3, 4, 5})
}
```

### 结构体反射

```go
package main

import (
    "fmt"
    "reflect"
)

type User struct {
    Name    string `json:"name" validate:"required"`
    Age     int    `json:"age" validate:"min=0,max=150"`
    Email   string `json:"email" validate:"required,email"`
    private string // 私有字段
}

func InspectStruct(u interface{}) {
    v := reflect.ValueOf(u)
    t := reflect.TypeOf(u)
    
    fmt.Printf("结构体类型: %v\n", t)
    
    // 遍历所有字段
    for i := 0; i < v.NumField(); i++ {
        field := t.Field(i)
        value := v.Field(i)
        
        // 只处理可导出的字段
        if value.CanInterface() {
            fmt.Printf("字段名: %s, 类型: %v, 值: %v\n", 
                field.Name, field.Type, value.Interface())
            
            // 获取标签
            if jsonTag := field.Tag.Get("json"); jsonTag != "" {
                fmt.Printf("  JSON标签: %s\n", jsonTag)
            }
            
            if validateTag := field.Tag.Get("validate"); validateTag != "" {
                fmt.Printf("  验证标签: %s\n", validateTag)
            }
        } else {
            fmt.Printf("私有字段: %s\n", field.Name)
        }
    }
}

func SetFieldValue(u interface{}, fieldName string, value interface{}) error {
    v := reflect.ValueOf(u)
    
    // 确保u是指针且可设置
    if v.Kind() != reflect.Ptr || !v.Elem().CanSet() {
        return fmt.Errorf("无法设置字段值")
    }
    
    v = v.Elem()
    field := v.FieldByName(fieldName)
    
    if !field.IsValid() {
        return fmt.Errorf("字段 %s 不存在", fieldName)
    }
    
    if !field.CanSet() {
        return fmt.Errorf("字段 %s 不可设置", fieldName)
    }
    
    if !reflect.TypeOf(value).ConvertibleTo(field.Type()) {
        return fmt.Errorf("值类型不匹配")
    }
    
    field.Set(reflect.ValueOf(value).Convert(field.Type()))
    return nil
}

func main() {
    user := User{
        Name:    "张三",
        Age:     25,
        Email:   "zhangsan@example.com",
        private: "私有信息",
    }
    
    InspectStruct(user)
    
    fmt.Println("\n=== 修改字段值 ===")
    fmt.Printf("修改前: %+v\n", user)
    
    if err := SetFieldValue(&user, "Name", "李四"); err != nil {
        fmt.Printf("修改Name失败: %v\n", err)
    }
    
    if err := SetFieldValue(&user, "Age", 30); err != nil {
        fmt.Printf("修改Age失败: %v\n", err)
    }
    
    fmt.Printf("修改后: %+v\n", user)
}
```

### 方法反射

```go
package main

import (
    "fmt"
    "reflect"
)

type Calculator struct{}

func (c Calculator) Add(a, b int) int {
    return a + b
}

func (c Calculator) Multiply(a, b int) int {
    return a * b
}

func CallMethod(obj interface{}, methodName string, args ...interface{}) ([]reflect.Value, error) {
    v := reflect.ValueOf(obj)
    method := v.MethodByName(methodName)
    
    if !method.IsValid() {
        return nil, fmt.Errorf("方法 %s 不存在", methodName)
    }
    
    // 转换参数
    in := make([]reflect.Value, len(args))
    for i, arg := range args {
        in[i] = reflect.ValueOf(arg)
    }
    
    // 调用方法
    results := method.Call(in)
    return results, nil
}

func InspectMethods(obj interface{}) {
    t := reflect.TypeOf(obj)
    fmt.Printf("类型 %v 的方法:\n", t)
    
    for i := 0; i < t.NumMethod(); i++ {
        method := t.Method(i)
        fmt.Printf("  方法名: %s\n", method.Name)
        fmt.Printf("  类型: %v\n", method.Type)
    }
}

func main() {
    calc := Calculator{}
    
    // 检查方法
    InspectMethods(calc)
    
    // 调用方法
    if results, err := CallMethod(calc, "Add", 3, 5); err != nil {
        fmt.Printf("调用Add失败: %v\n", err)
    } else {
        fmt.Printf("Add(3, 5) = %v\n", results[0].Int())
    }
    
    if results, err := CallMethod(calc, "Multiply", 4, 6); err != nil {
        fmt.Printf("调用Multiply失败: %v\n", err)
    } else {
        fmt.Printf("Multiply(4, 6) = %v\n", results[0].Int())
    }
}
```

package main

import (

// 通用序列化器
type Serializer struct{}

func (s Serializer) ToJSON(v interface{}) ([]byte, error) {
    return json.Marshal(v)
}

func (s Serializer) FromJSON(data []byte, v interface{}) error {
    return json.Unmarshal(data, v)
}

// 通用验证器
type Validator struct{}

func (v Validator) ValidateStruct(s interface{}) error {
    rv := reflect.ValueOf(s)
    rt := reflect.TypeOf(s)
    
    // 如果是指针，获取指向的元素
    if rv.Kind() == reflect.Ptr {
        rv = rv.Elem()
        rt = rt.Elem()
    }
        
    if rv.Kind() != reflect.Struct {
        return fmt.Errorf("只能验证结构体")
    }
    
    for i := 0; i < rv.NumField(); i++ {
        field := rv.Field(i)
        fieldType := rt.Field(i)
        
        // 只验证可导出字段
        if field.CanInterface() {
            validateTag := fieldType.Tag.Get("validate")
            if validateTag != "" {
                if err := v.validateField(field, validateTag); err != nil {
                    return fmt.Errorf("字段 %s 验证失败: %v", fieldType.Name, err)
                }
            }
        }
    }
    
    return nil
}

func (v Validator) validateField(field reflect.Value, tag string) error {
    switch tag {
    case "required":
        if field.Kind() == reflect.String && field.String() == "" {
            return fmt.Errorf("字段为必填项")
        }
    case "email":
        if field.Kind() == reflect.String {
            email := field.String()
            if !isValidEmail(email) {
                return fmt.Errorf("邮箱格式不正确")
            }
        }
    }
    return nil
}

func isValidEmail(email string) bool {
    // 简单的邮箱验证
    return len(email) > 0 && 
           len(email) < 255 && 
           email[0] != '@' && 
           email[len(email)-1] != '@'
}

// 练习题1: 通用深拷贝函数
func DeepCopy(src interface{}) interface{} {
    return deepCopyValue(reflect.ValueOf(src)).Interface()
}

func deepCopyValue(v reflect.Value) reflect.Value {
    switch v.Kind() {
    case reflect.Ptr:
        if v.IsNil() {
            return reflect.Zero(v.Type())
        }
        newVal := reflect.New(v.Type().Elem())
        newVal.Elem().Set(deepCopyValue(v.Elem()))
        return newVal
    
    case reflect.Slice:
        if v.IsNil() {
            return reflect.Zero(v.Type())
        }
        newSlice := reflect.MakeSlice(v.Type(), v.Len(), v.Cap())
        for i := 0; i < v.Len(); i++ {
            newSlice.Index(i).Set(deepCopyValue(v.Index(i)))
        }
        return newSlice
    
    case reflect.Map:
        if v.IsNil() {
            return reflect.Zero(v.Type())
        }
        newMap := reflect.MakeMap(v.Type())
        for _, key := range v.MapKeys() {
            newMap.SetMapIndex(key, deepCopyValue(v.MapIndex(key)))
        }
        return newMap
    
    case reflect.Struct:
        newStruct := reflect.New(v.Type()).Elem()
        for i := 0; i < v.NumField(); i++ {
            if v.Field(i).CanSet() {
                newStruct.Field(i).Set(deepCopyValue(v.Field(i)))
            }
        }
        return newStruct
    
    default:
        return v
    }
}

// 练习题2: 简化版ORM框架
type Model struct {
    tableName string
    fields    map[string]reflect.StructField
}

func NewModel(obj interface{}) *Model {
    t := reflect.TypeOf(obj)
    if t.Kind() == reflect.Ptr {
        t = t.Elem()
    }
    
    fields := make(map[string]reflect.StructField)
    tableName := strings.ToLower(t.Name()) + "s"
    
    for i := 0; i < t.NumField(); i++ {
        field := t.Field(i)
        dbTag := field.Tag.Get("db")
        if dbTag != "" {
            fields[dbTag] = field
        } else {
            fields[strings.ToLower(field.Name)] = field
        }
    }
    
    return &Model{
        tableName: tableName,
        fields:    fields,
    }
}

func (m *Model) TableName() string {
    return m.tableName
}

func (m *Model) Fields() []string {
    var fieldNames []string
    for name := range m.fields {
        fieldNames = append(fieldNames, name)
    }
    return fieldNames
}

func (m *Model) GenerateInsertSQL(obj interface{}) string {
    v := reflect.ValueOf(obj)
    if v.Kind() == reflect.Ptr {
        v = v.Elem()
    }
    
    var placeholders []string
    var values []string
    
    for _, field := range m.Fields() {
        placeholders = append(placeholders, "?")
        fieldValue := v.FieldByName(strings.Title(field))
        if fieldValue.IsValid() {
            values = append(values, fmt.Sprintf("%v", fieldValue.Interface()))
        }
    }
    
    sql := fmt.Sprintf("INSERT INTO %s (%s) VALUES (%s)",
        m.tableName,
        strings.Join(m.Fields(), ", "),
        strings.Join(placeholders, ", "))
    
    return sql + " -- Values: " + strings.Join(values, ", ")
}

// 练习题3: 配置文件解析器
type Config struct {
    fields map[string]ConfigField
}

type ConfigField struct {
    Name        string
    Type        string
    Description string
    Default     string
    Required    bool
}

func NewConfig(obj interface{}) *Config {
    t := reflect.TypeOf(obj)
    if t.Kind() == reflect.Ptr {
        t = t.Elem()
    }
    
    fields := make(map[string]ConfigField)
    
    for i := 0; i < t.NumField(); i++ {
        field := t.Field(i)
        configTag := field.Tag.Get("config")
        
        if configTag != "" {
            parts := strings.Split(configTag, ",")
            cfgField := ConfigField{
                Name: strings.ToLower(field.Name),
                Type: field.Type.String(),
            }
            
            for _, part := range parts {
                if part == "required" {
                    cfgField.Required = true
                } else if strings.HasPrefix(part, "desc:") {
                    cfgField.Description = strings.TrimPrefix(part, "desc:")
                } else if strings.HasPrefix(part, "default:") {
                    cfgField.Default = strings.TrimPrefix(part, "default:")
                }
            }
            
            fields[cfgField.Name] = cfgField
        }
    }
    
    return &Config{fields: fields}
}

func (c *Config) GenerateDocumentation() string {
    var doc strings.Builder
    doc.WriteString("# 配置文档\n\n")
    
    for _, field := range c.fields {
        doc.WriteString(fmt.Sprintf("## %s\n", field.Name))
        doc.WriteString(fmt.Sprintf("- 类型: %s\n", field.Type))
        if field.Description != "" {
            doc.WriteString(fmt.Sprintf("- 描述: %s\n", field.Description))
        }
        if field.Default != "" {
            doc.WriteString(fmt.Sprintf("- 默认值: %s\n", field.Default))
        }
        if field.Required {
            doc.WriteString("- 必填: 是\n")
        } else {
            doc.WriteString("- 必填: 否\n")
        }
        doc.WriteString("\n")
    }
    
    return doc.String()
}
