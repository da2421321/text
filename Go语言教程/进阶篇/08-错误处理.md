# 错误处理

## Go语言的错误处理哲学

Go语言采用显式的错误处理机制，而不是使用异常。这种设计使得错误处理更加清晰和可控。

## error接口

Go语言中错误通过实现`error`接口来表示：

```go
type error interface {
    Error() string
}
```

### 基本错误处理

```go
package main

import (
    "errors"
    "fmt"
    "os"
)

// 自定义函数返回错误
func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("除数不能为零")
    }
    return a / b, nil
}

func main() {
    // 处理可能返回错误的函数
    result, err := divide(10, 2)
    if err != nil {
        fmt.Printf("错误: %v\n", err)
    } else {
        fmt.Printf("结果: %.2f\n", result)
    }
    
    // 处理错误情况
    result, err = divide(10, 0)
    if err != nil {
        fmt.Printf("错误: %v\n", err)
    } else {
        fmt.Printf("结果: %.2f\n", result)
    }
    
    // 文件操作错误处理
    file, err := os.Open("nonexistent.txt")
    if err != nil {
        fmt.Printf("打开文件失败: %v\n", err)
        // 可以根据错误类型进行不同处理
        if os.IsNotExist(err) {
            fmt.Println("文件不存在")
        }
    } else {
        defer file.Close()
        fmt.Println("文件打开成功")
    }
}
```

### 使用fmt.Errorf创建错误

```go
package main

import (
    "fmt"
    "math"
)

func calculateSqrt(x float64) (float64, error) {
    if x < 0 {
        return 0, fmt.Errorf("无法计算负数 %.2f 的平方根", x)
    }
    return math.Sqrt(x), nil

func main() {
    numbers := []float64{4, 9, -1, 16}
    
    for _, num := range numbers {
        result, err := calculateSqrt(num)
        if err != nil {
            fmt.Printf("错误: %v\n", err)
        } else {
            fmt.Printf("%.2f 的平方根是 %.2f\n", num, result)
        }
    }
}
```

## 自定义错误类型

### 实现error接口

```go
package main

import (
    "fmt"
)

// 自定义错误类型
type DivideError struct {
    Dividend float64
    Divisor  float64
}

func (e DivideError) Error() string {
    return fmt.Sprintf("无法计算 %.2f / %.2f，因为除数为零", e.Dividend, e.Divisor)
}

func divideWithErrorType(a, b float64) (float64, error) {
    if b == 0 {
        return 0, DivideError{
            Dividend: a,
            Divisor:  b,
        }
    }
    return a / b, nil
}

func main() {
    result, err := divideWithErrorType(10, 0)
    if err != nil {
        // 类型断言检查错误类型
        
        if divideErr, ok := err.(DivideError); ok {
            fmt.Printf("除法错误: %v\n", divideErr)
            fmt.Printf("被除数: %.2f, 除数: %.2f\n", divideErr.Dividend, divideErr.Divisor)
        } else {
            fmt.Printf("其他错误: %v\n", err)
        }
    } else {
        fmt.Printf("结果: %.2f\n", result)
    }
}
```

### 使用errors包

```go
package main

import (
    "errors"
    "fmt"
)

var (
    ErrInvalidAge  = errors.New("年龄无效")
    ErrInvalidName = errors.New("姓名无效")
)

type Person struct {
    Name string
    Age  int
}

func (p Person) Validate() error {
    if p.Name == "" {
        return ErrInvalidName
    }
    if p.Age < 0 || p.Age > 150 {
        return ErrInvalidAge
    }
    return nil
}

func main() {
    persons := []Person{
        {"张三", 25},
        {"", 30},
        {"李四", -5},
        {"", -1},
    }
    
    for _, person := range persons {
        if err := person.Validate(); err != nil {
            switch err {
            case ErrInvalidName:
                fmt.Printf("验证失败: 姓名无效 - %+v\n", person)
            case ErrInvalidAge:
                fmt.Printf("验证失败: 年龄无效 - %+v\n", person)
            default:
                fmt.Printf("验证失败: %v - %+v\n", err, person)
            }
        } else {
            fmt.Printf("验证成功: %+v\n", person)
        }
    }
}
```

## panic和recover

panic用于处理真正异常的情况，recover用于捕获panic。

### panic的使用

```go
package main

import "fmt"

func divideWithPanic(a, b int) int {
    if b == 0 {
        panic("除数不能为零")
    }
    return a / b
}

func main() {
    defer func() {
        if r := recover(); r != nil {
            fmt.Printf("捕获到panic: %v\n", r)
        }
    }()
    
    fmt.Println("开始计算")
    result := divideWithPanic(10, 0)
    fmt.Printf("结果: %d\n", result)
    fmt.Println("计算完成")
}
```

### recover的使用

```go
package main

import (
    "fmt"
    "log"
)

func protect(g func()) {
    defer func() {
        if err := recover(); err != nil {
            log.Printf("发生panic: %v", err)
        }
    }()
    g()
}

func riskyFunction() {
    panic("这是一个危险的操作")
}

func safeFunction() {
    fmt.Println("这是一个安全的操作")
}

func main() {
    protect(riskyFunction)
    protect(safeFunction)
    fmt.Println("程序继续执行")
}
```

## 错误包装和解包

### 使用fmt.Errorf包装错误

```go
package main

import (
    "fmt"
    "io"
    "os"
)

func processFile(filename string) error {
    file, err := os.Open(filename)
    if err != nil {
        return fmt.Errorf("打开文件失败 %s: %w", filename, err)
    }
    defer file.Close()
    
    // 模拟文件处理
    _, err = io.ReadAll(file)
    if err != nil {
        return fmt.Errorf("读取文件失败 %s: %w", filename, err)
    }
    
    return nil
}

func main() {
    err := processFile("nonexistent.txt")
    if err != nil {
        fmt.Printf("处理文件时发生错误: %v\n", err)
        
        // 检查是否包含特定类型的错误
        if errors.Is(err, os.ErrNotExist) {
            fmt.Println("文件不存在错误")
        }
    }
}
```

### 自定义错误包装

```go
package main

import (
    "fmt"
)

type AppError struct {
    Op  string
    Err error
}

func (e AppError) Error() string {
    return fmt.Sprintf("操作 %s 失败: %v", e.Op, e.Err)
}

func (e AppError) Unwrap() error {
    return e.Err
}

func readFile(filename string) error {
    // 模拟文件读取错误
    return fmt.Errorf("权限被拒绝")
}

func processConfig(filename string) error {
    if err := readFile(filename); err != nil {
        return AppError{
            Op:  "读取配置文件",
            Err: err,
        }
    }
    return nil
}

func main() {
    err := processConfig("config.json")
    if err != nil {
        fmt.Printf("错误: %v\n", err)
        
        // 使用errors.Is检查底层错误
        var appErr AppError
        if errors.As(err, &appErr) {
            fmt.Printf("应用错误，操作: %s\n", appErr.Op)
        }
    }
}
```

## 代码示例

```go
package main

import (
    "errors"
    "fmt"
    "strconv"
    "time"
)

// 数据库操作模拟
type Database struct {
    connected bool
}

func (db *Database) Connect() error {
    // 模拟连接失败
    if time.Now().Unix()%2 == 0 {
        db.connected = true
        return nil
    }
    return errors.New("数据库连接失败")
}

func (db *Database) Query(sql string) ([]map[string]interface{}, error) {
    if !db.connected {
        return nil, errors.New("数据库未连接")
    }
    
    // 模拟查询结果
    result := []map[string]interface{}{
        {"id": 1, "name": "张三", "age": 25},
        {"id": 2, "name": "李四", "age": 30},
    }
    return result, nil
}

func (db *Database) Close() error {
    if !db.connected {
        return errors.New("数据库未连接")
    }
    db.connected = false
    return nil
}

// 业务逻辑层
type UserService struct {
    db *Database
}

func (us *UserService) GetUserByID(id string) (map[string]interface{}, error) {
    // 验证参数
    userID, err := strconv.Atoi(id)
    if err != nil {
        return nil, fmt.Errorf("无效的用户ID %s: %w", id, err)
    }
    
    if userID <= 0 {
        return nil, fmt.Errorf("用户ID必须大于0，当前值: %d", userID)
    }
    
    // 查询数据库
    users, err := us.db.Query(fmt.Sprintf("SELECT * FROM users WHERE id = %d", userID))
    if err != nil {
        return nil, fmt.Errorf("查询用户失败: %w", err)
    }
    
    if len(users) == 0 {
        return nil, fmt.Errorf("未找到ID为%s的用户", id)
    }
    
    return users[0], nil
}

func main() {
    db := &Database{}
    userService := &UserService{db: db}
    
    // 尝试连接数据库
    if err := db.Connect(); err != nil {
        fmt.Printf("数据库连接失败: %v\n", err)
        return
    }
    defer db.Close()
    
    // 测试获取用户
    testIDs := []string{"1", "2", "abc", "-1", "999"}
    
    for _, id := range testIDs {
        fmt.Printf("\n尝试获取用户ID: %s\n", id)
        user, err := userService.GetUserByID(id)
        if err != nil {
            fmt.Printf("获取用户失败: %v\n", err)
            
            // 根据错误类型进行不同处理
            var invalidIDError error
            if errors.As(err, &invalidIDError) {
                fmt.Println("  -> 这是一个ID验证错误")
            }
        } else {
            fmt.Printf("获取用户成功: %+v\n", user)
        }
    }
}
```

1. 创建一个计算器程序，对除零等错误进行适当处理
2. 实现一个文件处理系统，包含打开、读取、写入等操作的错误处理
3. 设计一个用户认证系统，包含登录、注册等功能的错误处理

```go
// 练习题参考答案

package main

import (
    "errors"
    "fmt"
    "strconv"
    "strings"
)

// 练习题1: 计算器程序
type Calculator struct{}

func (c Calculator) Add(a, b float64) float64 {
    return a + b
}

func (c Calculator) Subtract(a, b float64) float64 {
    return a - b
}

func (c Calculator) Multiply(a, b float64) float64 {
    return a * b
}

func (c Calculator) Divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("除数不能为零")
    }
    return a / b, nil
}

func (c Calculator) Power(base, exp float64) (float64, error) {
    if base == 0 && exp < 0 {
        return 0, errors.New("0的负数次幂未定义")
    }
    // 简化实现，实际应使用math.Pow
    result := 1.0
    for i := 0; i < int(exp); i++ {
        result *= base
    }
    return result, nil
}

func (c Calculator) Calculate(expression string) (float64, error) {
    parts := strings.Fields(expression)
    if len(parts) != 3 {
        return 0, errors.New("表达式格式错误，应为: 数字 运算符 数字")
    }
    
    a, err := strconv.ParseFloat(parts[0], 64)
    if err != nil {
        return 0, fmt.Errorf("无效的第一个数字: %w", err)
    }
    
    operator := parts[1]
    
    b, err := strconv.ParseFloat(parts[2], 64)
    if err != nil {
        return 0, fmt.Errorf("无效的第二个数字: %w", err)
    }
    
    switch operator {
    case "+":
        return c.Add(a, b), nil
    case "-":
        return c.Subtract(a, b), nil
    case "*":
        return c.Multiply(a, b), nil
    case "/":
        return c.Divide(a, b)
    case "^":
        return c.Power(a, b)
    default:
        return 0, fmt.Errorf("不支持的运算符: %s", operator)
    }
}

// 练习题2: 文件处理系统
type FileError struct {
    Operation string
    Filename  string
    Err       error
}

func (e FileError) Error() string {
    return fmt.Sprintf("文件操作 %s 失败 (%s): %v", e.Operation, e.Filename, e.Err)
}

func (e FileError) Unwrap() error {
    return e.Err
}

type FileManager struct {
    files map[string]string
}

func NewFileManager() *FileManager {
    return &FileManager{
        files: make(map[string]string),
    }
}

func (fm *FileManager) CreateFile(filename, content string) error {
    if filename == "" {
        return FileError{
            Operation: "创建文件",
            Filename:  filename,
            Err:       errors.New("文件名不能为空"),
        }
    }
    
    fm.files[filename] = content
    return nil
}

func (fm *FileManager) ReadFile(filename string) (string, error) {
    if filename == "" {
        return "", FileError{
            Operation: "读取文件",
            Filename:  filename,
            Err:       errors.New("文件名不能为空"),
        }
    }
    
    content, exists := fm.files[filename]
    if !exists {
        return "", FileError{
            Operation: "读取文件",
            Filename:  filename,
            Err:       errors.New("文件不存在"),
        }
    }
    
    return content, nil
}

func (fm *FileManager) WriteFile(filename, content string) error {
    if filename == "" {
        return FileError{
            Operation: "写入文件",
            Filename:  filename,
            Err:       errors.New("文件名不能为空"),
        }
    }
    
    if _, exists := fm.files[filename]; !exists {
        return FileError{
            Operation: "写入文件",
            Filename:  filename,
            Err:       errors.New("文件不存在"),
        }
    }
    
    fm.files[filename] = content
    return nil
}

func (fm *FileManager) DeleteFile(filename string) error {
    if filename == "" {
        return FileError{
            Operation: "删除文件",
            Filename:  filename,
            Err:       errors.New("文件名不能为空"),
        }
    }
    
    if _, exists := fm.files[filename]; !exists {
        return FileError{
            Operation: "删除文件",
            Filename:  filename,
            Err:       errors.New("文件不存在"),
        }
    }
    
    delete(fm.files, filename)
    return nil
}

// 练习题3: 用户认证系统
type AuthError struct {
    Operation string
    Username  string
    Err       error
}

func (e AuthError) Error() string {
    return fmt.Sprintf("认证操作 %s 失败 (用户: %s): %v", e.Operation, e.Username, e.Err)
}

func (e AuthError) Unwrap() error {
    return e.Err
}

type User struct {
    Username string
    Password string
    Email    string
}

type AuthService struct {
    users map[string]User
}

func NewAuthService() *AuthService {
    return &AuthService{
        users: make(map[string]User),
    }
}

func (as *AuthService) Register(username, password, email string) error {
    if username == "" {
        return AuthError{
            Operation: "注册",
            Username:  username,
            Err:       errors.New("用户名不能为空"),
        }
    }
    
    if password == "" {
        return AuthError{
            Operation: "注册",
            Username:  username,
            Err:       errors.New("密码不能为空"),
        }
    }
    
    if email == "" {
        return AuthError{
            Operation: "注册",
            Username:  username,
            Err:       errors.New("邮箱不能为空"),
        }
    }
    
    if _, exists := as.users[username]; exists {
        return AuthError{
            Operation: "注册",
            Username:  username,
            Err:       errors.New("用户名已存在"),
        }
    }
    
    as.users[username] = User{
        Username: username,
        Password: password,
        Email:    email,
    }
    
    return nil
}

func (as *AuthService) Login(username, password string) error {
    if username == "" {
        return AuthError{
            Operation: "登录",
            Username:  username,
            Err:       errors.New("用户名不能为空"),
        }
    }
    
    if password == "" {
        return AuthError{
            Operation: "登录",
            Username:  username,
            Err:       errors.New("密码不能为空"),
        }
    }
    
    user, exists := as.users[username]
    if !exists {
        return AuthError{
            Operation: "登录",
            Username:  username,
            Err:       errors.New("用户不存在"),
        }
    }
    
    if user.Password != password {
        return AuthError{
            Operation: "登录",
            Username:  username,
            Err:       errors.New("密码错误"),
        }
    }
    
    return nil
}

func (as *AuthService) GetUser(username string) (User, error) {
    user, exists := as.users[username]
    if !exists {
        return User{}, AuthError{
            Operation: "获取用户信息",
            Username:  username,
            Err:       errors.New("用户不存在"),
        }
    }
    return user, nil
}

    if content, err := fm.ReadFile("test.txt"); err != nil {
```
