# 面向对象编程

## Go语言的面向对象特性

Go语言没有传统意义上的类（class）概念，而是通过结构体（struct）和方法（method）来实现面向对象编程。

## 结构体与方法

### 定义结构体和方法

```go
package main

import "fmt"

// 定义一个矩形结构体
type Rectangle struct {
    Width, Height float64
}

// 为Rectangle定义方法
func (r Rectangle) Area() float64 {
    return r.Width * r.Height
}

func (r Rectangle) Perimeter() float64 {
    return 2 * (r.Width + r.Height)
}

// 使用指针接收者的方法
func (r *Rectangle) Scale(factor float64) {
    r.Width *= factor
    r.Height *= factor
}

func main() {
    rect := Rectangle{Width: 10, Height: 5}
    
    fmt.Printf("矩形: 宽=%.2f, 高=%.2f\n", rect.Width, rect.Height)
    fmt.Printf("面积: %.2f\n", rect.Area())
    fmt.Printf("周长: %.2f\n", rect.Perimeter())
    
    // 使用指针接收者的方法
    rect.Scale(2)
    fmt.Printf("缩放后 - 宽=%.2f, 高=%.2f\n", rect.Width, rect.Height)
    fmt.Printf("缩放后面积: %.2f\n", rect.Area())
}
```

**运行结果：**
```
矩形: 宽=10.00, 高=5.00
面积: 50.00
周长: 30.00
缩放后 - 宽=20.00, 高=10.00
缩放后面积: 200.00
```

### 值接收者 vs 指针接收者

```go
package main

import "fmt"

type Point struct {
    X, Y int
}

// 值接收者 - 不会修改原始结构体
func (p Point) MoveByValue(dx, dy int) Point {
    return Point{X: p.X + dx, Y: p.Y + dy}
}

// 指针接收者 - 会修改原始结构体
func (p *Point) MoveByPointer(dx, dy int) {
    p.X += dx
    p.Y += dy
}

func main() {
    p1 := Point{1, 2}
    fmt.Printf("原始点: %+v\n", p1)
    
    // 值接收者
    p2 := p1.MoveByValue(3, 4)
    fmt.Printf("移动后(值接收者): %+v\n", p2)
    fmt.Printf("原始点: %+v\n", p1) // 原始点未改变
    
    // 指针接收者
    p1.MoveByPointer(5, 6)
    fmt.Printf("移动后(指针接收者): %+v\n", p1) // 原始点已改变
}
```

**运行结果：**
```
原始点: {X:1 Y:2}
移动后(值接收者): {X:4 Y:6}
原始点: {X:1 Y:2}
移动后(指针接收者): {X:6 Y:8}
```

## 接口（Interface）

接口是Go语言实现多态的关键机制。

### 定义和实现接口

```go
package main

import (
    "fmt"
    "math"
)

// 定义几何形状接口
type Shape interface {
    Area() float64
    Perimeter() float64
}

// 矩形结构体
type Rectangle struct {
    Width, Height float64
}

func (r Rectangle) Area() float64 {
    return r.Width * r.Height
}

func (r Rectangle) Perimeter() float64 {
    return 2 * (r.Width + r.Height)
}

// 圆形结构体
type Circle struct {
    Radius float64
}

func (c Circle) Area() float64 {
    return math.Pi * c.Radius * c.Radius
}

func (c Circle) Perimeter() float64 {
    return 2 * math.Pi * c.Radius
}

// 计算形状信息的函数
func PrintShapeInfo(s Shape) {
    fmt.Printf("面积: %.2f, 周长: %.2f\n", s.Area(), s.Perimeter())
}

func main() {
    rect := Rectangle{Width: 10, Height: 5}
    circle := Circle{Radius: 3}
    
    fmt.Println("矩形信息:")
    PrintShapeInfo(rect)
    
    fmt.Println("圆形信息:")
    PrintShapeInfo(circle)
}
```

**运行结果：**
```
矩形信息:
面积: 50.00, 周长: 30.00
圆形信息:
面积: 28.27, 周长: 18.85
```

### 空接口和类型断言

```go
package main

import "fmt"

// 空接口可以表示任何类型
func PrintValue(v interface{}) {
    fmt.Printf("值: %v, 类型: %T\n", v, v)
}

func main() {
    // 空接口可以存储任何类型的值
    var i interface{}
    i = 42
    PrintValue(i)
    
    i = "Hello"
    PrintValue(i)
    
    i = true
    PrintValue(i)
    
    // 类型断言
    var val interface{} = "Hello, World!"
    
    // 方式1: 直接断言（如果类型不匹配会panic）
    str := val.(string)
    fmt.Printf("字符串值: %s\n", str)
    
    // 方式2: 安全断言（返回两个值）
    if str, ok := val.(string); ok {
        fmt.Printf("安全断言 - 字符串值: %s\n", str)
    }
    
    // 断言为其他类型
    if num, ok := val.(int); ok {
        fmt.Printf("数字值: %d\n", num)
    } else {
        fmt.Println("不是数字类型")
    }

    // 使用switch进行类型判断
    switch v := val.(type) {
    case string:
        fmt.Printf("字符串类型: %s\n", v)
    case int:
        fmt.Printf("整数类型: %d\n", v)
    case bool:
        fmt.Printf("布尔类型: %t\n", v)
    default:
        fmt.Printf("未知类型: %T\n", v)
    }
}
```

**运行结果：**
```
值: 42, 类型: int
值: Hello, 类型: string
值: true, 类型: bool
字符串值: Hello, World!
安全断言 - 字符串值: Hello, World!
不是数字类型
字符串类型: Hello, World!
```

## 组合与嵌入

Go语言通过组合而非继承来实现代码复用。

### 结构体嵌入

```go
package main

import "fmt"

// 基础结构体
type Person struct {
    Name string
    Age  int
}

func (p Person) Greet() {
    fmt.Printf("Hello, I'm %s\n", p.Name)
}

// 通过嵌入实现"继承"
type Student struct {
    Person    // 嵌入Person结构体
    School string
    Grade  int
}

func (s Student) Study() {
    fmt.Printf("%s is studying at %s\n", s.Name, s.School)
}

type Teacher struct {
    Person
    Subject string
    Salary  float64
}

func (t Teacher) Teach() {
    fmt.Printf("%s is teaching %s\n", t.Name, t.Subject)
}

func main() {
    student := Student{
        Person: Person{Name: "张三", Age: 20},
        School: "清华大学",
        Grade:  3,
    }
    
    teacher := Teacher{
        Person:  Person{Name: "李老师", Age: 35},
        Subject: "数学",
        Salary:  8000,
    }
    
    // 可以直接访问嵌入结构体的字段和方法
    fmt.Printf("学生: %s, 年龄: %d, 学校: %s\n", 
        student.Name, student.Age, student.School)
    student.Greet()  // 调用嵌入结构体的方法
    student.Study()
    
    fmt.Printf("老师: %s, 年龄: %d, 科目: %s\n", 
        teacher.Name, teacher.Age, teacher.Subject)
    teacher.Greet()
    teacher.Teach()
}
```

**运行结果：**
```
学生: 张三, 年龄: 20, 学校: 清华大学
Hello, I'm 张三
张三 is studying at 清华大学
老师: 李老师, 年龄: 35, 科目: 数学
Hello, I'm 李老师
李老师 is teaching 数学
```

### 接口嵌入

```go
package main

import "fmt"

// 定义基础接口
type Reader interface {
    Read() string
}

type Writer interface {
    Write(data string)
}

// 通过嵌入组合接口
type ReadWriter interface {
    Reader
    Writer
}

// 实现ReadWriter接口的结构体
type File struct {
    Name    string
    Content string
}

func (f *File) Read() string {
    return f.Content
}

func (f *File) Write(data string) {
    f.Content = data
}

func main() {
    file := &File{Name: "test.txt"}
    
    // 实现了ReadWriter接口
    var rw ReadWriter = file
    
    rw.Write("Hello, World!")
    fmt.Printf("读取内容: %s\n", rw.Read())
    
    // 也可以作为Reader或Writer使用
    var reader Reader = file
    var writer Writer = file
    
    writer.Write("Go语言")
    fmt.Printf("Reader读取: %s\n", reader.Read())
}
```

**运行结果：**
```
读取内容: Hello, World!
Reader读取: Go语言
```

## 代码示例

```go
package main

import (
    "fmt"
    "math"
)

// 定义动物接口
type Animal interface {
    Speak() string
    Move() string
}

// 定义哺乳动物接口，嵌入Animal接口
type Mammal interface {
    Animal
    FeedMilk() string
}

// 基础动物结构体
type BaseAnimal struct {
    Name string
}

func (b BaseAnimal) Speak() string {
    return fmt.Sprintf("%s发出声音", b.Name)
}

func (b BaseAnimal) Move() string {
    return fmt.Sprintf("%s在移动", b.Name)
}

// 狗结构体
type Dog struct {
    BaseAnimal
    Breed string
}

func (d Dog) Speak() string {
    return fmt.Sprintf("%s汪汪叫", d.Name)
}

func (d Dog) FeedMilk() string {
    return fmt.Sprintf("%s在哺乳", d.Name)
}

// 鸟结构体
type Bird struct {
    BaseAnimal
    CanFly bool
}

func (b Bird) Speak() string {
    return fmt.Sprintf("%s啾啾叫", b.Name)
}

func (b Bird) Move() string {
    if b.CanFly {
        return fmt.Sprintf("%s在飞翔", b.Name)
    }
    return fmt.Sprintf("%s在行走", b.Name)
}

// 计算动物信息的函数
func DescribeAnimal(a Animal) {
    fmt.Println(a.Speak())
    fmt.Println(a.Move())
}

func DescribeMammal(m Mammal) {
    DescribeAnimal(m)
    fmt.Println(m.FeedMilk())
}

func main() {
    dog := Dog{
        BaseAnimal: BaseAnimal{Name: "旺财"},
        Breed:      "金毛",
    }
    
    bird := Bird{
        BaseAnimal: BaseAnimal{Name: "小鸟"},
        CanFly:     true,
    }
    
    fmt.Println("=== 狗的信息 ===")
    DescribeAnimal(dog)
    // 由于Dog实现了Mammal接口，也可以这样调用
    DescribeMammal(dog)
    
    fmt.Println("\n=== 鸟的信息 ===")
    DescribeAnimal(bird)
    
    // 使用接口切片
    animals := []Animal{dog, bird}
    
    fmt.Println("\n=== 动物列表 ===")
    for _, animal := range animals {
        fmt.Println("---")
        DescribeAnimal(animal)
    }
}
```

**运行结果：**
```
=== 狗的信息 ===
旺财汪汪叫
旺财在移动
旺财汪汪叫
旺财在移动
旺财在哺乳

=== 鸟的信息 ===
小鸟啾啾叫
小鸟在飞翔

=== 动物列表 ===
---
旺财汪汪叫
旺财在移动
---
小鸟啾啾叫
小鸟在飞翔
```

## 练习题

1. 创建一个几何图形系统，包含矩形、圆形和三角形，都实现面积和周长计算
2. 实现一个简单的支付系统，支持信用卡、支付宝和微信支付
3. 创建一个员工管理系统，包含不同类型的员工（全职、兼职、实习生）

```go
// 练习题参考答案

package main

import (
    "fmt"
    "math"
)

// 练习题1: 几何图形系统
type Geometry interface {
    Area() float64
    Perimeter() float64
}

type Rectangle struct {
    Width, Height float64
}

func (r Rectangle) Area() float64 {
    return r.Width * r.Height
}

func (r Rectangle) Perimeter() float64 {
    return 2 * (r.Width + r.Height)
}

type Circle struct {
    Radius float64
}

func (c Circle) Area() float64 {
    return math.Pi * c.Radius * c.Radius
}

func (c Circle) Perimeter() float64 {
    return 2 * math.Pi * c.Radius
}

type Triangle struct {
    SideA, SideB, SideC float64
}

func (t Triangle) Area() float64 {
    // 使用海伦公式计算三角形面积
    s := (t.SideA + t.SideB + t.SideC) / 2
    return math.Sqrt(s * (s - t.SideA) * (s - t.SideB) * (s - t.SideC))
}

func (t Triangle) Perimeter() float64 {
    return t.SideA + t.SideB + t.SideC
}

// 练习题2: 支付系统
type Payment interface {
    Pay(amount float64) string
}

type CreditCard struct {
    CardNumber string
}

func (c CreditCard) Pay(amount float64) string {
    return fmt.Sprintf("使用信用卡(%s)支付了%.2f元", c.CardNumber, amount)
}

type Alipay struct {
    Account string
}

func (a Alipay) Pay(amount float64) string {
    return fmt.Sprintf("使用支付宝(%s)支付了%.2f元", a.Account, amount)
}

type WeChatPay struct {
    Account string
}

func (w WeChatPay) Pay(amount float64) string {
    return fmt.Sprintf("使用微信支付(%s)支付了%.2f元", w.Account, amount)
}

// 练习题3: 员工管理系统
type Employee interface {
    GetInfo() string
    CalculateSalary() float64
}

type FullTimeEmployee struct {
    Name        string
    MonthlySalary float64
}

func (f FullTimeEmployee) GetInfo() string {
    return fmt.Sprintf("全职员工: %s", f.Name)
}

func (f FullTimeEmployee) CalculateSalary() float64 {
    return f.MonthlySalary
}

type PartTimeEmployee struct {
    Name       string
    HourlyRate float64
    Hours      float64
}

func (p PartTimeEmployee) GetInfo() string {
    return fmt.Sprintf("兼职员工: %s", p.Name)
}

func (p PartTimeEmployee) CalculateSalary() float64 {
    return p.HourlyRate * p.Hours
}

type Intern struct {
    Name      string
    Stipend   float64
}

func (i Intern) GetInfo() string {
    return fmt.Sprintf("实习生: %s", i.Name)
}

func (i Intern) CalculateSalary() float64 {
    return i.Stipend
}

func main() {
    // 练习题1测试
    fmt.Println("=== 几何图形系统 ===")
    shapes := []Geometry{
        Rectangle{Width: 10, Height: 5},
        Circle{Radius: 3},
        Triangle{SideA: 3, SideB: 4, SideC: 5},
    }
    
    for _, shape := range shapes {
        fmt.Printf("面积: %.2f, 周长: %.2f\n", shape.Area(), shape.Perimeter())
    }
    
    // 练习题2测试
    fmt.Println("\n=== 支付系统 ===")
    payments := []Payment{
        CreditCard{CardNumber: "1234****5678"},
        Alipay{Account: "alipay@example.com"},
        WeChatPay{Account: "wechat@example.com"},
    }
    
    for _, payment := range payments {
        fmt.Println(payment.Pay(100.50))
    }
    
    // 练习题3测试
    fmt.Println("\n=== 员工管理系统 ===")
    employees := []Employee{
        FullTimeEmployee{Name: "张三", MonthlySalary: 15000},
        PartTimeEmployee{Name: "李四", HourlyRate: 50, Hours: 80},
        Intern{Name: "王五", Stipend: 3000},
    }
    
    for _, employee := range employees {
        fmt.Printf("%s, 月薪: %.2f元\n", employee.GetInfo(), employee.CalculateSalary())
    }
}
```
